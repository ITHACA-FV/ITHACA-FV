410 // Method to compute the lifting function for temperature
411 void unsteadyBB::liftSolveT()
412 {
413  for (label k = 0; k < inletIndexT.rows(); k++)
414  {
415  Time& runTime = _runTime();
416  fvMesh& mesh = _mesh();
417  volScalarField& T = _T();
418  volVectorField& U = _U();
419  surfaceScalarField& phi = _phi();
420  phi = linearInterpolate(U) & mesh.Sf();
421  simpleControl simple(mesh);
422  IOMRFZoneList& MRF = _MRF();
423  singlePhaseTransportModel& laminarTransport = _laminarTransport();
424  volScalarField& nut = _nut();
425  volScalarField& alphat = _alphat();
426  dimensionedScalar& nu = _nu();
427  dimensionedScalar& Pr = _Pr();
428  dimensionedScalar& Prt = _Prt();
429  label BCind = inletIndexT(k, 0);
430  volScalarField Tlift("Tlift" + name(k), T);
431  instantList Times = runTime.times();
432  runTime.setTime(Times[1], 1);
433  Info << "Solving a lifting Problem" << endl;
434  scalar t1 = 1;
435  scalar t0 = 0;
436  alphat = turbulence->nut() / Prt;
437  alphat.correctBoundaryConditions();
438  volScalarField alphaEff("alphaEff", turbulence->nu() / Pr + alphat);
439 
440  for (label j = 0; j < T.boundaryField().size(); j++)
441  {
442  if (j == BCind)
443  {
444  assignBC(Tlift, j, t1);
445  assignIF(Tlift, t0);
446  }
447  else if (T.boundaryField()[BCind].type() == "fixedValue")
448  {
449  assignBC(Tlift, j, t0);
450  assignIF(Tlift, t0);
451  }
452  else
453  {
454  }
455  }
456 
457  while (simple.correctNonOrthogonal())
458  {
459  fvScalarMatrix TEqn
460  (
461  fvm::div(phi, Tlift)
462  - fvm::laplacian(alphaEff, Tlift)
463  );
464  TEqn.solve();
465  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
466  << " ClockTime = " << runTime.elapsedClockTime() << " s"
467  << nl << endl;
468  }
469 
470  Tlift.write();
471  liftfieldT.append(Tlift);
472  }
473 }

596 // * * * * * * * * * * * * * * Momentum Eq. Methods * * * * * * * * * * * * * //
597 Eigen::MatrixXd unsteadyBB::pressure_gradient_term(label NUmodes,
598  label NPrghmodes,
599  label NSUPmodes)
600 {
601  label K1size = NUmodes + NSUPmodes + liftfield.size();
602  label K2size = NPrghmodes;
603  Eigen::MatrixXd K_matrix(K1size, K2size);
604  // Create PTRLIST with lift, velocities and supremizers
605  PtrList<volVectorField> Together(0);
606  PtrList<volScalarField> TogetherT(0);
607  dimensionedVector g = _g();
608 
609  if (liftfield.size() != 0)
610  {
611  for (label k = 0; k < liftfield.size(); k++)
612  {
613  Together.append(liftfield[k]);
614  }
615  }
616 
617  if (NUmodes != 0)
618  {
619  for (label k = 0; k < NUmodes; k++)
620  {
621  Together.append(Umodes[k]);
622  }
623  }
624 
625  if (NSUPmodes != 0)
626  {
627  for (label k = 0; k < NSUPmodes; k++)
628  {
629  Together.append(supmodes[k]);
630  }
631  }
632 
633  // Project everything
634  for (label i = 0; i < K1size; i++)
635  {
636  for (label j = 0; j < K2size; j++)
637  {
638  K_matrix(i, j) = fvc::domainIntegrate(Together[i] &
639  fvc::reconstruct(fvc::snGrad(Prghmodes[j]) *
640  Prghmodes[j].mesh().magSf())).value();
641  }
642  }
643 
644  // Export the matrix
645  ITHACAstream::SaveDenseMatrix(K_matrix, "./ITHACAoutput/Matrices/",
646  "K_" + name(liftfield.size()) + "_" + name(NUmodes)
647  + "_" + name(NSUPmodes) + "_" + name(NPrghmodes));
648  return K_matrix;
649 }

935 // Method to compute the lifting function for velocity
936 void unsteadyBB::liftSolve()
937 {
938  for (label k = 0; k < inletIndex.rows(); k++)
939  {
940  Time& runTime = _runTime();
941  surfaceScalarField& phi = _phi();
942  fvMesh& mesh = _mesh();
943  volScalarField& p = _p();
944  volVectorField& U = _U();
945  volScalarField& p_rgh = _p_rgh();
946  volScalarField& UliftBC = _UliftBC();
947  volScalarField& T = _T();
948  dimensionedScalar& nu = _nu();
949  IOMRFZoneList& MRF = _MRF();
950  label BCind = inletIndex(k, 0);
951  volVectorField Ulift("Ulift" + name(k), U);
952  instantList Times = runTime.times();
953  runTime.setTime(Times[1], 1);
954  pisoControl potentialFlow(mesh, "potentialFlow");
955  Info << "Solving a lifting Problem" << endl;
956  Vector<double> v1(0, 0, 0);
957  v1[inletIndex(k, 1)] = 1;
958  Vector<double> v0(0, 0, 0);
959 
960  for (label j = 0; j < U.boundaryField().size(); j++)
961  {
962  if (j == BCind)
963  {
964  assignBC(Ulift, j, v1);
965  }
966  else if (U.boundaryField()[BCind].type() == "fixedValue")
967  {
968  assignBC(Ulift, j, v0);
969  }
970  else
971  {
972  }
973 
974  assignIF(Ulift, v0);
975  phi = linearInterpolate(Ulift) & mesh.Sf();
976  }
977 
978  Info << "Constructing velocity potential field Phi\n" << endl;
979  volScalarField Phi
980  (
981  IOobject
982  (
983  "Phi",
984  runTime.timeName(),
985  mesh,
986  IOobject::READ_IF_PRESENT,
987  IOobject::NO_WRITE
988  ),
989  mesh,
990  dimensionedScalar("Phi", dimLength * dimVelocity, 0),
991  UliftBC.boundaryField().types()
992  );
993  label PhiRefCell = 0;
994  scalar PhiRefValue = 0.0;
995  setRefCell
996  (
997  Phi,
998  potentialFlow.dict(),
999  PhiRefCell,
1000  PhiRefValue
1001  );
1002  mesh.setFluxRequired(Phi.name());
1003  runTime.functionObjects().start();
1004  MRF.makeRelative(phi);
1005  adjustPhi(phi, Ulift, UliftBC);
1006 
1007  while (potentialFlow.correctNonOrthogonal())
1008  {
1009  fvScalarMatrix PhiEqn
1010  (
1011  fvm::laplacian(dimensionedScalar("1", dimless, 1), Phi)
1012  ==
1013  fvc::div(phi)
1014  );
1015  PhiEqn.setReference(PhiRefCell, PhiRefValue);
1016  PhiEqn.solve();
1017 
1018  if (potentialFlow.finalNonOrthogonalIter())
1019  {
1020  phi -= PhiEqn.flux();
1021  }
1022  }
1023 
1024  MRF.makeAbsolute(phi);
1025  Info << "Continuity error = "
1026  << mag(fvc::div(phi))().weightedAverage(mesh.V()).value()
1027  << endl;
1028  Ulift = fvc::reconstruct(phi);
1029  Ulift.correctBoundaryConditions();
1030  Info << "Interpolated velocity error = "
1031  << (sqrt(sum(sqr((fvc::interpolate(U) & mesh.Sf()) - phi)))
1032  / sum(mesh.magSf())).value()
1033  << endl;
1034  Ulift.write();
1035  liftfield.append(Ulift);
1036  }
1037 }
