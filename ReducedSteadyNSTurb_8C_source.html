68 int newtonSteadyNSTurb::operator()(const Eigen::VectorXd& x,
69  Eigen::VectorXd& fvec) const
70 {
71  Eigen::VectorXd aTmp(Nphi_u);
72  Eigen::VectorXd bTmp(Nphi_p);
73  aTmp = x.head(Nphi_u);
74  bTmp = x.tail(Nphi_p);
75  // Convective term
76  Eigen::MatrixXd cc(1, 1);
77  // Mom Term
78  Eigen::VectorXd m1 = problem->bTotalMatrix * aTmp * nu;
79  // Gradient of pressure
80  Eigen::VectorXd m2 = problem->K_matrix * bTmp;
81  // Pressure Term
82  Eigen::VectorXd m3 = problem->P_matrix * aTmp;
83  // Penalty term
84  Eigen::MatrixXd penaltyU = Eigen::MatrixXd::Zero(Nphi_u, N_BC);
85 
86  // Term for penalty method
87  if (problem->bcMethod == "penalty")
88  {
89  for (label l = 0; l < N_BC; l++)
90  {
91  penaltyU.col(l) = bc(l) * problem->bcVelVec[l] - problem->bcVelMat[l] *
92  aTmp;
93  }
94  }
95 
96  for (label i = 0; i < Nphi_u; i++)
97  {
98  cc = aTmp.transpose() * Eigen::SliceFromTensor(problem->C_tensor, 0,
99  i) * aTmp - gNut.transpose() *
100  Eigen::SliceFromTensor(problem->cTotalTensor, 0, i) * aTmp;
101  fvec(i) = m1(i) - cc(0, 0) - m2(i);
102 
103  if (problem->bcMethod == "penalty")
104  {
105  fvec(i) += ((penaltyU * tauU)(i, 0));
106  }
107  }
108 
109  for (label j = 0; j < Nphi_p; j++)
110  {
111  label k = j + Nphi_u;
112  fvec(k) = m3(j);
113  }
114 
115  if (problem->bcMethod == "lift")
116  {
117  for (label j = 0; j < N_BC; j++)
118  {
119  fvec(j) = x(j) - bc(j);
120  }
121  }
122 
123  return 0;
124 }

138 void ReducedSteadyNSTurb::solveOnlineSUP(Eigen::MatrixXd velNow)
139 {
140  y.resize(Nphi_u + Nphi_p, 1);
141  y.setZero();
142 
143  for (label j = 0; j < N_BC; j++)
144  {
145  y(j) = velNow(j, 0);
146  }
147 
148  Color::Modifier red(Color::FG_RED);
149  Color::Modifier green(Color::FG_GREEN);
150  Color::Modifier def(Color::FG_DEFAULT);
151  Eigen::HybridNonLinearSolver<newtonSteadyNSTurb> hnls(newtonObject);
152  newtonObject.bc.resize(N_BC);
153  newtonObject.tauU = tauU;
154 
155  for (label j = 0; j < N_BC; j++)
156  {
157  newtonObject.bc(j) = velNow(j, 0);
158  }
159 
160  if (problem->viscCoeff == "L2")
161  {
162  for (label i = 0; i < nphiNut; i++)
163  {
164  newtonObject.gNut = problem->nutCoeff;
165  }
166  }
167  else if (problem->viscCoeff == "RBF")
168  {
169  for (label i = 0; i < nphiNut; i++)
170  {
171  newtonObject.gNut(i) = problem->rbfSplines[i]->eval(velNow);
172  rbfCoeff = newtonObject.gNut;
173  }
174  }
175  else
176  {
177  Info << "The way to compute the eddy viscosity coefficients has to be either L2 or RBF"
178  << endl;
179  exit(0);
180  }
181 
182  volScalarField nutTmp("nutRec", problem->nutModes[0] * 0);
183 
184  for (label j = 0; j < nphiNut; j++)
185  {
186  nutTmp += problem->nutModes[j] * newtonObject.gNut(j);
187  }
188 
189  nutRec.append(nutTmp);
190  newtonObject.nu = nu;
191  hnls.solve(y);
192  Eigen::VectorXd res(y);
193  newtonObject.operator()(y, res);
194  std::cout << "################## Online solve N° " << count_online_solve <<
195  " ##################" << std::endl;
196  std::cout << "Solving for the parameter: " << velNow << std::endl;
197 
198  if (res.norm() < 1e-5)
199  {
200  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
201  hnls.iter << " iterations " << def << std::endl << std::endl;
202  }
203  else
204  {
205  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
206  hnls.iter << " iterations " << def << std::endl << std::endl;
207  }
208 
209  count_online_solve += 1;
210 }

213 void ReducedSteadyNSTurb::reconstructSUP(fileName folder, int printEvery)
214 {
215  mkDir(folder);
216  ITHACAutilities::createSymLink(folder);
217  int counter = 0;
218  int nextWrite = 0;
219 
220  for (label i = 0; i < online_solution.size(); i++)
221  {
222  if (counter == nextWrite)
223  {
224  volVectorField uRec("uRec", Umodes[0] * 0);
225 
226  for (label j = 0; j < Nphi_u; j++)
227  {
228  uRec += Umodes[j] * online_solution[i](j + 1, 0);
229  }
230 
231  ITHACAstream::exportSolution(uRec, name(online_solution[i](0, 0)), folder);
232  volScalarField pRec("pRec", problem->Pmodes[0] * 0);
233 
234  for (label j = 0; j < Nphi_p; j++)
235  {
236  pRec += problem->Pmodes[j] * online_solution[i](j + Nphi_u + 1, 0);
237  }
238 
239  ITHACAstream::exportSolution(pRec, name(online_solution[i](0, 0)), folder);
240  nextWrite += printEvery;
241  UREC.append(uRec);
242  PREC.append(pRec);
243  }
244 
245  counter++;
246  }
247 }
