95 /*
96  * Returns the Hessian evaluated at x.
97  * The Hessian is an n x n matrix,
98  * where n is the dimension of x.
99  */
100 DenseMatrix BSpline::evalHessian(DenseVector x) const
101 {
102  checkInput(x);
103 
104  #ifndef NDEBUG
105  if (!pointInDomain(x))
106  throw Exception("BSpline::evalHessian: Evaluation at point outside domain.");
107  #endif // NDEBUG
108 
109  DenseMatrix H;
110  H.setZero(1,1);
111  DenseMatrix identity = DenseMatrix::Identity(numVariables, numVariables);
112  DenseMatrix caug = kroneckerProduct(identity, coefficients.transpose());
113  DenseMatrix DB = basis.evalBasisHessian(x);
114  H = caug*DB;
115 
116  // Fill in upper triangular of Hessian
117  for (size_t i = 0; i < numVariables; ++i)
118  for (size_t j = i+1; j < numVariables; ++j)
119  H(i,j) = H(j,i);
120 
121  return H;
122 }

231 void BSpline::reduceSupport(std::vector<double> lb, std::vector<double> ub, bool doRegularizeKnotVectors)
232 {
233  if (lb.size() != numVariables || ub.size() != numVariables)
234  throw Exception("BSpline::reduceSupport: Inconsistent vector sizes!");
235 
236  std::vector<double> sl = basis.getSupportLowerBound();
237  std::vector<double> su = basis.getSupportUpperBound();
238 
239  for (unsigned int dim = 0; dim < numVariables; dim++)
240  {
241  // Check if new domain is empty
242  if (ub.at(dim) <= lb.at(dim) || lb.at(dim) >= su.at(dim) || ub.at(dim) <= sl.at(dim))
243  throw Exception("BSpline::reduceSupport: Cannot reduce B-spline domain to empty set!");
244 
245  // Check if new domain is a strict subset
246  if (su.at(dim) < ub.at(dim) || sl.at(dim) > lb.at(dim))
247  throw Exception("BSpline::reduceSupport: Cannot expand B-spline domain!");
248 
249  // Tightest possible
250  sl.at(dim) = lb.at(dim);
251  su.at(dim) = ub.at(dim);
252  }
253 
254  if (doRegularizeKnotVectors)
255  {
256  regularizeKnotVectors(sl, su);
257  }
258 
259  // Remove knots and control points that are unsupported with the new bounds
260  if (!removeUnsupportedBasisFunctions(sl, su))
261  {
262  throw Exception("BSpline::reduceSupport: Failed to remove unsupported basis functions!");
263  }
264 }

293 // Computes knot averages: assumes that basis is initialized!
294 DenseMatrix BSpline::computeKnotAverages() const
295 {
296  // Calculate knot averages for each knot vector
297  std::vector<DenseVector> mu_vectors;
298  for (unsigned int i = 0; i < numVariables; i++)
299  {
300  std::vector<double> knots = basis.getKnotVector(i);
301  DenseVector mu = DenseVector::Zero(basis.getNumBasisFunctions(i));
302 
303  for (unsigned int j = 0; j < basis.getNumBasisFunctions(i); j++)
304  {
305  double knotAvg = 0;
306  for (unsigned int k = j+1; k <= j+basis.getBasisDegree(i); k++)
307  {
308  knotAvg += knots.at(k);
309  }
310  mu(j) = knotAvg/basis.getBasisDegree(i);
311  }
312  mu_vectors.push_back(mu);
313  }
314 
315  // Calculate vectors of ones (with same length as corresponding knot average vector)
316  std::vector<DenseVector> knotOnes;
317  for (unsigned int i = 0; i < numVariables; i++)
318  knotOnes.push_back(DenseVector::Ones(mu_vectors.at(i).rows()));
319 
320  // Fill knot average matrix one column at the time
321  DenseMatrix knot_averages = DenseMatrix::Zero(basis.getNumBasisFunctions(), numVariables);
322 
323  for (unsigned int i = 0; i < numVariables; i++)
324  {
325  DenseMatrix mu_ext(1,1); mu_ext(0,0) = 1;
326  for (unsigned int j = 0; j < numVariables; j++)
327  {
328  DenseMatrix temp = mu_ext;
329  if (i == j)
330  mu_ext = Eigen::kroneckerProduct(temp, mu_vectors.at(j));
331  else
332  mu_ext = Eigen::kroneckerProduct(temp, knotOnes.at(j));
333  }
334  if (mu_ext.rows() != basis.getNumBasisFunctions())
335  throw Exception("BSpline::computeKnotAverages: Incompatible size of knot average matrix.");
336  knot_averages.block(0, i, basis.getNumBasisFunctions(), 1) = mu_ext;
337  }
338 
339  return knot_averages;
340 }

351 void BSpline::regularizeKnotVectors(std::vector<double> &lb, std::vector<double> &ub)
352 {
353  // Add and remove controlpoints and knots to make the b-spline p-regular with support [lb, ub]
354  if (!(lb.size() == numVariables && ub.size() == numVariables))
355  throw Exception("BSpline::regularizeKnotVectors: Inconsistent vector sizes.");
356 
357  for (unsigned int dim = 0; dim < numVariables; dim++)
358  {
359  unsigned int multiplicityTarget = basis.getBasisDegree(dim) + 1;
360 
361  // Inserting many knots at the time (to save number of B-spline coefficient calculations)
362  // NOTE: This method generates knot insertion matrices with more nonzero elements than
363  // the method that inserts one knot at the time. This causes the preallocation of
364  // kronecker product matrices to become too small and the speed deteriorates drastically
365  // in higher dimensions because reallocation is necessary. This can be prevented by
366  // precomputing the number of nonzeros when preallocating memory (see myKroneckerProduct).
367  int numKnotsLB = multiplicityTarget - basis.getKnotMultiplicity(dim, lb.at(dim));
368  if (numKnotsLB > 0)
369  {
370  insertKnots(lb.at(dim), dim, numKnotsLB);
371  }
372 
373  int numKnotsUB = multiplicityTarget - basis.getKnotMultiplicity(dim, ub.at(dim));
374  if (numKnotsUB > 0)
375  {
376  insertKnots(ub.at(dim), dim, numKnotsUB);
377  }
378  }
379 }

381 bool BSpline::removeUnsupportedBasisFunctions(std::vector<double> &lb, std::vector<double> &ub)
382 {
383  if (lb.size() != numVariables || ub.size() != numVariables)
384  throw Exception("BSpline::removeUnsupportedBasisFunctions: Incompatible dimension of domain bounds.");
385 
386  SparseMatrix A = basis.reduceSupport(lb, ub);
387 
388  if (coefficients.size() != A.rows())
389  return false;
390 
391  // Remove unsupported control points (basis functions)
392  updateControlPoints(A.transpose());
393 
394  return true;
395 }

410 std::string BSpline::getDescription() const
411 {
412  std::string description("BSpline of degree");
413  auto degrees = getBasisDegrees();
414  // See if all degrees are the same.
415  bool equal = true;
416  for (size_t i = 1; i < degrees.size(); ++i)
417  {
418  equal = equal && (degrees.at(i) == degrees.at(i-1));
419  }
420 
421  if(equal)
422  {
423  description.append(" ");
424  description.append(std::to_string(degrees.at(0)));
425  }
426  else
427  {
428  description.append("s (");
429  for (size_t i = 0; i < degrees.size(); ++i)
430  {
431  description.append(std::to_string(degrees.at(i)));
432  if (i + 1 < degrees.size())
433  {
434  description.append(", ");
435  }
436  }
437  description.append(")");
438  }
439 
440  return description;
