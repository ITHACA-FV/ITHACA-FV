91 // Operator to evaluate the residual for the supremizer approach
92 int newtonUnsteadyNSTurbSUP::operator()(const Eigen::VectorXd& x,
93  Eigen::VectorXd& fvec) const
94 {
95  Eigen::VectorXd a_dot(Nphi_u);
96  Eigen::VectorXd a_tmp(Nphi_u);
97  Eigen::VectorXd bTmp(Nphi_p);
98  a_tmp = x.head(Nphi_u);
99  bTmp = x.tail(Nphi_p);
100  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
101  // Convective term
102  Eigen::MatrixXd cc(1, 1);
103  // Mom Term
104  Eigen::VectorXd m1 = problem->bTotalMatrix * a_tmp * nu;
105  // Gradient of pressure
106  Eigen::VectorXd m2 = problem->K_matrix * bTmp;
107  // Mass Term
108  Eigen::VectorXd m5 = problem->M_matrix * a_dot;
109  // Pressure Term
110  Eigen::VectorXd m3 = problem->P_matrix * a_tmp;
111 
112  for (label i = 0; i < Nphi_u; i++)
113  {
114  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp - gNut.transpose() *
115  problem->cTotalMatrix[i] * a_tmp;
116  fvec(i) = - m5(i) + m1(i) - cc(0, 0) - m2(i);
117  }
118 
119  for (label j = 0; j < Nphi_p; j++)
120  {
121  label k = j + Nphi_u;
122  fvec(k) = m3(j);
123  }
124 
125  for (label j = 0; j < N_BC; j++)
126  {
127  fvec(j) = x(j) - bc(j);
128  }
129 
130  return 0;
131 }

144 // Operator to evaluate the residual for the supremizer approach
145 int newton_UnsteadyNSTurb_PPE::operator()(const Eigen::VectorXd& x,
146  Eigen::VectorXd& fvec) const
147 {
148  Eigen::VectorXd a_dot(Nphi_u);
149  Eigen::VectorXd a_tmp(Nphi_u);
150  Eigen::VectorXd bTmp(Nphi_p);
151  a_tmp = x.head(Nphi_u);
152  bTmp = x.tail(Nphi_p);
153  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
154  // Convective terms
155  Eigen::MatrixXd cc(1, 1);
156  Eigen::MatrixXd gg(1, 1);
157  Eigen::MatrixXd bb(1, 1);
158  // Mom Term
159  Eigen::VectorXd m1 = problem->B_matrix * a_tmp * nu;
160  // Gradient of pressure
161  Eigen::VectorXd m2 = problem->K_matrix * bTmp;
162  // Mass Term
163  Eigen::VectorXd m5 = problem->M_matrix * a_dot;
164  // Pressure Term
165  Eigen::VectorXd m3 = problem->D_matrix * bTmp;
166  // BC PPE
167  Eigen::VectorXd m6 = problem->BC1_matrix * a_tmp * nu;
168  // BC PPE
169  Eigen::VectorXd m7 = problem->BC3_matrix * a_tmp * nu;
170 
171  for (label i = 0; i < Nphi_u; i++)
172  {
173  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp - gNut.transpose() *
174  problem->cTotalMatrix[i] * a_tmp;
175  fvec(i) = - m5(i) + m1(i) - cc(0, 0) - m2(i);
176  }
177 
178  for (label j = 0; j < Nphi_p; j++)
179  {
180  label k = j + Nphi_u;
181  gg = a_tmp.transpose() * problem->G_matrix[j] * a_tmp;
182  bb = a_tmp.transpose() * problem->BC2_matrix[j] * a_tmp;
183  //fvec(k) = m3(j, 0) - gg(0, 0) - m6(j, 0) + bb(0, 0);
184  fvec(k) = m3(j, 0) + gg(0, 0) - m7(j, 0);
185  }
186 
187  for (label j = 0; j < N_BC; j++)
188  {
189  fvec(j) = x(j) - bc(j);
190  }
191 
192  return 0;
193 }

207 void ReducedUnsteadyNSTurb::solveOnlineSUP(Eigen::MatrixXd& velNow,
208  label startSnap)
209 {
210  // Create and resize the solution vector
211  y.resize(Nphi_u + Nphi_p, 1);
212  y.setZero();
213  y.head(Nphi_u) = ITHACAutilities::get_coeffs(Usnapshots[startSnap], Umodes);
214  y.tail(Nphi_p) = ITHACAutilities::get_coeffs_ortho(Psnapshots[startSnap],
215  Pmodes);
216 
217  // Change initial condition for the lifting function
218  for (label j = 0; j < N_BC; j++)
219  {
220  y(j) = velNow(j, 0);
221  }
222 
223  // Set some properties of the newton object
224  newtonObjectSUP.nu = nu;
225  newtonObjectSUP.y_old = y;
226  newtonObjectSUP.dt = dt;
227  newtonObjectSUP.bc.resize(N_BC);
228 
229  for (label j = 0; j < N_BC; j++)
230  {
231  newtonObjectSUP.bc(j) = velNow(j, 0);
232  }
233 
234  // Set number of online solutions
235  int Ntsteps = static_cast<int>((finalTime - tstart) / dt);
236  online_solution.resize(Ntsteps);
237  // Set the initial time
238  time = tstart;
239  // Counting variable
240  int counter = 0;
241  // Create vector to store temporal solution and save initial condition as first solution
242  Eigen::MatrixXd tmp_sol(Nphi_u + Nphi_p + 1, 1);
243  tmp_sol(0) = time;
244  tmp_sol.col(0).tail(y.rows()) = y;
245 
246  if (time != 0)
247  {
248  online_solution[counter] = tmp_sol;
249  counter ++;
250  }
251 
252  // Create nonlinear solver object
253  Eigen::HybridNonLinearSolver<newtonUnsteadyNSTurbSUP> hnls(newtonObjectSUP);
254  // Set output colors for fancy output
255  Color::Modifier red(Color::FG_RED);
256  Color::Modifier green(Color::FG_GREEN);
257  Color::Modifier def(Color::FG_DEFAULT);
258 
259  // This part up to the while loop is just to compute the eddy viscosity field at time = startTime if time is not equal to zero
260  if (time != 0)
261  {
262  volScalarField nutTmp("nutRec", problem->nutModes[0] * 0);
263  Eigen::VectorXd gNut0;
264  gNut0.resize(nphiNut);
265  std::vector<double> tv0;
266  tv0.resize(1);
267  tv0[0] = time;
268 
269  for (label i = 0; i < nphiNut; i++)
270  {
271  gNut0(i) = problem->rbfSplines[i]->eval(tv0);
272  }
273 
274  for (label j = 0; j < nphiNut; j++)
275  {
276  nutTmp += problem->nutModes[j] * gNut0(j);
277  }
278 
279  nutRec.append(nutTmp);
280  }
281 
282  // Start the time loop
283  while (time < finalTime)
284  {
285  time = time + dt;
286  std::vector<double> tv;
287  tv.resize(1);
288  tv[0] = time;
289 
290  for (label i = 0; i < nphiNut; i++)
291  {
292  newtonObjectSUP.gNut(i) = problem->rbfSplines[i]->eval(tv);
293  }
294 
295  volScalarField nutTmp("nutRec", problem->nutModes[0] * 0);
296 
297  for (label j = 0; j < nphiNut; j++)
298  {
299  nutTmp += problem->nutModes[j] * newtonObjectSUP.gNut(j);
300  }
301 
302  nutRec.append(nutTmp);
303  Eigen::VectorXd res(y);
304  res.setZero();
305  hnls.solve(y);
306 
307  for (label j = 0; j < N_BC; j++)
308  {
309  y(j) = velNow(j, 0);
310  }
311 
312  newtonObjectSUP.operator()(y, res);
313  newtonObjectSUP.y_old = y;
314  std::cout << "################## Online solve N° " << count_online_solve <<
315  " ##################" << std::endl;
316  Info << "Time = " << time << endl;
317  std::cout << "Solving for the parameter: " << velNow << std::endl;
318 
319  if (res.norm() < 1e-5)
320  {
321  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
322  hnls.iter << " iterations " << def << std::endl << std::endl;
323  }
324  else
325  {
326  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
327  hnls.iter << " iterations " << def << std::endl << std::endl;
328  }
329 
330  count_online_solve += 1;
331  tmp_sol(0) = time;
332  tmp_sol.col(0).tail(y.rows()) = y;
333 
334  if (counter >= online_solution.size())
335  {
336  online_solution.append(tmp_sol);
337  }
338  else
339  {
340  online_solution[counter] = tmp_sol;
341  }
342 
343  counter ++;
344  }
345 
346  // Save the solution
347  ITHACAstream::exportMatrix(online_solution, "red_coeff", "python",
348  "./ITHACAoutput/red_coeff");
349  ITHACAstream::exportMatrix(online_solution, "red_coeff", "matlab",
350  "./ITHACAoutput/red_coeff");
351  count_online_solve += 1;
352 }
