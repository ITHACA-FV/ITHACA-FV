218 DEIM<T, S>::DEIM (T& s, int MaxModes, word FunctionName)
219  :
220  SnapShotsMatrix(s),
221  MaxModes(MaxModes),
222  FunctionName(FunctionName)
223 {
224  Eigen::MatrixXd A;
225  Eigen::VectorXd b;
226  Eigen::VectorXd c;
227  Eigen::VectorXd r;
228  Eigen::VectorXd rho(1);
229  modes = ITHACAPOD::DEIMmodes(SnapShotsMatrix, MaxModes, FunctionName);
230  MatrixModes = Foam2Eigen::PtrList2Eigen(modes);
231  int ind_max, c1;
232  double max = MatrixModes.cwiseAbs().col(0).maxCoeff(&ind_max, &c1);
233  rho(0) = max;
234  magicPoints.append(ind_max);
235  U = MatrixModes.col(0);
236  P.resize(MatrixModes.rows(), 1);
237  P.insert(ind_max, 0) = 1;
238  for (int i = 1; i < MaxModes; i++)
239  {
240  A = P.transpose() * U;
241  b = P.transpose() * MatrixModes.col(i);
242  c = A.lu().solve(b);
243  r = MatrixModes.col(i) - U * c;
244  max = r.cwiseAbs().maxCoeff(&ind_max, &c1);
245  P.conservativeResize(MatrixModes.rows(), i + 1);
246  P.insert(ind_max, i) = 1;
247  U.conservativeResize(MatrixModes.rows(), i + 1);
248  U.col(i) = MatrixModes.col(i);
249  rho.conservativeResize(i + 1);
250  rho(i) = max;
251  magicPoints.append(ind_max);
252  }
253  MatrixOnline = U * ((P.transpose() * U).inverse());
254 }
