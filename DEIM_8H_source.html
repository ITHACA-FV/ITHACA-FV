215 DEIM<T, S>::DEIM (T& s, int MaxModes, word FunctionName)
216  :
217  SnapShotsMatrix(s),
218  MaxModes(MaxModes),
219  FunctionName(FunctionName)
220 {
221  Eigen::MatrixXd A;
222  Eigen::VectorXd b;
223  Eigen::VectorXd c;
224  Eigen::VectorXd r;
225  Eigen::VectorXd rho(1);
226  modes = ITHACAPOD::DEIMmodes(SnapShotsMatrix, MaxModes, FunctionName);
227  MatrixModes = Foam2Eigen::PtrList2Eigen(modes);
228  int ind_max, c1;
229  double max = MatrixModes.cwiseAbs().col(0).maxCoeff(&ind_max, &c1);
230  rho(0) = max;
231  magicPoints.append(ind_max);
232  U = MatrixModes.col(0);
233  P.resize(MatrixModes.rows(), 1);
234  P.insert(ind_max, 0) = 1;
235 
236  for (int i = 1; i < MaxModes; i++)
237  {
238  A = P.transpose() * U;
239  b = P.transpose() * MatrixModes.col(i);
240  c = A.lu().solve(b);
241  r = MatrixModes.col(i) - U * c;
242  max = r.cwiseAbs().maxCoeff(&ind_max, &c1);
243  P.conservativeResize(MatrixModes.rows(), i + 1);
244  P.insert(ind_max, i) = 1;
245  U.conservativeResize(MatrixModes.rows(), i + 1);
246  U.col(i) = MatrixModes.col(i);
247  rho.conservativeResize(i + 1);
248  rho(i) = max;
249  magicPoints.append(ind_max);
250  }
251 
252  MatrixOnline = U * ((P.transpose() * U).inverse());
253 }
