64 Eigen::MatrixXd reducedProblem::solveLinearSys(List<Eigen::MatrixXd> LinSys,
65  Eigen::MatrixXd x, Eigen::VectorXd& residual, const Eigen::MatrixXd& bc,
66  const std::string solverType)
67 
68 {
69  M_Assert(solverType == "fullPivLu" || solverType == "partialPivLu"
70  || solverType == "householderQR" || solverType == "colPivHouseholderQR"
71  || solverType == "fullPivHouseholderQR"
72  || solverType == "CompleteOrthogonalDecomposition" || solverType == "llt"
73  || solverType == "ldlt" || solverType == "bdcSvd"
74  || solverType == "jacobiSvd", "solver not defined");
75  Eigen::MatrixXd y;
76 
77  for (label i = 0; i < bc.size(); i++)
78  {
79  LinSys[0].row(i) *= 0;
80  LinSys[0](i, i) = 1;
81  LinSys[1](i, 0) = bc(i);
82  }
83 
84  residual = LinSys[0] * x - LinSys[1];
85 
86  if (solverType == "fullPivLu")
87  {
88  y = LinSys[0].fullPivLu().solve(LinSys[1]);
89  }
90  else if (solverType == "partialPivLu")
91  {
92  y = LinSys[0].partialPivLu().solve(LinSys[1]);
93  }
94  else if (solverType == "householderQr")
95  {
96  y = LinSys[0].householderQr().solve(LinSys[1]);
97  }
98  else if (solverType == "colPivHouseholderQr")
99  {
100  y = LinSys[0].colPivHouseholderQr().solve(LinSys[1]);
101  }
102  else if (solverType == "fullPivHouseholderQr")
103  {
104  y = LinSys[0].fullPivHouseholderQr().solve(LinSys[1]);
105  }
106  else if (solverType == "completeOrthogonalDecomposition")
107  {
108  y = LinSys[0].completeOrthogonalDecomposition().solve(LinSys[1]);
109  }
110  else if (solverType == "llt")
111  {
112  y = LinSys[0].llt().solve(LinSys[1]);
113  }
114  else if (solverType == "ldlt")
115  {
116  y = LinSys[0].ldlt().solve(LinSys[1]);
117  }
118  else if (solverType == "bdcSvd")
119  {
120  y = LinSys[0].bdcSvd().solve(LinSys[1]);
121  }
122  else if (solverType == "jacobiSvd")
123  {
124  y = LinSys[0].jacobiSvd().solve(LinSys[1]);
125  }
126 
127  return y;
128 }

144 Eigen::MatrixXd onlineInterp::getInterpCoeffRBF(std::vector<SPLINTER::RBFSpline>
145  rbfVec, Eigen::MatrixXd mu_interp)
146 {
147  M_Assert(mu_interp.cols() == Nmu_samples,
148  "Matrix 'mu_interp' must have same number and order of columns (i.e. parameters) as the matrix 'mu_samples'.");
149  label Nmodes = rbfVec.size();
150  // Nsamples (snapshots) to evaluate the interpolator at
151  label Nsamples = mu_interp.rows();
152  Eigen::MatrixXd muInterpRefined(mu_interp.rows(), 0);
153 
154  // Discard columns of mu_interp that have constant value for all elements, and use a refined matrix muInterpRefined
155  for (label i = 0; i < mu_interp.cols(); i++)
156  {
157  double firstVal = mu_interp(0, i);
158  bool isConst = (mu_interp.col(i).array() == firstVal).all();
159 
160  // Consider only columns with non-constant values
161  if (isConst == 0)
162  {
163  muInterpRefined.conservativeResize(muInterpRefined.rows(),
164  muInterpRefined.cols() + 1);
165  muInterpRefined.col(muInterpRefined.cols() - 1) = mu_interp.col(i);
166  }
167  }
168 
169  label Nmu = muInterpRefined.cols();
170  Eigen::MatrixXd coeff_interp(Nmodes, Nsamples);
171 
172  for (label i = 0; i < Nmodes; i++)
173  {
174  for (label j = 0; j < Nsamples; j++)
175  {
176  SPLINTER::DenseVector x(Nmu);
177 
178  for (label k = 0; k < Nmu; k++)
179  {
180  x(k) = muInterpRefined(j, k);
181  }
182 
183  coeff_interp(i, j) = rbfVec[i].eval(x);
184  }
185  }
186 
187  return coeff_interp;
188 }
