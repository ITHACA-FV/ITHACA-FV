78 Eigen::MatrixXd onlineInterp::getInterpCoeffRBF(std::vector<SPLINTER::RBFSpline>
79  rbfVec, Eigen::MatrixXd mu_interp)
80 {
81  M_Assert(mu_interp.cols() == Nmu_samples,
82  "Matrix 'mu_interp' must have same number and order of columns (i.e. parametes) as the matrix 'mu_samples'.");
83  label Nmodes = rbfVec.size();
84  // Nsamples (snapshots) to evaluate the interpolator at
85  label Nsamples = mu_interp.rows();
86  Eigen::MatrixXd muInterpRefined(mu_interp.rows(), 0);
87 
88  // Discard columns of mu_interp that have constant value for all elements, and use a refined matrix muInterpRefined
89  for (label i = 0; i < mu_interp.cols(); i++)
90  {
91  double firstVal = mu_interp(0, i);
92  bool isConst = (mu_interp.col(i).array() == firstVal).all();
93 
94  // Consider only columns with non-constant values
95  if (isConst == 0)
96  {
97  muInterpRefined.conservativeResize(muInterpRefined.rows(),
98  muInterpRefined.cols() + 1);
99  muInterpRefined.col(muInterpRefined.cols() - 1) = mu_interp.col(i);
100  }
101  }
102 
103  label Nmu = muInterpRefined.cols();
104  Eigen::MatrixXd coeff_interp(Nmodes, Nsamples);
105 
106  for (label i = 0; i < Nmodes; i++)
107  {
108  for (label j = 0; j < Nsamples; j++)
109  {
110  SPLINTER::DenseVector x(Nmu);
111 
112  for (label k = 0; k < Nmu; k++)
113  {
114  x(k) = muInterpRefined(j, k);
115  }
116 
117  coeff_interp(i, j) = rbfVec[i].eval(x);
118  }
119  }
120 
121  return coeff_interp;
122 }
