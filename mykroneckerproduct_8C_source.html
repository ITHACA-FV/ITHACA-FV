16 /*
17  * Implementation of Kronecker product.
18  * Eigen has an implementation of the Kronecker product,
19  * but it is very slow due to poor memory reservation.
20  * See: https://forum.kde.org/viewtopic.php?f=74&t=106955&p=309990&hilit=kronecker#p309990
21  * When Eigen update their implementation, and officially support it, we switch to that.
22  */
23 SparseMatrix myKroneckerProduct(const SparseMatrix &A, const SparseMatrix &B)
24 {
25  SparseMatrix AB(A.rows()*B.rows(), A.cols()*B.cols());
26 
27  // Reserve memory for AB
28 
29  //AB.reserve(A.nonZeros()*B.nonZeros()); // Does not reserve inner vectors (slow)
30  //int innernnz = std::ceil(A.nonZeros()*B.nonZeros()/AB.outerSize());
31  //AB.reserve(Eigen::VectorXi::Constant(AB.outerSize(), innernnz)); // Assumes equal distribution of non-zeros (slow)
32 
33  // Calculate exact number of non-zeros for each inner vector
34  Eigen::VectorXi nnzA = Eigen::VectorXi::Zero(A.outerSize());
35  Eigen::VectorXi nnzB = Eigen::VectorXi::Zero(B.outerSize());
36  Eigen::VectorXi nnzAB = Eigen::VectorXi::Zero(AB.outerSize());
37  //innerNonZeros.setZero();
38 
39  for (int jA = 0; jA < A.outerSize(); ++jA)
40  {
41  int nnz = 0;
42  for (SparseMatrix::InnerIterator itA(A,jA); itA; ++itA) nnz++;
43  nnzA(jA) = nnz;
44  }
45 
46  for (int jB = 0; jB < B.outerSize(); ++jB)
47  {
48  int nnz = 0;
49  for (SparseMatrix::InnerIterator itB(B,jB); itB; ++itB) nnz++;
50  nnzB(jB) = nnz;
51  }
52 
53  int innz = 0;
54  for (int i = 0; i < nnzA.rows(); ++i)
55  {
56  for (int j = 0; j < nnzB.rows(); ++j)
57  {
58  nnzAB(innz) = nnzA(i)*nnzB(j);
59  innz++;
60  }
61  }
62 
63  AB.reserve(nnzAB);
64 
65  // Non-zero tolerance
66  double tolerance = std::numeric_limits<SparseMatrix::Scalar>::epsilon();
67 
68  // Compute Kronecker product
69  for (int jA = 0; jA < A.outerSize(); ++jA)
70  {
71  for (SparseMatrix::InnerIterator itA(A,jA); itA; ++itA)
72  {
73  if (std::abs(itA.value()) > tolerance)
74  {
75  int jrow = itA.row()*B.rows();
76  int jcol = itA.col()*B.cols();
77 
78  for (int jB = 0; jB < B.outerSize(); ++jB)
79  {
80  for (SparseMatrix::InnerIterator itB(B,jB); itB; ++itB)
81  {
82  if (std::abs(itA.value()*itB.value()) > tolerance)
83  {
84  int row = jrow + itB.row();
85  int col = jcol + itB.col();
86  AB.insert(row,col) = itA.value()*itB.value();
87  }
88  }
89  }
90  }
91  }
92  }
93  AB.makeCompressed();
94  return AB;
95 }
