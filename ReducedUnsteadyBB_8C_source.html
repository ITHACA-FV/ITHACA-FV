66

68  {
69  LUmodes.append(problem->supmodes[k]);

71

91 int newton_unsteadyBB_sup::operator()(const Eigen::VectorXd& x,
92  Eigen::VectorXd& fvec) const
93 {
94  Eigen::VectorXd a_dot(Nphi_u);
95  Eigen::VectorXd a_tmp(Nphi_u);
96  Eigen::VectorXd b_tmp(Nphi_prgh);
97  Eigen::VectorXd c_dot(Nphi_t);
98  Eigen::VectorXd c_tmp(Nphi_t);
99  a_tmp = x.head(Nphi_u);
100  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
101  b_tmp = x.segment(Nphi_u, Nphi_prgh);
102  c_tmp = x.tail(Nphi_t);
103  c_dot = (x.tail(Nphi_t) - y_old.tail(Nphi_t)) / dt;
104  // Convective term
105  Eigen::MatrixXd cc(1, 1);
106  // Diffusive Term
107  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
108  // Mass Term Velocity
109  Eigen::VectorXd M5 = problem->M_matrix * a_dot;
110  // Gradient of pressure
111  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
112  // Continuity
113  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
114  // Buoyancy Term
115  Eigen::VectorXd M10 = problem->H_matrix * c_tmp;
116  // Convective term temperature
117  Eigen::MatrixXd qq(1, 1);
118  // diffusive term temperature
119  Eigen::VectorXd M6 = problem->Y_matrix * c_tmp * (nu / Pr);
120  // Mass Term Temperature
121  Eigen::VectorXd M8 = problem->W_matrix * c_dot;
122 
123  for (label i = 0; i < Nphi_u; i++)
124  {
125  cc = a_tmp.transpose() * Eigen::SliceFromTensor(problem->C_tensor, 0,
126  i) * a_tmp;
127  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M10(i) - M2(i);
128  }
129 
130  for (label j = 0; j < Nphi_prgh; j++)
131  {
132  label k = j + Nphi_u;
133  fvec(k) = M3(j);
134  }
135 
136  for (label j = 0; j < Nphi_t; j++)
137  {
138  label k = j + Nphi_u + Nphi_prgh;
139  qq = a_tmp.transpose() * problem->Q_matrix[j] * c_tmp;
140  fvec(k) = -M8(j) + M6(j) - qq(0, 0);
141  }
142 
143  for (label j = 0; j < N_BC; j++)
144  {
145  fvec(j) = x(j) - BC(j);
146  }
147 
148  for (label j = 0; j < N_BC_t; j++)
149  {
150  label k = j + Nphi_u + Nphi_prgh;
151  fvec(k) = x(k) - BC_t(j);
152  }
153 
154  return 0;
155 }

203  Eigen::VectorXd M7 = problem->BC3_matrix * a_tmp * nu;
204  // diffusive term temperature

223  }
224

226  {
227  label k = j + Nphi_u + Nphi_prgh;
228  qq = a_tmp.transpose() * problem->Q_matrix[j] * c_tmp;
229  fvec(k) = -M8(j) + M9(j) - qq(0, 0);
230  }
231 
232  // for (label j = 0; j < N_BC; j++)

258

261  time = time + dt;
262  counter ++;
263  }
264 
265  // Set size of online solution
266  online_solutiont.resize(Nphi_u + Nphi_prgh + Nphi_t + 1, counter + 1);
267  // Set initial condition for online solve
268  volScalarField T_IC("T_IC", problem->Tfield[0]);
269

271  {
272  for (label i = 0; i < N_BC_t; i++)
273  {
274  if (j == problem->inletIndexT(i, 0))

281  }

283 
284  // Create and resize the solution vector

289  LUmodes);

302  newton_object_sup.dt = dt;
303  newton_object_sup.Pr = Pr;
304  newton_object_sup.BC_t.resize(N_BC_t);

310  newton_object_sup.BC_t(j) = temp_now_BC(j, 0);
311  }
312

314  {
315  newton_object_sup.BC(j) = vel_now_BC(j, 0);
316  }
317

319  time = tstart;
320  // Create vector to store temporal solution and save initial condition as first solution
321  Eigen::MatrixXd tmp_sol(Nphi_u + Nphi_prgh + Nphi_t + 1, 1);
322  tmp_sol(0) = time;
323  tmp_sol.col(0).tail(y.rows()) = y;
324  online_solutiont.col(0) = tmp_sol;

326  Eigen::HybridNonLinearSolver<newton_unsteadyBB_sup> hnls(newton_object_sup);
327  // Set output colors for fancy output
328  Color::Modifier red(Color::FG_RED);
329  Color::Modifier green(Color::FG_GREEN);
330  Color::Modifier def(Color::FG_DEFAULT);
331

333  for (label i = 1; i < online_solutiont.cols(); i++)
334  {
335  time = time + dt;
336  Eigen::VectorXd res(y);
337  res.setZero();
338  hnls.solve(y);
339 
340  for (label j = 0; j < N_BC; j++)
341  {
342  y(j) = vel_now_BC(j, 0);
343  }
344 
345  for (label j = 0; j < N_BC_t; j++)
346  {
347  label k = j + Nphi_prgh + Nphi_u;
348  y(k) = temp_now_BC(j, 0);
349  }
350 
351  newton_object_sup.operator()(y, res);
352  newton_object_sup.y_old = y;
353  Info << "Time = " << time << endl;
354 
355  if (res.norm() < 1e-5)
356  {
357  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
358  hnls.iter << " iterations " << def << std::endl << std::endl;
359  }
360  else

362  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
363  hnls.iter << " iterations " << def << std::endl << std::endl;
364  }
365 
366  tmp_sol(0) = time;
367  tmp_sol.col(0).tail(y.rows()) = y;

369  }
370 
371  // Save the current solution
372  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "python",
373  "./ITHACAoutput/red_coeff/" + name(NParaSet) + "/");
374  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "matlab",
375  "./ITHACAoutput/red_coeff/" + name(NParaSet) + "/");
376  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "eigen",
377  "./ITHACAoutput/red_coeff/" + name(NParaSet) + "/");
378  return online_solutiont;
379 }
