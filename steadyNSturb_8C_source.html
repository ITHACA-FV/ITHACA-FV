122 List < Eigen::MatrixXd > steadyNSturb::turbulence_term1(label NUmodes,
123  label NSUPmodes, label Nnutmodes)
124 {
125  label Csize = NUmodes + NSUPmodes + liftfield.size();
126  List < Eigen::MatrixXd > CT1_matrix;
127  CT1_matrix.setSize(Csize);
128 
129  for (label j = 0; j < Csize; j++)
130  {
131  CT1_matrix[j].resize(Nnutmodes, Csize);
132  CT1_matrix[j] = CT1_matrix[j] * 0;
133  }
134 
135  PtrList<volVectorField> Together(0);
136 
137  // Create PTRLIST with lift, velocities and supremizers
138 
139  if (liftfield.size() != 0)
140  {
141  for (label k = 0; k < liftfield.size(); k++)
142  {
143  Together.append(liftfield[k]);
144  }
145  }
146 
147  if (NUmodes != 0)
148  {
149  for (label k = 0; k < NUmodes; k++)
150  {
151  Together.append(Umodes[k]);
152  }
153  }
154 
155  if (NSUPmodes != 0)
156  {
157  for (label k = 0; k < NSUPmodes; k++)
158  {
159  Together.append(supmodes[k]);
160  }
161  }
162 
163  for (label i = 0; i < Csize; i++)
164  {
165  Info << "Filling layer number " << i + 1 << " in the matrix CT1_matrix" << endl;
166 
167  for (label j = 0; j < Nnutmodes; j++)
168  {
169  for (label k = 0; k < Csize; k++)
170  {
171  CT1_matrix[i](j, k) = fvc::domainIntegrate(Together[i] & fvc::laplacian(
172  nuTmodes[j], Together[k])).value();
173  }
174  }
175  }
176 
177  // Export the matrix
178  ITHACAstream::exportMatrix(CT1_matrix, "CT1_matrix", "python",
179  "./ITHACAoutput/Matrices/");
180  ITHACAstream::exportMatrix(CT1_matrix, "CT1_matrix", "matlab",
181  "./ITHACAoutput/Matrices/");
182  ITHACAstream::exportMatrix(CT1_matrix, "CT1_matrix", "eigen",
183  "./ITHACAoutput/Matrices/CT1");
184  return CT1_matrix;
185 }

258 Eigen::MatrixXd steadyNSturb::BT_turbulence(label NUmodes, label NSUPmodes)
259 {
260  label BTsize = NUmodes + NSUPmodes + liftfield.size();
261  Eigen::MatrixXd BT_matrix(BTsize, BTsize);
262  BT_matrix = BT_matrix * 0;
263  // Create PTRLIST with lift, velocities and supremizers
264  PtrList<volVectorField> Together(0);
265 
266  if (liftfield.size() != 0)
267  {
268  for (label k = 0; k < liftfield.size(); k++)
269  {
270  Together.append(liftfield[k]);
271  }
272  }
273 
274  if (NUmodes != 0)
275  {
276  for (label k = 0; k < NUmodes; k++)
277  {
278  Together.append(Umodes[k]);
279  }
280  }
281 
282  if (NSUPmodes != 0)
283  {
284  for (label k = 0; k < NSUPmodes; k++)
285  {
286  Together.append(supmodes[k]);
287  }
288  }
289 
290  // Project everything
291  for (label i = 0; i < BTsize; i++)
292  {
293  for (label j = 0; j < BTsize; j++)
294  {
295  BT_matrix(i, j) = fvc::domainIntegrate(Together[i] & (fvc::div(dev((T(fvc::grad(
296  Together[j]))))))).value();
297  }
298  }
299 
300  // Export the matrix
301  ITHACAstream::exportMatrix(BT_matrix, "BT_matrix", "python",
302  "./ITHACAoutput/Matrices/");
303  ITHACAstream::exportMatrix(BT_matrix, "BT_matrix", "matlab",
304  "./ITHACAoutput/Matrices/");
305  ITHACAstream::exportMatrix(BT_matrix, "BT_matrix", "eigen",
306  "./ITHACAoutput/Matrices/");
307  return BT_matrix;

311  label Nnut)

325  "CT2_matrix");

329  NUmodes = NU;

336  P_matrix = divergence_term(NUmodes, NPmodes, NSUPmodes);

340  CT2_matrix = turbulence_term2(NUmodes, NSUPmodes, Nnutmodes);
341  }
342

344  C_total_matrix.setSize(C_matrix.size());
345 
346  for (label i = 0; i < C_matrix.size(); i++)

348  C_total_matrix[i] = CT2_matrix[i] + CT1_matrix[i];
349  }
350 
351  NUmodes = NU;
352  NPmodes = NP;
353  NSUPmodes = NSUP;
354  Nnutmodes = Nnut;
355  // Get the coeffs for interpolation (the orthonormal one is used because basis are orthogonal)

357  ITHACAstream::exportMatrix(Ncoeff, "Ncoeff", "python",
358  "./ITHACAoutput/Matrices/");

361 
362  for (int i = 0; i < Nnutmodes; i++)
363  {
364  SAMPLES[i] = new SPLINTER::DataTable(1, 1);
365 
366  for (int j = 0; j < Ncoeff.cols(); j++)
367  {
368  SAMPLES[i]->addSample(mu.row(j), Ncoeff(i, j));
369  }

372  SPLINTER::RadialBasisFunctionType::GAUSSIAN);
