104 List < Eigen::MatrixXd > steadyNSturb::turbulence_term1(label NUmodes, label NSUPmodes, label Nnutmodes)
105 {
106  label Csize = NUmodes + NSUPmodes + liftfield.size();
107  List < Eigen::MatrixXd > CT1_matrix;
108 
109  CT1_matrix.setSize(Csize);
110 
111  for (label j = 0; j < Csize; j++)
112  {
113  CT1_matrix[j].resize(Nnutmodes, Csize);
114  CT1_matrix[j] = CT1_matrix[j] *0;
115  }
116 
117  PtrList<volVectorField> Together(0);
118 
119  // Create PTRLIST with lift, velocities and supremizers
120 
121  if (liftfield.size() != 0)
122  {
123  for (label k = 0; k < liftfield.size(); k++)
124  {
125  Together.append(liftfield[k]);
126  }
127  }
128  if (NUmodes != 0)
129  {
130  for (label k = 0; k < NUmodes; k++)
131  {
132  Together.append(Umodes[k]);
133  }
134  }
135  if (NSUPmodes != 0)
136  {
137  for (label k = 0; k < NSUPmodes; k++)
138  {
139  Together.append(supmodes[k]);
140  }
141  }
142 
143 
144 
145  for (label i = 0; i < Csize; i++)
146  {
147 
148  Info << "Filling layer number " << i+1 << " in the matrix CT1_matrix" << endl;
149 
150  for (label j = 0; j < Nnutmodes; j++)
151  {
152  for (label k = 0;k < Csize; k++)
153  {
154  CT1_matrix[i](j,k) = fvc::domainIntegrate(Together[i]&fvc::laplacian(nuTmodes[j],Together[k])).value();
155  }
156  }
157  }
158  // Export the matrix
159  ITHACAstream::exportMatrix(CT1_matrix, "CT1_matrix", "python", "./ITHACAoutput/Matrices/");
160  ITHACAstream::exportMatrix(CT1_matrix, "CT1_matrix", "matlab", "./ITHACAoutput/Matrices/");
161  ITHACAstream::exportMatrix(CT1_matrix, "CT1_matrix", "eigen", "./ITHACAoutput/Matrices/CT1");
162  return CT1_matrix;
163 }

287 {
288

302  NUmodes = NU;
303  NPmodes = NP;
304  NSUPmodes = NSUP;

307  B_matrix = diffusive_term(NUmodes, NPmodes, NSUPmodes);
308  C_matrix = convective_term(NUmodes, NPmodes, NSUPmodes);
309  K_matrix = pressure_gradient_term(NUmodes, NPmodes, NSUPmodes);
310  P_matrix = divergence_term(NUmodes, NPmodes, NSUPmodes);
311  M_matrix = mass_term(NUmodes, NPmodes, NSUPmodes);
312  BT_matrix = BT_turbulence(NUmodes,NSUPmodes);
313  CT1_matrix = turbulence_term1(NUmodes,NSUPmodes,Nnutmodes);
314  CT2_matrix = turbulence_term2(NUmodes,NSUPmodes,Nnutmodes);
315  }
316

318 
319  C_total_matrix.setSize(C_matrix.size());
320 
321  for (label i = 0; i < C_matrix.size(); i++)
322  {
323  C_total_matrix[i] = CT2_matrix[i] + CT1_matrix[i];
324  }
325 
326  NUmodes = NU;
327  NPmodes = NP;
328  NSUPmodes = NSUP;
329  Nnutmodes = Nnut;
330  // Get the coeffs for interpolation (the orthonormal one is used because basis are orthogonal)
331  Eigen::MatrixXd Ncoeff = ITHACAutilities::get_coeffs_ortho(nutFields, nuTmodes);
332  ITHACAstream::exportMatrix(Ncoeff, "Ncoeff", "python", "./ITHACAoutput/Matrices/");

334  SAMPLES.resize(Nnutmodes);

338  {
339  SAMPLES[i] = new SPLINTER::DataTable(1,1);
340  for(int j = 0; j < Ncoeff.cols(); j++)
341  {
342  SAMPLES[i]->addSample(mu.row(j),Ncoeff(i,j));

344

346  std::cout << "Constructing RadialBasisFunction for mode " << i+1 << std::endl;
347  }
