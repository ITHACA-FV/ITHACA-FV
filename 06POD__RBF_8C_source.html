158 int main(int argc, char* argv[])

160  //return 0;
161  tutorial06 example(argc, argv);
162  word par_offline("./par_offline"); // the offline samples.
163  word par_new("./par_online"); // the samples which will be used in the online stage for checking the reduced order model.
164  Eigen::MatrixXd par_online = ITHACAstream::readMatrix(par_new);

168  example.inletIndex(0, 1) = 0;

172  int NmodesU = para.ITHACAdict->lookupOrDefault<int>("NmodesU", 5);
173  int NmodesP = para.ITHACAdict->lookupOrDefault<int>("NmodesP", 5);
174  int NmodesSUP = para.ITHACAdict->lookupOrDefault<int>("NmodesSUP", 5);
175  int NmodesNUT = para.ITHACAdict->lookupOrDefault<int>("NmodesNUT", 5);
176  int NmodesProject = para.ITHACAdict->lookupOrDefault<int>("NmodesProject", 5);
177  int NmodesMatrixRec = para.ITHACAdict->lookupOrDefault<int>("NmodesMatrixRec",
178  5);
179  example.offlineSolve();

181  example.solvesupremizer();
182  //example.liftSolve();
183  ITHACAstream::read_fields(example.liftfield, example.U, "./lift/");
184  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
185  ITHACAstream::exportFields(example.Uomfield, "./ITHACAoutput/Offline",
186  "Uofield");
187  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example.podex,
188  example.supex, 0, NmodesProject);
189  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example.podex,
190  example.supex, 0, NmodesProject);
191  ITHACAPOD::getModes(example.nutFields, example.nuTmodes, example.podex,
192  example.supex, 0, NmodesProject);
193  ITHACAPOD::getModes(example.supfield, example.supmodes, example.podex,
194  example.supex, 1, NmodesProject);
195  PtrList<volVectorField> rec_field;
196  PtrList<volVectorField> lift_and_modes;
197  PtrList<volScalarField> rec_field_p;
198  PtrList<volScalarField> rec_field_nu;
199  Eigen::MatrixXd Mass_matrix_P = ITHACAutilities::get_mass_matrix(
200  example.Pmodes);
201  Eigen::MatrixXd Mass_matrix_nut = ITHACAutilities::get_mass_matrix(
202  example.nuTmodes);
203  ITHACAstream::exportMatrix(Mass_matrix_nut, "Mass_matrix_nut", "python",
204  "./ITHACAoutput/Matrices_L2/");
205  ITHACAstream::exportMatrix(Mass_matrix_P, "Mass_matrix_P", "python",
206  "./ITHACAoutput/Matrices_L2/");
207  example.projectSUP("./Matrices", NmodesU, NmodesP, NmodesSUP,
208  NmodesNUT); // Read the matrices if they already exist and if not compute them, you have to delete the folder Matrices if you want to re-do the computations with different number of modes
209  Info << "after projectSUP" << endl;
210  reducedSteadyNSturb pod_rbf(
211  example); //pod_rbf is an object of the class that used RBF in the online stage and thus takes into account the eddy viscosity in that stage
212  Info << "after pod_rbf" << endl;
213  pod_rbf.nu = 1e-3;
214 
215  for (label k = 0; k < par_online.rows(); k++)
216  {
217  Eigen::MatrixXd vel_now(2, 1);
218  vel_now(0, 0) = par_online(k, 0);
219  vel_now(1, 0) = par_online(k, 1);
220  pod_rbf.solveOnline_sup(vel_now);
221  Eigen::MatrixXd tmp_sol(pod_rbf.y.rows() + 1, 1);
222  tmp_sol(0) = k + 1;
223  tmp_sol.col(0).tail(pod_rbf.y.rows()) = pod_rbf.y;
224  pod_rbf.online_solution.append(tmp_sol);
225  }
226 
227  ITHACAstream::exportMatrix(pod_rbf.online_solution, "red_coeff", "python",
228  "./ITHACAoutput/red_coeff");
229  ITHACAstream::exportMatrix(pod_rbf.online_solution, "red_coeff", "matlab",
230  "./ITHACAoutput/red_coeff");

232  "./ITHACAoutput/red_coeff");
233  pod_rbf.reconstruct_sup("./ITHACAoutput/Reconstruction/");
234  ITHACAstream::exportFields(pod_rbf.nutREC, "./ITHACAoutput/Reconstruction",
235  "nuRec");
236  reducedSteadyNS pod_normal(
237  example); //pod_normal is an object of the classical POD model which does not use RBF in the online stage, here we want to make a comparison between the results of the two models
238  pod_normal.nu = 1e-3;
239 
240  for (label k = 0; k < par_online.rows(); k++)
241  {
242  Eigen::MatrixXd vel_now(2, 1);
243  vel_now(0, 0) = par_online(k, 0);
244  vel_now(1, 0) = par_online(k, 1);
245  pod_normal.solveOnline_sup(vel_now);

247  tmp_sol(0) = k + 1;

249  pod_normal.online_solution.append(tmp_sol);
250  }
251 
252  ITHACAstream::exportMatrix(pod_normal.online_solution, "red_coeffnew", "python",
253  "./ITHACAoutput/red_coeffnew");
254  ITHACAstream::exportMatrix(pod_normal.online_solution, "red_coeffnew", "matlab",
255  "./ITHACAoutput/red_coeffnew");
256  ITHACAstream::exportMatrix(pod_normal.online_solution, "red_coeffnew", "eigen",
257  "./ITHACAoutput/red_coeffnew");
258  pod_normal.reconstruct_sup("./ITHACAoutput/Lam_Rec/");
259  example.offlineSolve(par_online, "./ITHACAoutput/high_fidelity_online/");
260  exit(0);
