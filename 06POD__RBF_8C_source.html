152 int main(int argc, char* argv[])
153 {
154  // Construct the tutorial06 object
155  tutorial06 example(argc, argv);
156  // Read parameters samples for the offline stage and the ones for the online stages
157  word par_offline("./par_offline");
158  word par_new("./par_online");
159  Eigen::MatrixXd par_online = ITHACAstream::readMatrix(par_new);
160  example.mu = ITHACAstream::readMatrix(par_offline);
161  // Set the inlet boundaries where we have parameterized boundary conditions
162  example.inletIndex.resize(2, 2);
163  example.inletIndex(0, 0) = 0;
164  example.inletIndex(0, 1) = 0;
165  example.inletIndex(1, 0) = 0;
166  example.inletIndex(1, 1) = 1;
167  ITHACAparameters* para = ITHACAparameters::getInstance(example._mesh(),
168  example._runTime());
169  // Read parameters from ITHACAdict file
170  int NmodesU = para->ITHACAdict->lookupOrDefault<int>("NmodesU", 5);
171  int NmodesP = para->ITHACAdict->lookupOrDefault<int>("NmodesP", 5);
172  int NmodesSUP = para->ITHACAdict->lookupOrDefault<int>("NmodesSUP", 5);
173  int NmodesNUT = para->ITHACAdict->lookupOrDefault<int>("NmodesNUT", 5);
174  int NmodesProject = para->ITHACAdict->lookupOrDefault<int>("NmodesProject", 5);
175  // Perform The Offline Solve;
176  example.offlineSolve();
177  // Read the lift functions
178  ITHACAstream::read_fields(example.liftfield, example.U, "./lift/");
179  // Create the homogeneous set of snapshots for the velocity field
180  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
181  // Export the homogeneous velocity snapshots
182  ITHACAstream::exportFields(example.Uomfield, "./ITHACAoutput/Offline",
183  "Uofield");
184  // Perform a POD decomposition for the velocity, the pressure and the eddy viscosity
185  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example._U().name(),
186  example.podex,
187  example.supex, 0, NmodesProject);
188  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example._U().name(),
189  example.podex,
190  example.supex, 0, NmodesProject);
191  ITHACAPOD::getModes(example.nutFields, example.nutModes, example._nut().name(),
192  example.podex,
193  example.supex, 0, NmodesProject);
194  // Solve the supremizer problem based on the pressure modes
195  example.solvesupremizer("modes");
196  // Compute the reduced order matrices
197  example.projectSUP("./Matrices", NmodesU, NmodesP, NmodesSUP,
198  NmodesNUT);
199  // Create an object of the turbulent class
200  ReducedSteadyNSTurb pod_rbf(
201  example);
202  // Set value of the reduced viscosity and the penalty factor
203  pod_rbf.nu = 1e-3;
204  pod_rbf.tauU.resize(2, 1);
205  // We create the matrix rbfCoeff which will store the values of the interpolation results for the eddy viscosity field
206  Eigen::MatrixXd rbfCoeff;
207  rbfCoeff.resize(NmodesNUT, par_online.rows());
208 
209  // Perform an online solve for the new values of inlet velocities
210  for (label k = 0; k < par_online.rows(); k++)
211  {
212  Eigen::MatrixXd velNow(2, 1);
213  velNow(0, 0) = par_online(k, 0);
214  velNow(1, 0) = par_online(k, 1);
215  pod_rbf.tauU(0, 0) = 0;
216  pod_rbf.tauU(1, 0) = 0;
217  pod_rbf.solveOnlineSUP(velNow);
218  rbfCoeff.col(k) = pod_rbf.rbfCoeff;
219  Eigen::MatrixXd tmp_sol(pod_rbf.y.rows() + 1, 1);
220  tmp_sol(0) = k + 1;
221  tmp_sol.col(0).tail(pod_rbf.y.rows()) = pod_rbf.y;
222  pod_rbf.online_solution.append(tmp_sol);
223  }
224 
225  // Save the matrix of interpolated eddy viscosity coefficients
226  ITHACAstream::exportMatrix(rbfCoeff, "rbfCoeff", "python",
227  "./ITHACAoutput/Matrices/");
228  ITHACAstream::exportMatrix(rbfCoeff, "rbfCoeff", "matlab",
229  "./ITHACAoutput/Matrices/");
230  // Save the online solution
231  ITHACAstream::exportMatrix(pod_rbf.online_solution, "red_coeff", "python",
232  "./ITHACAoutput/red_coeff");
233  ITHACAstream::exportMatrix(pod_rbf.online_solution, "red_coeff", "matlab",
234  "./ITHACAoutput/red_coeff");
235  ITHACAstream::exportMatrix(pod_rbf.online_solution, "red_coeff", "eigen",
236  "./ITHACAoutput/red_coeff");
237  // Reconstruct and export the solution
238  pod_rbf.reconstructSUP("./ITHACAoutput/Reconstruction/");
239  // Save the reduced eddy viscosity fields
240  ITHACAstream::exportFields(pod_rbf.nutRec, "./ITHACAoutput/Reconstruction",
241  "nutRec");
242  // Create an object of the laminar class
243  reducedSteadyNS pod_normal(
244  example);
245  // Set value of the reduced viscosity and the penalty factor
246  pod_normal.nu = 1e-3;
247  pod_normal.tauU.resize(2, 1);
248 
249  // Perform an online solve for the new values of inlet velocities
250  for (label k = 0; k < par_online.rows(); k++)
251  {
252  Eigen::MatrixXd vel_now(2, 1);
253  vel_now(0, 0) = par_online(k, 0);
254  vel_now(1, 0) = par_online(k, 1);
255  pod_normal.tauU(0, 0) = 0;
256  pod_normal.tauU(1, 0) = 0;
257  pod_normal.solveOnline_sup(vel_now);
258  Eigen::MatrixXd tmp_sol(pod_normal.y.rows() + 1, 1);
259  tmp_sol(0) = k + 1;
260  tmp_sol.col(0).tail(pod_normal.y.rows()) = pod_normal.y;
261  pod_normal.online_solution.append(tmp_sol);
262  }
263 
264  // Save the online solution
265  ITHACAstream::exportMatrix(pod_normal.online_solution, "red_coeffnew", "python",
266  "./ITHACAoutput/red_coeffnew");
267  ITHACAstream::exportMatrix(pod_normal.online_solution, "red_coeffnew", "matlab",
268  "./ITHACAoutput/red_coeffnew");
269  ITHACAstream::exportMatrix(pod_normal.online_solution, "red_coeffnew", "eigen",
270  "./ITHACAoutput/red_coeffnew");
271  // Reconstruct and export the solution
272  pod_normal.reconstruct_sup("./ITHACAoutput/Lam_Rec/");
273  // Solve the full order problem for the online velocity values for the purpose of comparison
274  // if (ITHACAutilities::check_folder("./ITHACAoutput/Offline_check") == false)
275  // {
276  // example.offlineSolve(par_online, "./ITHACAoutput/Offline_check/");
277  // ITHACAutilities::createSymLink("./ITHACAoutput/Offline_check");
278  // }
279  exit(0);
280 }
