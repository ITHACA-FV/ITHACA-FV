66 int newton_steadyNS::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
67 {
68  Eigen::VectorXd a_tmp(Nphi_u);
69  Eigen::VectorXd b_tmp(Nphi_p);
70  a_tmp = x.head(Nphi_u);
71  b_tmp = x.tail(Nphi_p);
72  // Convective term
73  Eigen::MatrixXd cc(1, 1);
74  // Mom Term
75  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
76  // Gradient of pressure
77  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
78  // Pressure Term
79  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
80 
81  for (label i = 0; i < Nphi_u; i++)
82  {
83  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp;
84  fvec(i) = M1(i) - cc(0, 0) - M2(i);
85  }
86  for (label j = 0; j < Nphi_p; j++)
87  {
88  label k = j + Nphi_u;
89  fvec(k) = M3(j);
90  }
91  for (label j = 0; j < N_BC; j++)
92  {
93  fvec(j) = x(j) - BC(j);
94  }
95  return 0;
96 }

115 void reducedSteadyNS::solveOnline_sup(Eigen::MatrixXd vel_now)
116 {
117 
118  y.resize(Nphi_u + Nphi_p, 1);
119  y.setZero();
120 
121  for (label j = 0; j < N_BC; j++)
122  {
123  y(j) = vel_now(j, 0);
124  }
125  Color::Modifier red(Color::FG_RED);
126  Color::Modifier green(Color::FG_GREEN);
127  Color::Modifier def(Color::FG_DEFAULT);
128  Eigen::HybridNonLinearSolver<newton_steadyNS> hnls(newton_object);
129  newton_object.BC.resize(N_BC);
130  for (label j = 0; j < N_BC; j++)
131  {
132  newton_object.BC(j) = vel_now(j, 0);
133  }
134  newton_object.nu = nu;
135  hnls.solve(y);
136 
137  Eigen::VectorXd res(y);
138  newton_object.operator()(y, res);
139 
140  std::cout << "################## Online solve N° " << count_online_solve << " ##################" << std::endl;
141  std::cout << "Solving for the parameter: " << vel_now << std::endl;
142  if (res.norm() < 1e-5)
143  {
144  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
145  }
146  else
147  {
148  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
149  }
150  count_online_solve += 1;
151 }

219 double reducedSteadyNS::inf_sup_constant()
220 {
221  double a;
222  Eigen::VectorXd sup(Nphi_u);
223  Eigen::VectorXd inf(Nphi_p);
224  for (int i = 0; i < Nphi_p; i++)
225  {
226  for (int j = 0; j < Nphi_u; j++)
227  {
228  sup(j) = fvc::domainIntegrate(fvc::div(Umodes[j]) * Pmodes[i]).value() / ITHACAutilities::H1seminorm(Umodes[j]) / ITHACAutilities::L2norm(Pmodes[i]);
229  }
230  inf(i) = sup.maxCoeff();
231  }
232  a = inf.minCoeff();
233  return a;
234 }
