82 int newton_steadyNS::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
83 {
84  Eigen::VectorXd a_tmp(Nphi_u);
85  Eigen::VectorXd b_tmp(Nphi_p);
86  a_tmp = x.head(Nphi_u);
87  b_tmp = x.tail(Nphi_p);
88  // Convective term
89  Eigen::MatrixXd cc(1, 1);
90  // Mom Term
91  Eigen::VectorXd M1 = B_matrix * a_tmp *nu;
92  // Gradient of pressure
93  Eigen::VectorXd M2 = K_matrix * b_tmp;
94  // Pressure Term
95  Eigen::VectorXd M3 = P_matrix * a_tmp;
96 
97  for (label i = 0; i < Nphi_u; i++)
98  {
99  cc = a_tmp.transpose() * C_matrix[i] * a_tmp;
100  fvec(i) = M1(i) - cc(0, 0) - M2(i);
101  }
102  for (label j = 0; j < Nphi_p; j++)
103  {
104  label k = j + Nphi_u;
105  fvec(k) = M3(j);
106  }
107  for (label j = 0; j < N_BC; j++)
108  {
109  fvec(j) = x(j) - BC(j);
110  }
111  return 0;
112 }

271 {

274  Eigen::VectorXd inf(Nphi_p);
275 
276 
277  for (int i = 0; i < Nphi_p; i++)
