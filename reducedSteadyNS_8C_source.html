83 int newton_steadyNS::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
84 {
85  Eigen::VectorXd a_tmp(Nphi_u);
86  Eigen::VectorXd b_tmp(Nphi_p);
87  a_tmp = x.head(Nphi_u);
88  b_tmp = x.tail(Nphi_p);
89  // Convective term
90  Eigen::MatrixXd cc(1, 1);
91  // Mom Term
92  Eigen::VectorXd M1 = B_matrix * a_tmp *nu;
93  // Gradient of pressure
94  Eigen::VectorXd M2 = K_matrix * b_tmp;
95  // Pressure Term
96  Eigen::VectorXd M3 = P_matrix * a_tmp;
97 
98  for (label i = 0; i < Nphi_u; i++)
99  {
100  cc = a_tmp.transpose() * C_matrix[i] * a_tmp;
101  fvec(i) = M1(i) - cc(0, 0) - M2(i);
102  }
103  for (label j = 0; j < Nphi_p; j++)
104  {
105  label k = j + Nphi_u;
106  fvec(k) = M3(j);
107  }
108  for (label j = 0; j < N_BC; j++)
109  {
110  fvec(j) = x(j) - BC(j);
111  }
112  return 0;
113 }

152 
153  for (label j = 0; j < N_BC; j++)

162  newton_object.BC.resize(N_BC);
163 
164  for (label j = 0; j < N_BC; j++)
165  {
166  newton_object.BC(j) = vel_now(j, 0);
167  }
168 
169  newton_object.nu = nu;
170 
171  hnls.solve(y);
172  //lm.minimize(y);
173  Eigen::VectorXd res(y);
174  newton_object.operator()(y, res);
175 
176  std::cout << "################## Online solve N° " << count_online_solve << " ##################" << std::endl;
177  std::cout << "Solving for the parameter: " << vel_now << std::endl;
178  if (res.norm() < 1e-5)
179  {
180  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
181  }
182  else
183  {
184  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;

271 double reducedSteadyNS::inf_sup_constant()
272 {
273  double a;
274  Eigen::VectorXd sup(Nphi_u);
275  Eigen::VectorXd inf(Nphi_p);
276 
277 
278  for (int i = 0; i < Nphi_p; i++)
279  {
280  for (int j = 0; j < Nphi_u; j++)
281  {
282  sup(j) = fvc::domainIntegrate(fvc::div(Umodes[j])*Pmodes[i]).value()/ITHACAutilities::H1seminorm(Umodes[j])/ITHACAutilities::L2norm(Pmodes[i]);
283  }
284  inf(i) = sup.maxCoeff();
285  }
