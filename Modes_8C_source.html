38 List<Eigen::MatrixXd> Modes<T>::toEigen()
39 {
40  NBC = 0;
41 
42  for (int i = 0; i < (this->first()).boundaryFieldRef().size(); i++)
43  {
44  if ((this->first()).boundaryFieldRef()[i].type() != "processor")
45  {
46  NBC++;
47  }
48  }
49 
50  EigenModes.resize(NBC + 1);
51  EigenModes[0] = Foam2Eigen::PtrList2Eigen(this->toPtrList());
52  List<Eigen::MatrixXd> BC = Foam2Eigen::PtrList2EigenBC(this->toPtrList());
53 
54  for (int i = 0; i < NBC; i++)

56  EigenModes[i + 1] = BC[i];
57  }

62 template<class T>
63 List<Eigen::MatrixXd> Modes<T>::project(fvMatrix<T>& Af, int numberOfModes)
64 {
65  List<Eigen::MatrixXd> LinSys;
66  LinSys.resize(2);
67 
68  if (EigenModes.size() == 0)
69  {
70  toEigen();
71  }
72 
73  Eigen::SparseMatrix<double> Ae;
74  Eigen::VectorXd be;
75  Foam2Eigen::fvMatrix2Eigen(Af, Ae, be);
76 
77  if (numberOfModes == 0)
78  {
79  LinSys[0] = EigenModes[0].transpose() * Ae * EigenModes[0];
80  LinSys[1] = EigenModes[0].transpose() * be;
81  }
82  else
83  {
84  M_Assert(numberOfModes <= EigenModes[0].cols(),
85  "Number of required modes for projection is higher then the number of available ones");
86  LinSys[0] = ((EigenModes[0]).leftCols(numberOfModes)).transpose() * Ae *
87  (EigenModes[0]).leftCols(numberOfModes);
88  LinSys[1] = ((EigenModes[0]).leftCols(numberOfModes)).transpose() * be;
89  }
90 
91  return LinSys;
92 }

203 template<class T>
204 PtrList<GeometricField<T, fvPatchField, volMesh>>
205  Modes<T>::projectSnapshots(
206  PtrList<GeometricField<T, fvPatchField, volMesh>> snapshots,
207  PtrList<volScalarField> Volumes,
208  int numberOfModes,
209  word innerProduct)
210 {
211  if (EigenModes.size() == 0)
212  {
213  toEigen();
214  }
215 
216  M_Assert(snapshots.size() == Volumes.size(),
217  "The number of snapshots and the number of volumes vectors must be equal");
218  M_Assert(numberOfModes <= this->size(),
219  "The number of Modes used for the projection cannot be bigger than the number of available modes");
220  M_Assert(innerProduct == "L2" || innerProduct == "Frobenius",
221  "The chosen inner product is not implemented");
222  int dim = std::nearbyint(EigenModes[0].rows() /
223  Volumes[0].size()); //Checking if volumes and modes have the same size that means check if the problem is vector or scalar
224  Eigen::MatrixXd totVolumes(Volumes[0].size()*dim, Volumes.size());
225 
226  for (int i = 0; i < Volumes.size(); i++)
227  {
228  totVolumes.col(i) = Foam2Eigen::field2Eigen(Volumes[i]);
229  }
230 
231  totVolumes.replicate(dim, 1);
232  Eigen::MatrixXd Modes;
233 
234  if (numberOfModes == 0)
235  {
236  Modes = EigenModes[0];
237  }
238  else
239  {
240  Modes = EigenModes[0].leftCols(numberOfModes);
241  }
242 
243  Eigen::MatrixXd M;
244  PtrList<GeometricField<T, fvPatchField, volMesh>> projSnap;
245  Eigen::MatrixXd projSnapI;
246  Eigen::MatrixXd projSnapCoeff;
247 
248  for (int i = 0; i < snapshots.size(); i++)
249  {
250  Eigen::MatrixXd F_eigen = Foam2Eigen::field2Eigen(snapshots[i]);
251 
252  if (innerProduct == "L2")
253  {
254  M = Modes.transpose() * (totVolumes.col(i)).asDiagonal() * Modes;
255  projSnapI = Modes.transpose() * (totVolumes.col(i)).asDiagonal() * F_eigen;
256  }
257  else //Frobenius
258  {
259  M = Modes.transpose() * Modes;
260  projSnapI = Modes.transpose() * F_eigen;
261  }
262 
263  projSnapCoeff = M.fullPivLu().solve(projSnapI);
264  projSnap.append(reconstruct(projSnapCoeff, "projSnap"));
265  }
266 
267  return projSnap;
268 }

283 PtrList<GeometricField<T, fvPatchField, volMesh>>
284  Modes<T>::projectSnapshots(
285  PtrList<GeometricField<T, fvPatchField, volMesh>> snapshots,
286  int numberOfModes,
287  word innerProduct)
288 {
289  M_Assert(numberOfModes <= this->size(),
290  "The number of Modes used for the projection cannot be bigger than the number of available modes");
291  M_Assert(innerProduct == "L2" || innerProduct == "Frobenius",
292  "The chosen inner product is not implemented");
293  Eigen::MatrixXd Modes;
294 
295  if (numberOfModes == 0)
296  {
297  Modes = EigenModes[0];
298  }
299  else
300  {
301  Modes = EigenModes[0].leftCols(numberOfModes);
302  }
303 
304  if (EigenModes.size() == 0)
305  {
306  toEigen();
307  }
308 
309  Eigen::MatrixXd M_vol;
310  Eigen::MatrixXd M;
311  PtrList<GeometricField<T, fvPatchField, volMesh>> projSnap;
312  Eigen::MatrixXd projSnapI;
313  Eigen::MatrixXd projSnapCoeff;
314 
315  for (int i = 0; i < snapshots.size(); i++)
316  {
317  Eigen::MatrixXd F_eigen = Foam2Eigen::field2Eigen(snapshots[i]);
318 
319  if (innerProduct == "L2")
320  {
321  M_vol = ITHACAutilities::getMassMatrixFV(snapshots[i]);
322  }
323  else if (innerProduct == "Frobenius")
324  {
325  M_vol = Eigen::VectorXd::Identity(F_eigen.rows(), 1);
326  }
327  else
328  {
329  std::cout << "Inner product not defined" << endl;
330  exit(0);
331  }
332 
333  M = Modes.transpose() * M_vol.asDiagonal() * Modes;
334  projSnapI = Modes.transpose() * M_vol.asDiagonal() * F_eigen;
335  projSnapCoeff = M.fullPivLu().solve(projSnapI);
336  projSnap.append(reconstruct(projSnapCoeff, "projSnap"));
337  }
338 
339  return projSnap;
340 }
