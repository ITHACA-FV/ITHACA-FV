62 template<class Type, template<class> class PatchField, class GeoMesh>
63 List<Eigen::MatrixXd> Modes<Type, PatchField, GeoMesh>::project(
64  fvMatrix<Type>& Af, label numberOfModes,
65  word projType)
66 {
67  M_Assert(projType == "G" || projType == "PG",
68  "Projection type can be G for Galerking or PG for Petrov-Galerkin");
69  List<Eigen::MatrixXd> LinSys;
70  LinSys.resize(2);
71 
72  if (EigenModes.size() == 0)
73  {
74  toEigen();
75  }
76 
77  Eigen::SparseMatrix<double> Ae;
78  Eigen::VectorXd be;
79  Foam2Eigen::fvMatrix2Eigen(Af, Ae, be);
80 
81  if (numberOfModes == 0)
82  {
83  if (projType == "G")
84  {
85  LinSys[0] = EigenModes[0].transpose() * Ae * EigenModes[0];
86  LinSys[1] = EigenModes[0].transpose() * be;
87  }
88 
89  if (projType == "PG")
90  {
91  LinSys[0] = (Ae * EigenModes[0]).transpose() * Ae * EigenModes[0];
92  LinSys[1] = (Ae * EigenModes[0]).transpose() * be;
93  }
94  }
95  else
96  {
97  M_Assert(numberOfModes <= EigenModes[0].cols(),
98  "Number of required modes for projection is higher then the number of available ones");
99 
100  if (projType == "G")
101  {
102  LinSys[0] = ((EigenModes[0]).leftCols(numberOfModes)).transpose() * Ae *
103  (EigenModes[0]).leftCols(numberOfModes);
104  LinSys[1] = ((EigenModes[0]).leftCols(numberOfModes)).transpose() * be;
105  }
106 
107  if (projType == "PG")
108  {
109  LinSys[0] = (Ae * ((EigenModes[0]).leftCols(numberOfModes))).transpose() * Ae *
110  (EigenModes[0]).leftCols(numberOfModes);
111  LinSys[1] = (Ae * ((EigenModes[0]).leftCols(numberOfModes))).transpose() * be;
112  }
113  }
114 
115  return LinSys;
116 }

120  GeometricField<Type, PatchField, GeoMesh>&
121  field, label numberOfModes, word projType, fvMatrix<Type>* Af)
122 {

126  auto vol = ITHACAutilities::getMassMatrixFV(field);
127  Eigen::MatrixXd projField;
128 
129  if (EigenModes.size() == 0)
130  {
131  toEigen();
132  }
133 
134  if (numberOfModes == 0)
135  {

137  {

139  }

143  "Using a Petrov-Galerkin projection you have to provide also the system matrix");
144  Eigen::SparseMatrix<double> Ae;
145  Eigen::VectorXd be;
146  Foam2Eigen::fvMatrix2Eigen(*Af, Ae, be);
147  projField = (Ae * EigenModes[0]).transpose() * vol.asDiagonal() * fieldEig;
148  }
149  }
150  else
151  {
152  M_Assert(numberOfModes <= EigenModes[0].cols(),
153  "Number of required modes for projection is higher then the number of available ones");
154 
155  if (projType == "G")
156  {
157  projField = ((EigenModes[0]).leftCols(numberOfModes)).transpose() *
158  vol.asDiagonal() * fieldEig;
159  }
160  else if (projType == "PG")
161  {
162  M_Assert(Af != NULL,
163  "Using a Petrov-Galerkin projection you have to provide also the system matrix");
164  Eigen::SparseMatrix<double> Ae;
165  Eigen::VectorXd be;
166  Foam2Eigen::fvMatrix2Eigen(*Af, Ae, be);
167  projField = (Ae * ((EigenModes[0]).leftCols(numberOfModes))).transpose() *
168  vol.asDiagonal() * fieldEig;
169  }
170  }
171 
172  return projField;

246 Modes<Type, PatchField, GeoMesh>::reconstruct(
247  GeometricField<Type, PatchField, GeoMesh>& inputField,
248  Eigen::MatrixXd Coeff,
249  word Name)
250 {
251  if (EigenModes.size() == 0)
252  {
253  toEigen();
254  }
255 
256  label Nmodes = Coeff.rows();
257  Eigen::VectorXd InField = EigenModes[0].leftCols(Nmodes) * Coeff;
258 
259  if (inputField.name() == "nut")
260  {
261  InField = (InField.array() < 0).select(0, InField);
262  }
263 
264  inputField = Foam2Eigen::Eigen2field(inputField, InField);
265  inputField.rename(Name);
266 
267  for (label i = 0; i < NBC; i++)
268  {
269  Eigen::VectorXd BF = EigenModes[i + 1].leftCols(Nmodes) * Coeff;
270 
271  if (inputField.name() == "nut")
272  {
273  BF = (BF.array() < 0).select(0, BF);
274  }
275 
276  ITHACAutilities::assignBC(inputField, i, BF);
277  }
278 
279  return inputField;
280 }

301 template<class Type, template<class> class PatchField, class GeoMesh>
302 void Modes<Type, PatchField, GeoMesh>::projectSnapshots(
303  PtrList<GeometricField<Type, PatchField, GeoMesh>> snapshots,
304  PtrList<GeometricField<Type, PatchField, GeoMesh>>& projSnapshots,
305  PtrList<volScalarField> Volumes,
306  label numberOfModes,
307  word innerProduct)
308 {
309  if (EigenModes.size() == 0)
310  {
311  toEigen();
312  }
313 
314  M_Assert(snapshots.size() == Volumes.size(),
315  "The number of snapshots and the number of volumes vectors must be equal");
316  M_Assert(numberOfModes <= this->size(),
317  "The number of Modes used for the projection cannot be bigger than the number of available modes");
318  M_Assert(innerProduct == "L2" || innerProduct == "Frobenius",
319  "The chosen inner product is not implemented");
320  projSnapshots.resize(snapshots.size());
321  label dim = std::nearbyint(EigenModes[0].rows() /
322  Volumes[0].size()); //Checking if volumes and modes have the same size that means check if the problem is vector or scalar
323  Eigen::MatrixXd totVolumes(Volumes[0].size()*dim, Volumes.size());
324 
325  for (label i = 0; i < Volumes.size(); i++)
326  {
327  totVolumes.col(i) = Foam2Eigen::field2Eigen(Volumes[i]);
328  }
329 
330  totVolumes.replicate(dim, 1);
331  Eigen::MatrixXd Modes;
332 
333  if (numberOfModes == 0)
334  {
335  Modes = EigenModes[0];
336  }
337  else
338  {
339  Modes = EigenModes[0].leftCols(numberOfModes);
340  }
341 
342  Eigen::MatrixXd M;
343  Eigen::MatrixXd projSnapI;
344  Eigen::MatrixXd projSnapCoeff;
345 
346  for (label i = 0; i < snapshots.size(); i++)
347  {
348  GeometricField<Type, PatchField, GeoMesh> Fr = snapshots[0];
349  Eigen::MatrixXd F_eigen = Foam2Eigen::field2Eigen(snapshots[i]);
350 
351  if (innerProduct == "L2")
352  {
353  M = Modes.transpose() * (totVolumes.col(i)).asDiagonal() * Modes;
354  projSnapI = Modes.transpose() * (totVolumes.col(i)).asDiagonal() * F_eigen;
355  }
356  else //Frobenius
357  {
358  M = Modes.transpose() * Modes;
359  projSnapI = Modes.transpose() * F_eigen;
360  }
361 
362  projSnapCoeff = M.fullPivLu().solve(projSnapI);
363  reconstruct(Fr, projSnapCoeff, "projSnap");
364  projSnapshots.set(i, Fr.clone());
365  }
366 }

409  Eigen::MatrixXd projSnapCoeff;
