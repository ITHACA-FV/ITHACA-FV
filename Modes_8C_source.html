38 List<Eigen::MatrixXd> Modes<T>::toEigen()
39 {
40  NBC = 0;
41 
42  for (int i = 0; i < (this->first()).boundaryFieldRef().size(); i++)
43  {
44  if ((this->first()).boundaryFieldRef()[i].type() != "processor")
45  {
46  NBC++;
47  }
48  }
49 
50  EigenModes.resize(NBC + 1);

52  List<Eigen::MatrixXd> BC = Foam2Eigen::PtrList2EigenBC(this->toPtrList());
53

62 template<class T>
63 List<Eigen::MatrixXd> Modes<T>::project(fvMatrix<T>& Af, int numberOfModes)
64 {
65  List<Eigen::MatrixXd> LinSys;
66  LinSys.resize(2);
67 
68  if (EigenModes.size() == 0)
69  {
70  toEigen();
71  }
72 
73  Eigen::SparseMatrix<double> Ae;
74  Eigen::VectorXd be;
75  Foam2Eigen::fvMatrix2Eigen(Af, Ae, be);
76 
77  if (numberOfModes == 0)
78  {
79  LinSys[0] = EigenModes[0].transpose() * Ae * EigenModes[0];
80  LinSys[1] = EigenModes[0].transpose() * be;
81  }
82  else
83  {
84  M_Assert(numberOfModes <= EigenModes[0].cols(),
85  "Number of required modes for projection is higher then the number of available ones");
86  LinSys[0] = ((EigenModes[0]).leftCols(numberOfModes)).transpose() * Ae *
87  (EigenModes[0]).leftCols(numberOfModes);
88  LinSys[1] = ((EigenModes[0]).leftCols(numberOfModes)).transpose() * be;
89  }
90 
91  return LinSys;
92 }

96 GeometricField<T, fvPatchField, volMesh> Modes<T>::reconstruct(
97  Eigen::MatrixXd& Coeff,
98  word Name)
99 {
100  if (EigenModes.size() == 0)
101  {
102  toEigen();
103  }
104 
105  GeometricField<T, fvPatchField, volMesh> field(Name,
106  (this->toPtrList())[0] * 0);
107  int Nmodes = Coeff.rows();
108  Eigen::VectorXd InField = EigenModes[0].leftCols(Nmodes) * Coeff;
109  field = Foam2Eigen::Eigen2field(field, InField);
110 
111  for (int i = 0; i < NBC; i++)
112  {
113  Eigen::VectorXd BF = EigenModes[i + 1].leftCols(Nmodes) * Coeff;
114  ITHACAutilities::assignBC(field, i, BF);
115  }
116 
117  return field;

120 template<class T>
121 PtrList<GeometricField<T, fvPatchField, volMesh>>
122  Modes<T>::projectSnapshots(
123  PtrList<GeometricField<T, fvPatchField, volMesh>> snapshots,
124  PtrList<volScalarField> Volumes,
125  int numberOfModes,
126  word innerProduct)
127 {
128  if (EigenModes.size() == 0)
129  {
130  toEigen();
131  }
132 
133  M_Assert(snapshots.size() == Volumes.size(),
134  "The number of snapshots and the number of volumes vectors must be equal");
135  M_Assert(numberOfModes <= this->size(),
136  "The number of Modes used for the projection cannot be bigger than the number of available modes");
137  M_Assert(innerProduct == "L2" || innerProduct == "Frobenius",
138  "The chosen inner product is not implemented");
139  int dim = std::nearbyint(EigenModes[0].rows() /
140  Volumes[0].size()); //Checking if volumes and modes have the same size that means check if the problem is vector or scalar
141  Eigen::MatrixXd totVolumes(Volumes[0].size()*dim, Volumes.size());
142 
143  for (int i = 0; i < Volumes.size(); i++)
144  {
145  totVolumes.col(i) = Foam2Eigen::field2Eigen(Volumes[i]);
146  }
147 
148  totVolumes.replicate(dim, 1);
149  Eigen::MatrixXd Modes;
150 
151  if (numberOfModes == 0)
152  {
153  Modes = EigenModes[0];
154  }
155  else
156  {
157  Modes = EigenModes[0].leftCols(numberOfModes);
158  }
159 
160  Eigen::MatrixXd M;
161  PtrList<GeometricField<T, fvPatchField, volMesh>> projSnap;
162  Eigen::MatrixXd projSnapI;
163  Eigen::MatrixXd projSnapCoeff;
164 
165  for (int i = 0; i < snapshots.size(); i++)
166  {
167  Eigen::MatrixXd F_eigen = Foam2Eigen::field2Eigen(snapshots[i]);
168 
169  if (innerProduct == "L2")
170  {
171  M = Modes.transpose() * (totVolumes.col(i)).asDiagonal() * Modes;
172  projSnapI = Modes.transpose() * (totVolumes.col(i)).asDiagonal() * F_eigen;
173  }
174  else //Frobenius
175  {
176  M = Modes.transpose() * Modes;
177  projSnapI = Modes.transpose() * F_eigen;
178  }
179 
180  projSnapCoeff = M.fullPivLu().solve(projSnapI);
181  projSnap.append(reconstruct(projSnapCoeff, "projSnap"));
182  }
183 
184  return projSnap;
185 }

199 template<class T>
200 PtrList<GeometricField<T, fvPatchField, volMesh>>
201  Modes<T>::projectSnapshots(
202  PtrList<GeometricField<T, fvPatchField, volMesh>> snapshots,
203  int numberOfModes,
204  word innerProduct)
205 {
206  M_Assert(numberOfModes <= this->size(),
207  "The number of Modes used for the projection cannot be bigger than the number of available modes");
208  M_Assert(innerProduct == "L2" || innerProduct == "Frobenius",
209  "The chosen inner product is not implemented");
210  Eigen::MatrixXd Modes;
211 
212  if (numberOfModes == 0)
213  {
214  Modes = EigenModes[0];
215  }
216  else
217  {
218  Modes = EigenModes[0].leftCols(numberOfModes);
219  }
220 
221  if (EigenModes.size() == 0)
222  {
223  toEigen();
224  }
225 
226  Eigen::MatrixXd M_vol;
227  Eigen::MatrixXd M;
228  PtrList<GeometricField<T, fvPatchField, volMesh>> projSnap;
229  Eigen::MatrixXd projSnapI;
230  Eigen::MatrixXd projSnapCoeff;
231 
232  for (int i = 0; i < snapshots.size(); i++)
233  {
234  Eigen::MatrixXd F_eigen = Foam2Eigen::field2Eigen(snapshots[i]);
235 
236  if (innerProduct == "L2")
237  {
238  M_vol = ITHACAutilities::get_mass_matrix_FV(snapshots[i]);
239  }
240  else if (innerProduct == "Frobenius")
241  {
242  M_vol = Eigen::VectorXd::Identity(F_eigen.rows(), 1);
243  }
244  else
245  {
246  std::cout << "Inner product not defined" << endl;
247  exit(0);
248  }
249 
250  M = Modes.transpose() * M_vol.asDiagonal() * Modes;
251  projSnapI = Modes.transpose() * M_vol.asDiagonal() * F_eigen;
252  projSnapCoeff = M.fullPivLu().solve(projSnapI);
253  projSnap.append(reconstruct(projSnapCoeff, "projSnap"));
254  }
255 
256  return projSnap;
257 }
