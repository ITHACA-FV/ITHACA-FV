46reducedUnsteadyNST::reducedUnsteadyNST(unsteadyNST& FOMproblem)
47//problem(&FOMproblem)
48{
49 problem = & FOMproblem;
50 N_BC = problem->inletIndex.rows();
51 N_BC_t = problem->inletIndexT.rows();
52 Nphi_u = problem->B_matrix.rows();
53 Nphi_p = problem->K_matrix.cols();
54 Nphi_t = problem->Y_matrix.rows();
55
56 // Create locally the velocity modes
57 for (int k = 0; k < problem->liftfield.size(); k++)
58 {
59 Umodes.append((problem->liftfield[k]).clone());
60 }
61
62 for (int k = 0; k < problem->NUmodes; k++)
63 {
64 Umodes.append((problem->Umodes[k]).clone());
65 }
66
67 for (int k = 0; k < problem->NSUPmodes; k++)
68 {
69 Umodes.append((problem->supmodes[k]).clone());
70 }
71
72 // Create locally the pressure modes
73 for (int k = 0; k < problem->NPmodes; k++)
74 {
75 Pmodes.append((problem->Pmodes[k]).clone());
76 }
77
78 for (int k = 0; k < problem->liftfieldT.size(); k++)
79 {
80 Tmodes.append((problem->liftfieldT[k]).clone());
81 }
82
83 // Create locally the temperature modes
84 for (int k = 0; k < problem->NTmodes; k++)
85 {
86 Tmodes.append((problem->Tmodes[k]).clone());
87 }
88
89 // Store locally the snapshots for projections
90 for (int k = 0; k < problem->Ufield.size(); k++)
91 {
92 Usnapshots.append((problem->Ufield[k]).clone());
93 Psnapshots.append((problem->Pfield[k]).clone());
94 }
95
96 for (int k = 0; k < problem->Tfield.size(); k++)
97 {
98 Tsnapshots.append((problem->Tfield[k]).clone());
99 }
100
101 newton_object_sup = newton_unsteadyNST_sup(Nphi_u + Nphi_p, Nphi_u + Nphi_p,
102 FOMproblem);
103 newton_object_sup_t = newton_unsteadyNST_sup_t(Nphi_t, Nphi_t, FOMproblem);
104}

108//Operator to evaluate the residual for the supremizer approach
109int newton_unsteadyNST_sup::operator()(const Eigen::VectorXd& x,
110 Eigen::VectorXd& fvec) const
111{
112 Eigen::VectorXd a_dot(Nphi_u);
113 Eigen::VectorXd a_tmp(Nphi_u);
114 Eigen::VectorXd b_tmp(Nphi_p);
115 a_tmp = x.head(Nphi_u);
116 b_tmp = x.tail(Nphi_p);
117 a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
118 // Convective term
119 Eigen::MatrixXd cc(1, 1);
120 // Momentum Term
121 Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
122 // Gradient of pressure
123 Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
124 // Mass Term Velocity
125 Eigen::VectorXd M5 = problem->M_matrix * a_dot;
126 // Pressure Term
127 Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
128
129 for (int i = 0; i < Nphi_u; i++)
130 {
131 cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp;
132 fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M2(i);
133 }
134
135 for (int j = 0; j < Nphi_p; j++)
136 {
137 int k = j + Nphi_u;
138 fvec(k) = M3(j);
139 }
140
141 for (int j = 0; j < N_BC; j++)
142 {
143 fvec(j) = x(j) - BC(j);
144 }
145
146 return 0;
147}

158int newton_unsteadyNST_sup_t::operator()(const Eigen::VectorXd& t,
159 Eigen::VectorXd& fvect) const
160{
161 Eigen::VectorXd c_dot(Nphi_t);
162 Eigen::VectorXd c_tmp(Nphi_t);
163 c_tmp = t.head(Nphi_t);
164 c_dot = (t.head(Nphi_t) - z_old.head(Nphi_t)) / dt;
165 // Convective term temperature
166 Eigen::MatrixXd qq(1, 1);
167 // diffusive term temperature
168 Eigen::VectorXd M6 = problem->Y_matrix * c_tmp * DT;
169 // Mass Term Temperature
170 Eigen::VectorXd M8 = problem->MT_matrix * c_dot;
171
172 for (int i = 0; i < Nphi_t; i++)
173 {
174 qq = a_tmp.transpose() * problem-> Q_matrix[i] * c_tmp;
175 fvect(i) = -M8(i) + M6(i) - qq(0, 0);
176 }
177
178 for (int j = 0; j < N_BC_t; j++)
179 {
180 fvect(j) = t(j) - BC_t(j);
181 }
182
183 return 0;
184}

194// * * * * * * * * * * * * * * * Solve Functions * * * * * * * * * * * * * //
195void reducedUnsteadyNST::solveOnline_sup(Eigen::MatrixXd& vel_now,
196 Eigen::MatrixXd& temp_now, int startSnap)
197{
198 // Create and resize the solution vector
199 y.resize(Nphi_u + Nphi_p, 1);
200 y.setZero();
201 z.resize(Nphi_t, 1);
202 z.setZero();
203 // Set Initial Conditions
204 y.head(Nphi_u) = ITHACAutilities::getCoeffs(Usnapshots[startSnap], Umodes);
205 y.tail(Nphi_p) = ITHACAutilities::getCoeffs(Psnapshots[startSnap], Pmodes);
206 z.head(Nphi_t) = ITHACAutilities::getCoeffs(Tsnapshots[startSnap], Tmodes);
207
208 // Change initial condition for the lifting function
209 for (int j = 0; j < N_BC; j++)
210 {
211 y(j) = vel_now(j, 0);
212 }
213
214 for (int j = 0; j < N_BC_t; j++)
215 {
216 z(j) = temp_now(j, 0);
217 }
218
219 // Set some properties of the newton object
220 newton_object_sup.nu = nu;
221 newton_object_sup.y_old = y;
222 newton_object_sup.dt = dt;
223 newton_object_sup_t.DT = DT;
224 newton_object_sup_t.z_old = z;
225 newton_object_sup_t.dt = dt;
226 newton_object_sup.BC.resize(N_BC);
227 newton_object_sup_t.BC_t.resize(N_BC_t);
228
229 for (int j = 0; j < N_BC_t; j++)
230 {
231 newton_object_sup_t.BC_t(j) = temp_now(j, 0);
232 }
233
234 for (int j = 0; j < N_BC; j++)
235 {
236 newton_object_sup.BC(j) = vel_now(j, 0);
237 }
238
239 // Set number of online solutions
240 int Ntsteps = static_cast<int>((finalTime - tstart) / dt);
241 online_solution.resize(Ntsteps);
242 online_solutiont.resize(Ntsteps);
243 // Set the initial time
244 time = tstart;
245 // Counting variable
246 int counter = 0;
247 // Create vector to store temporal solution and save initial condition as first solution
248 Eigen::MatrixXd tmp_sol(Nphi_u + Nphi_p + 1, 1);
249 tmp_sol(0) = time;
250 tmp_sol.col(0).tail(y.rows()) = y;
251 online_solution[counter] = tmp_sol;
252 Eigen::MatrixXd tmp_solt(Nphi_t + 1, 1);
253 tmp_solt(0) = time;
254 tmp_solt.col(0).tail(z.rows()) = z;
255 online_solutiont[counter] = tmp_solt;
256 counter ++;
257 // Create nonlinear solver object
258 Eigen::HybridNonLinearSolver<newton_unsteadyNST_sup> hnls(newton_object_sup);
259 Eigen::HybridNonLinearSolver<newton_unsteadyNST_sup_t> hnlst(
260 newton_object_sup_t);
261 // Set output colors for fancy output
262 Color::Modifier red(Color::FG_RED);
263 Color::Modifier green(Color::FG_GREEN);
264 Color::Modifier def(Color::FG_DEFAULT);
265
266 // Start the time loop
267 while (time < finalTime)
268 {
269 time = time + dt;
270 Eigen::VectorXd res(y);
271 Eigen::VectorXd rest(z);
272 res.setZero();
273 rest.setZero();
274 hnls.solve(y);
275
276 for (int j = 0; j < N_BC; j++)
277 {
278 y(j) = vel_now(j, 0);
279 }
280
281 // set the a_temp
282 // solve for temperature
283 newton_object_sup_t.a_tmp = y.head(Nphi_u);
284 hnlst.solve(z);
285
286 for (int j = 0; j < N_BC_t; j++)
287 {
288 z(j) = temp_now(j, 0);
289 }
290
291 newton_object_sup.operator()(y, res);
292 newton_object_sup.y_old = y;
293 newton_object_sup_t.operator()(z, rest);
294 newton_object_sup_t.z_old = z;
295 std::cout << "################## Online solve N° " << count_online_solve <<
296 " ##################" << std::endl;
297 Info << "Time = " << time << endl;
298 std::cout << "Solving for the parameter: " << vel_now << std::endl;
299 std::cout << "Solving for the parameter: " << temp_now << std::endl;
300
301 if (res.norm() < 1e-5)
302 {
303 std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
304 hnls.iter << " iterations " << def << std::endl << std::endl;
305 }
306 else
307 {
308 std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
309 hnls.iter << " iterations " << def << std::endl << std::endl;
310 }
311
312 if (rest.norm() < 1e-5)
313 {
314 std::cout << green << "|F(x)| = " << rest.norm() << " - Minimun reached in " <<
315 hnlst.iter << " iterations " << def << std::endl << std::endl;
316 }
317 else
318 {
319 std::cout << red << "|F(x)| = " << rest.norm() << " - Minimun reached in " <<
320 hnlst.iter << " iterations " << def << std::endl << std::endl;
321 }
322
323 count_online_solve += 1;
324 tmp_sol(0) = time;
325 tmp_sol.col(0).tail(y.rows()) = y;
326
327 if (counter >= online_solution.size())
328 {
329 online_solution.append(tmp_sol);
330 }
331 else
332 {
333 online_solution[counter] = tmp_sol;
334 }
335
336 tmp_solt(0) = time;
337 tmp_solt.col(0).tail(z.rows()) = z;
338
339 if (counter >= online_solutiont.size())
340 {
341 online_solutiont.append(tmp_solt);
342 }
343 else
344 {
345 online_solutiont[counter] = tmp_solt;
346 }
347
348 counter ++;
349 }
350
351 // Save the solution
352 ITHACAstream::exportMatrix(online_solution, "red_coeff", "python",
353 "./ITHACAoutput/red_coeff");
354 ITHACAstream::exportMatrix(online_solution, "red_coeff", "matlab",
355 "./ITHACAoutput/red_coeff");
356 ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "python",
357 "./ITHACAoutput/red_coeff_t");
358 ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "matlab",
359 "./ITHACAoutput/red_coeff_t");
360 count_online_solve += 1;
361}

364void reducedUnsteadyNST::reconstruct_sup(fileName folder, int printevery)
365{
366 mkDir(folder);
367 ITHACAutilities::createSymLink(folder);
368 int counter = 0;
369 int nextwrite = 0;
370 int counter2 = 1;
371
372 for (int i = 0; i < online_solution.size(); i++)
373 {
374 if (counter == nextwrite)
375 {
376 volVectorField U_rec("U_rec", Umodes[0] * 0);
377
378 for (int j = 0; j < Nphi_u; j++)
379 {
380 U_rec += Umodes[j] * online_solution[i](j + 1, 0);
381 }
382
383 //problem.exportSolution(U_rec, name(online_solution[i](0, 0)), folder);
384 ITHACAstream::exportSolution(U_rec, name(counter2), folder);
385 volScalarField P_rec("P_rec", Pmodes[0] * 0);
386
387 for (int j = 0; j < Nphi_p; j++)
388 {
389 P_rec += Pmodes[j] * online_solution[i](j + Nphi_u + 1, 0);
390 }
391
392 //problem.exportSolution(P_rec, name(online_solution[i](0, 0)), folder);
393 ITHACAstream::exportSolution(P_rec, name(counter2), folder);
394 nextwrite += printevery;
395 counter2 ++;
396 UREC.append((U_rec).clone());
397 PREC.append((P_rec).clone());
398 }
399
400 counter++;
401 }
402}
