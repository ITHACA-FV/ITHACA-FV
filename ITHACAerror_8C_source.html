40 template<typename T>
41 double errorFrobRel(GeometricField<T, fvPatchField, volMesh>& field1,
42  GeometricField<T, fvPatchField, volMesh>& field2)
43 {
44  double err;
45  GeometricField<T, fvPatchField, volMesh> errField = field1 - field2;
46 
47  if (frobNorm(field1) <= 1e-6)
48  {
49  err = 0;
50  }
51  else
52  {
53  err = frobNorm(errField) / frobNorm(field1);
54  }
55 
56  return err;
57 }
