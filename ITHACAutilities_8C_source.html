441 void ITHACAutilities::setBoxToValue(volScalarField& field, Eigen::MatrixXd Box, double value)
442 {
443  for (label i = 0; i < field.internalField().size(); i++)
444  {
445  auto cx = field.mesh().C()[i].component(vector::X);
446  auto cy = field.mesh().C()[i].component(vector::Y);
447  auto cz = field.mesh().C()[i].component(vector::Z);
448  if (cx >= Box(0, 0) && cy >= Box(0, 1) && cz >= Box(0, 2) && cx <= Box(1, 0) && cy <= Box(1, 1) && cz <= Box(1, 2) )
449  {
450  field.ref()[i] = value;
451  }
452  }
453  for (label i = 0; i < field.boundaryField().size(); i++)
454  {
455  for (label j = 0; j < field.boundaryField()[i].size(); j++)
456  {
457  if (field.boundaryField()[i].type() == "fixedValue" || field.boundaryField()[i].type() == "calculated")
458  {
459  auto cx = field.mesh().C().boundaryField()[i][j][0];
460  auto cy = field.mesh().C().boundaryField()[i][j][1];
461  auto cz = field.mesh().C().boundaryField()[i][j][2];
462  if (cx >= Box(0, 0) && cy >= Box(0, 1) && cz >= Box(0, 2) && cx <= Box(1, 0) && cy <= Box(1, 1) && cz <= Box(1, 2) )
463  {
464  field.boundaryFieldRef()[i][j] = value;

466  }
