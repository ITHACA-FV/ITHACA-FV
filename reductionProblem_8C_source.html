69 void reductionProblem::genRandPar(int Tsize)
70 {
71  mu.resize(Tsize, Pnumber);
72  std::srand(std::time(0));
73  auto rand = Eigen::MatrixXd::Random(Tsize, Pnumber);
74  auto dx = mu_range.col(1) - mu_range.col(0);
75  Eigen::MatrixXd k;
76  k = (rand.array() + double(1)) / 2;
77 
78  for (int i = 0; i < Pnumber; i ++)
79  {
80  k.col(i) = (k.col(i) * dx(i)).array() + mu_range(i, 0);
81 
82  }
83  mu = k;
84 }

115 // Assign a BC for a vector field
116 void reductionProblem::assignBC(volScalarField& s, label BC_ind, double& value)
117 {
118  if (s.boundaryField()[BC_ind].type() == "fixedValue")
119  {
120  for (label i = 0; i < s.boundaryField()[BC_ind].size(); i++)
121  {
122  s.boundaryFieldRef()[BC_ind][i] = value;
123  }
124  }
125  else if (s.boundaryField()[BC_ind].type() == "fixedGradient")
126  {
127  fixedGradientFvPatchScalarField& Tpatch = refCast<fixedGradientFvPatchScalarField>(s.boundaryFieldRef()[BC_ind]);
128  scalarField& gradTpatch = Tpatch.gradient();
129  forAll(gradTpatch, faceI)
130  {
131  gradTpatch[faceI] = value;
132  }
133  }
134  else if (s.boundaryField()[BC_ind].type() == "fixedFluxPressure")
135  {
136  for (label i = 0; i < s.boundaryField()[BC_ind].size(); i++)
137  {
138  s.boundaryFieldRef()[BC_ind][i] = value;
139  }
140  }
141  else if (s.boundaryField()[BC_ind].type() == "freestream")
142  {
143  for (label i = 0; i < s.boundaryField()[BC_ind].size(); i++)
144  {
145  s.boundaryFieldRef()[BC_ind][i] = value;
146  }
147  freestreamFvPatchField<scalar>& Tpatch = refCast<freestreamFvPatchField<scalar> >(s.boundaryFieldRef()[BC_ind]);
148  scalarField& gradTpatch = Tpatch.freestreamValue();
149  forAll(gradTpatch, faceI)
150  {
151  gradTpatch[faceI] = value;
152  }
153 
154  }
155  else if (s.boundaryField()[BC_ind].type() == "calculated")
156  {
157  for (label i = 0; i < s.boundaryField()[BC_ind].size(); i++)
158  {
159  s.boundaryFieldRef()[BC_ind][i] = value;
160  }
161  }
162  else if (s.boundaryField()[BC_ind].type() == "empty")
163  {
164 
165  }
166 }
