161 // Method to compute the lifting function
162 void steadyNS::liftSolve()
163 {
164  for (label k = 0; k < inletIndex.rows(); k++)
165  {
166  Time& runTime = _runTime();
167  surfaceScalarField& phi = _phi();
168  fvMesh& mesh = _mesh();
169  fv::options& fvOptions = _fvOptions();
170  volScalarField p = _p();
171  volVectorField U = _U();
172  IOMRFZoneList& MRF = _MRF();
173  label BCind = inletIndex(k, 0);
174  volVectorField Ulift("Ulift" + name(k), U);
175  instantList Times = runTime.times();
176  runTime.setTime(Times[1], 1);
177 
178  pisoControl potentialFlow(mesh, "potentialFlow");
179 
180  Info << "Solving a lifting Problem" << endl;
181 
182  Vector<double> v1(0, 0, 0);
183  v1[inletIndex(k, 1)] = 1;
184  Vector<double> v0(0, 0, 0);
185 
186  for (label j = 0; j < U.boundaryField().size(); j++)
187  {
188  if (j == BCind)
189  {
190  assignBC(Ulift, j, v1);
191  }
192  else if (U.boundaryField()[BCind].type() == "fixedValue")
193  {
194  assignBC(Ulift, j, v0);
195  }
196  else
197  {
198 
199  }
200  assignIF(Ulift, v0);
201  phi = linearInterpolate(Ulift) & mesh.Sf();
202  }
203  Info << "Constructing velocity potential field Phi\n" << endl;
204  volScalarField Phi
205  (
206  IOobject
207  (
208  "Phi",
209  runTime.timeName(),
210  mesh,
211  IOobject::READ_IF_PRESENT,
212  IOobject::NO_WRITE
213  ),
214  mesh,
215  dimensionedScalar("Phi", dimLength * dimVelocity, 0),
216  p.boundaryField().types()
217  );
218 
219  label PhiRefCell = 0;
220  scalar PhiRefValue = 0;
221  setRefCell
222  (
223  Phi,
224  potentialFlow.dict(),
225  PhiRefCell,
226  PhiRefValue
227  );
228 
229  mesh.setFluxRequired(Phi.name());
230 
231  runTime.functionObjects().start();
232 
233  MRF.makeRelative(phi);
234  adjustPhi(phi, Ulift, p);
235 
236 
237  while (potentialFlow.correctNonOrthogonal())
238  {
239  fvScalarMatrix PhiEqn
240  (
241  fvm::laplacian(dimensionedScalar("1", dimless, 1), Phi)
242  ==
243  fvc::div(phi)
244  );
245 
246  PhiEqn.setReference(PhiRefCell, PhiRefValue);
247  PhiEqn.solve();
248 
249  if (potentialFlow.finalNonOrthogonalIter())
250  {
251  phi -= PhiEqn.flux();
252  }
253  }
254 
255  MRF.makeAbsolute(phi);
256 
257  Info << "Continuity error = "
258  << mag(fvc::div(phi))().weightedAverage(mesh.V()).value()
259  << endl;
260 
261  Ulift = fvc::reconstruct(phi);
262  Ulift.correctBoundaryConditions();
263 
264  Info << "Interpolated velocity error = "
265  << (sqrt(sum(sqr((fvc::interpolate(U) & mesh.Sf()) - phi)))
266  / sum(mesh.magSf())).value()
267  << endl;
268  Ulift.write();
269  liftfield.append(Ulift);
270  }
271 }

309 Eigen::MatrixXd steadyNS::diffusive_term(label NUmodes, label NPmodes, label NSUPmodes)
310 {
311  label Bsize = NUmodes + NSUPmodes + liftfield.size();
312  Eigen::MatrixXd B_matrix;
313  B_matrix.resize(Bsize, Bsize);
314 
315  PtrList<volVectorField> Together(0);
316  for (label k = 0; k < liftfield.size(); k++)
317  {
318  Together.append(liftfield[k]);
319  }
320  for (label k = 0; k < NUmodes; k++)
321  {
322  Together.append(Umodes[k]);
323  }
324  for (label k = 0; k < NSUPmodes; k++)
325  {
326  Together.append(supmodes[k]);
327  }
328 
329  // Project everything
330  for (label i = 0; i < Bsize; i++)
331  {
332  for (label j = 0; j < Bsize; j++)
333  {
334  B_matrix(i, j) = fvc::domainIntegrate(Together[i] & fvc::laplacian(dimensionedScalar("1", dimless, 1), Together[j])).value();
335  }
336  }
337  // Export the matrix
338  ITHACAstream::exportMatrix(B_matrix, "B", "python", "./ITHACAoutput/Matrices/");
339  ITHACAstream::exportMatrix(B_matrix, "B", "matlab", "./ITHACAoutput/Matrices/");

341  return B_matrix;
