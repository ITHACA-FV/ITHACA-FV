424 template<typename T>
425 Eigen::MatrixXd ITHACAutilities::get_mass_matrix_Eigen(PtrList<T>& fields)
426 {
427  Eigen::MatrixXd F = Foam2Eigen::PtrList2Eigen(fields);
428  Eigen::VectorXd V = Foam2Eigen::field2Eigen(fields[0].mesh().V());
429  Eigen::MatrixXd VM = V.asDiagonal();
430  Eigen::MatrixXd M = F.transpose()*VM*F;
431  return M;
