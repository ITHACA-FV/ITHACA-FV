463 template<typename T>
464 Eigen::MatrixXd ITHACAutilities::get_mass_matrix_Eigen(PtrList<T>& fields)
465 {
466  Eigen::MatrixXd F = Foam2Eigen::PtrList2Eigen(fields);
467  Eigen::VectorXd V = Foam2Eigen::field2Eigen(fields[0].mesh().V());
468  Eigen::MatrixXd VM = V.asDiagonal();
469  Eigen::MatrixXd M = F.transpose()*VM*F;
470  return M;
