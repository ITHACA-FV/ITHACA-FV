475 template<typename T>
476 Eigen::MatrixXd ITHACAutilities::get_mass_matrix_Eigen(PtrList<T>& fields)
477 {
478  Eigen::MatrixXd F = Foam2Eigen::PtrList2Eigen(fields);
479  Eigen::VectorXd V = Foam2Eigen::field2Eigen(fields[0].mesh().V());
480  Eigen::MatrixXd VM = V.asDiagonal();
481  Eigen::MatrixXd M = F.transpose() * VM * F;
482  return M;
