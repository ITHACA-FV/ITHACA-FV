62 void SteadyNSSimple::getTurbRBF(label NNutModes)
63 {
64  if (NNutModes == 0)
65  {
66  NNutModes = nutModes.size();
67  }
68 
69  coeffL2 = ITHACAutilities::getCoeffs(nutFields, nutModes, NNutModes);
70  samples.resize(NNutModes);
71  rbfSplines.resize(NNutModes);
72  Eigen::MatrixXd weights;
73 
74  for (label i = 0; i < NNutModes; i++) // i is the nnumber of th mode
75  {
76  word weightName = "wRBF_M" + name(i + 1);
77 
78  if (ITHACAutilities::check_file("./ITHACAoutput/weights/" + weightName))
79  {
80  samples[i] = new SPLINTER::DataTable(1, 1);
81 
82  for (label j = 0; j < coeffL2.cols();
83  j++) // j is the number of the nut snapshot
84  {
85  samples[i]->addSample(mu.row(j), coeffL2(i, j));
86  }
87 
88  ITHACAstream::ReadDenseMatrix(weights, "./ITHACAoutput/weights/", weightName);
89  rbfSplines[i] = new SPLINTER::RBFSpline(*samples[i],
90  SPLINTER::RadialBasisFunctionType::GAUSSIAN, weights);
91  std::cout << "Constructing RadialBasisFunction for mode " << i + 1 << std::endl;
92  }
93  else
94  {
95  samples[i] = new SPLINTER::DataTable(1, 1);
96 
97  for (label j = 0; j < coeffL2.cols();
98  j++) // j is the number of the nut snapshot
99  {
100  samples[i]->addSample(mu.row(j), coeffL2(i, j));
101  }
102 
103  rbfSplines[i] = new SPLINTER::RBFSpline(*samples[i],
104  SPLINTER::RadialBasisFunctionType::GAUSSIAN);
105  ITHACAstream::SaveDenseMatrix(rbfSplines[i]->weights,
106  "./ITHACAoutput/weights/", weightName);
107  std::cout << "Constructing RadialBasisFunction for mode " << i + 1 << std::endl;
108  }
109  }

112 void SteadyNSSimple::truthSolve2(List<scalar> mu_now, word Folder)
113 {
114  Time& runTime = _runTime();
115  volScalarField& p = _p();
116  volVectorField& U = _U();
117  fvMesh& mesh = _mesh();
118  surfaceScalarField& phi = _phi();
119  simpleControl& simple = _simple();
120  singlePhaseTransportModel& laminarTransport = _laminarTransport();
121  scalar residual = 1;
122  scalar uresidual = 1;
123  Vector<double> uresidual_v(0, 0, 0);
124  scalar presidual = 1;
125  scalar csolve = 0;
126  turbulence->read();
127  std::ofstream res_os;
128  std::ofstream snaps_os;
129  std::ofstream iters;
130  std::ofstream res_U;
131  std::ofstream res_P;
132  res_os.open(Folder + "/residuals", std::ios_base::app);
133  snaps_os.open(Folder + "/snaps", std::ios_base::app);
134  iters.open(Folder + "/iters", std::ios_base::app);
135  res_U.open(Folder + name(counter) + "/res_U", std::ios_base::app);
136  res_P.open(Folder + name(counter) + "/res_P", std::ios_base::app);
137  folderN = 0;
138  saver = 0;
139  middleStep = para->ITHACAdict->lookupOrDefault<label>("middleStep", 20);
140  middleExport = para->ITHACAdict->lookupOrDefault<bool>("middleExport", true);
141 #if OFVER == 6
142 
143  while (simple.loop(runTime) && residual > tolerance && csolve < maxIter )
144 #else
145  while (simple.loop() && residual > tolerance && csolve < maxIter )
146 #endif
147  {
148  csolve++;
149  saver++;
150  Info << "Time = " << runTime.timeName() << nl << endl;
151  volScalarField nueff = turbulence->nuEff();
152  fvVectorMatrix UEqn
153  (
154  fvm::div(phi, U)
155  - fvm::laplacian(nueff, U)
156  - fvc::div(nueff * dev2(T(fvc::grad(U))))
157  );
158  UEqn.relax();
159 
160  if (simple.momentumPredictor())
161  {
162  uresidual_v = solve(UEqn == - fvc::grad(p)).initialResidual();
163  }
164 
165  volVectorField U2(U);
166  scalar C = 0;
167 
168  for (label i = 0; i < 3; i++)
169  {
170  if (C < uresidual_v[i])
171  {
172  C = uresidual_v[i];
173  }
174  }
175 
176  uresidual = C;
177  volScalarField rAU(1.0 / UEqn.A());
178  volVectorField HbyA(constrainHbyA(1.0 / UEqn.A() * UEqn.H(), U, p));
179  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
180  adjustPhi(phiHbyA, U, p);
181  tmp<volScalarField> rAtU(rAU);
182 
183  if (simple.consistent())
184  {
185  rAtU = 1.0 / (1.0 / rAU - UEqn.H1());
186  phiHbyA +=
187  fvc::interpolate(rAtU() - rAU) * fvc::snGrad(p) * mesh.magSf();
188  HbyA -= (rAU - rAtU()) * fvc::grad(p);
189  }
190 
191  label i = 0;
192 
193  while (simple.correctNonOrthogonal())
194  {
195  fvScalarMatrix pEqn
196  (
197  fvm::laplacian(rAtU(), p) == fvc::div(phiHbyA)
198  );
199  pEqn.setReference(pRefCell, pRefValue);
200 
201  if (i == 0)
202  {
203  presidual = pEqn.solve().initialResidual();
204  }
205  else
206  {
207  pEqn.solve().initialResidual();
208  }
209 
210  if (simple.finalNonOrthogonalIter())
211  {
212  phi = phiHbyA - pEqn.flux();
213  }
214 
215  i++;
216  }
217 
218 #include "continuityErrs.H"
219  p.relax();
220  // Momentum corrector
221  U = HbyA - rAtU() * fvc::grad(p);
222  U.correctBoundaryConditions();
223  residual = max(presidual, uresidual);
224  Info << "Time = " << runTime.timeName() << nl << endl;
225  laminarTransport.correct();
226  turbulence->correct();
227 
228  if (saver == middleStep && middleExport)
229  {
230  saver = 0;
231  folderN++;
232  // if (folderN % 2 == 0)
233  // {
234  ITHACAstream::exportSolution(U, name(folderN), Folder + name(counter));
235  // }
236  // else
237  // {
238  // ITHACAstream::exportSolution(U2, name(folderN), Folder + name(counter));
239  // }
240  ITHACAstream::exportSolution(p, name(folderN), Folder + name(counter));
241  Ufield.append(U.clone());
242  Pfield.append(p.clone());
243  res_U << uresidual << std::endl;
244  res_P << presidual << std::endl;
245 
246  if (ITHACAutilities::isTurbulent())
247  {
248  auto nut = mesh.lookupObject<volScalarField>("nut");
249  ITHACAstream::exportSolution(nut, name(folderN), Folder + name(counter));
250  nutFields.append(nut.clone());
251  }
252  }
253  }
254 
255  snaps_os << folderN + 1 << std::endl;
256  iters << csolve << std::endl;
257  res_os << residual << std::endl;
258  res_os.close();
259  res_U.close();
260  res_P.close();
261  snaps_os.close();
262  iters.close();
263  runTime.setTime(runTime.startTime(), 0);
264 
265  if (middleExport)
266  {
267  ITHACAstream::exportSolution(U, name(folderN + 1), Folder + name(counter));
268  ITHACAstream::exportSolution(p, name(folderN + 1), Folder + name(counter));
269  }
270  else
271  {
272  ITHACAstream::exportSolution(U, name(counter), Folder);
273  ITHACAstream::exportSolution(p, name(counter), Folder);
274  }
275 
276  if (ITHACAutilities::isTurbulent())
277  {
278  auto nut = mesh.lookupObject<volScalarField>("nut");
279  ITHACAstream::exportSolution(nut, name(folderN + 1), Folder + name(counter));
280  nutFields.append(nut.clone());
281  }
282 
283  Ufield.append(U.clone());
284  Pfield.append(p.clone());
285  counter++;
286  writeMu(mu_now);
287  // --- Fill in the mu_samples with parameters (mu) to be used for the POD sample points
288  mu_samples.conservativeResize(mu_samples.rows() + 1, mu_now.size());
289 
290  for (label i = 0; i < mu_now.size(); i++)
291  {
292  mu_samples(mu_samples.rows() - 1, i) = mu_now[i];
293  }
294 
295  // Resize to Unitary if not initialized by user (i.e. non-parametric problem)
296  if (mu.cols() == 0)
297  {
298  mu.resize(1, 1);
299  }
300 
301  if (mu_samples.rows() == mu.cols())
302  {
303  ITHACAstream::exportMatrix(mu_samples, "mu_samples", "eigen",
304  Folder);
305  }
