97  example.setParameters();

112  ITHACADMD<vector> DMDv(example.Ufield, example.writeEvery);

115  DMDv.reconstruct(exportFolder, exportFieldNameU);

121  DMDp.reconstruct(exportFolder, exportFieldNameP);
