127  void linearMovePts(double angle, List<vector>& points2Move)
128  {
129  double sMax;
130  scalarList x;
131  scalarList y;
132 
133  for (label i = 0; i < points2Move.size(); i++)
134  {
135  x.append(points2Move[i].component(0));
136  y.append(points2Move[i].component(1));
137  }
138 
139  double xMin = min(x);
140  double xMax = max(x);
141  double yMin = min(y);
142  double yMax = max(y);
143  sMax = (yMax - yMin) * std::tan(M_PI * angle / 180);
144 
145  for (label i = 0; i < points2Move.size(); i++)
146  {
147  points2Move[i].component(0) = points2Move[i].component(0) +
148  (yMax - points2Move[i].component(1)) / (yMax - yMin) * (xMax -
149  points2Move[i].component(0)) / (xMax - xMin) * sMax;
150  }
151  }
