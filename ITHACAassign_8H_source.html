284 template<typename type_f>
285 void normalizeFields(
286  PtrList<GeometricField<type_f, fvPatchField, volMesh>>& fields)
287 {
288  Eigen::MatrixXd eigenFields = Foam2Eigen::PtrList2Eigen(fields);
289  List<Eigen::MatrixXd> eigenFieldsBC = Foam2Eigen::PtrList2EigenBC(fields);
290 
291  for (label i = 0; i < fields.size(); i++)
292  {
293  double norm = L2Norm(fields[i]);
294  GeometricField<type_f, fvPatchField, volMesh> tmp(fields[0].name(),
295  fields[0] * 0);
296  Eigen::VectorXd vec = eigenFields.col(i) / norm;
297  tmp = Foam2Eigen::Eigen2field(tmp, vec);
298 
299  // Adjusting boundary conditions
300  for (int k = 0; k < tmp.boundaryField().size(); k++)
301  {
302  Eigen::MatrixXd vec = eigenFieldsBC[k].col(i) / norm;
303  assignBC(tmp, k, vec);
304  }
305 
306  fields.set(i, tmp);
307  }
308 }
