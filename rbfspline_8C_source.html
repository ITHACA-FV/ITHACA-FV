23 RBFSpline::RBFSpline(const DataTable &samples, RadialBasisFunctionType type, bool normalized)
24  : samples(samples),
25  normalized(normalized),
26  precondition(false),
27  dim(samples.getNumVariables()),
28  numSamples(samples.getNumSamples())
29 {
30  if (type == RadialBasisFunctionType::THIN_PLATE_SPLINE)
31  {
32  fn = std::shared_ptr<RadialBasisFunction>(new ThinPlateSpline());
33  }
34  else if (type == RadialBasisFunctionType::MULTIQUADRIC)
35  {
36  fn = std::shared_ptr<RadialBasisFunction>(new Multiquadric());
37  }
38  else if (type == RadialBasisFunctionType::INVERSE_QUADRIC)
39  {
40  fn = std::shared_ptr<RadialBasisFunction>(new InverseQuadric());
41  }
42  else if (type == RadialBasisFunctionType::INVERSE_MULTIQUADRIC)
43  {
44  fn = std::shared_ptr<RadialBasisFunction>(new InverseMultiquadric());
45  }
46  else if (type == RadialBasisFunctionType::GAUSSIAN)
47  {
48  fn = std::shared_ptr<RadialBasisFunction>(new Gaussian());
49  }
50  else
51  {
52  fn = std::shared_ptr<RadialBasisFunction>(new ThinPlateSpline());
53  }

55  /* Want to solve the linear system A*w = b,
56  * where w is the vector of weights.
57  * NOTE: the system is dense and by default badly conditioned.
58  * It should be solved by a specialized solver such as GMRES
59  * with preconditioning (e.g. ACBF) as in matlab.
60  * NOTE: Consider trying the Łukaszyk–Karmowski metric (for two variables)
61  */
62  //SparseMatrix A(numSamples,numSamples);
63  //A.reserve(numSamples*numSamples);
64  DenseMatrix A; A.setZero(numSamples, numSamples);
65  DenseMatrix b; b.setZero(numSamples,1);
66 
67  int i=0;
68  for(auto it1 = samples.cbegin(); it1 != samples.cend(); ++it1, ++i)
69  {
70  double sum = 0;
71  int j=0;
72  for(auto it2 = samples.cbegin(); it2 != samples.cend(); ++it2, ++j)
73  {
74  double val = fn->eval(dist(*it1, *it2));
75  if(val != 0)
76  {
77  //A.insert(i,j) = val;
78  A(i,j) = val;
79  sum += val;
80  }
81  }
82 
83  double y = it1->getY();
84  if(normalized) b(i) = sum*y;
85  else b(i) = y;
86  }
87 
88  //A.makeCompressed();
89 
90  if(precondition)
91  {
92  // Calcualte precondition matrix P
93  DenseMatrix P = computePreconditionMatrix();
94 
95  // Preconditioned A and b
96  DenseMatrix Ap = P*A;
97  DenseMatrix bp = P*b;
98 
99  A = Ap;
100  b = bp;
101  }
102 
103 #ifndef NDEBUG
104  std::cout << "Computing RBF weights using dense solver." << std::endl;
105 #endif // NDEBUG
106 
107  // SVD analysis
108  Eigen::JacobiSVD<DenseMatrix> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
109  auto svals = svd.singularValues();
110  double svalmax = svals(0);
111  double svalmin = svals(svals.rows()-1);
112  double rcondnum = (svalmax <= 0.0 || svalmin <= 0.0) ? 0.0 : svalmin/svalmax;
113 
114 #ifndef NDEBUG
115  std::cout << "The reciprocal of the condition number is: " << rcondnum << std::endl;
116  std::cout << "Largest/smallest singular value: " << svalmax << " / " << svalmin << std::endl;
117 #endif // NDEBUG
118 
119  // Solve for weights
120  weights = svd.solve(b);
121 
122 #ifndef NDEBUG
123  // Compute error. If it is used later on, move this statement above the NDEBUG
124  double err = (A*weights - b).norm() / b.norm();
125  std::cout << "Error: " << std::setprecision(10) << err << std::endl;
126 #endif // NDEBUG

144 double RBFSpline::eval(std::vector<double> x) const
145 {
146  assert(x.size() == dim);
147  double fval, sum = 0, sumw = 0;
148  int i = 0;
149  for(auto it = samples.cbegin(); it != samples.cend(); ++it, ++i)
150  {
151  fval = fn->eval(dist(x,it->getX()));
152  sumw += weights(i)*fval;
153  sum += fval;
154  }
155  return normalized ? sumw/sum : sumw;

213 DenseMatrix RBFSpline::computePreconditionMatrix() const
214 {
215  DenseMatrix P;
216  P.setZero(numSamples,numSamples);
217 
218  // Calculate precondition matrix P based on
219  // purely local approximate cardinal basis functions (ACBF)
220  int sigma = std::max(1.0, std::floor(0.1*numSamples)); // Local points to consider
221 
222  int i=0;
223  for(auto it1 = samples.cbegin(); it1 != samples.cend(); ++it1, ++i)
224  {
225  Point p1(it1->getX());
226 
227  // Shift data using p1 as origin
228  std::vector<Point> shifted_points;
229  int j=0;
230  for(auto it2 = samples.cbegin(); it2 != samples.cend(); ++it2, ++j)
231  {
232  Point p2(it2->getX());
233  Point p3(p2-p1);
234  p3.setIndex(j); // store index with point
235  shifted_points.push_back(p3);
236  }
237  std::sort(shifted_points.begin(), shifted_points.end());
238 
239  // Find sigma closest points to p1
240  std::vector<Point> points;
241  std::vector<int> indices;
242  for(int j=0; j<sigma; j++)
243  {
244  Point p(shifted_points.at(j));
245  indices.push_back(p.getIndex());
246  Point p2 = p+p1;
247  p2.setIndex(p.getIndex());
248  points.push_back(p2); // The resulting point has a different index than that of p
249 // cout << p.getIndex() << "/" << p1.getIndex() << "/" << p2.getIndex() << endl;
250 // assert(p.getIndex() == p2.getIndex());
251  }
252 
253  // Add some points far away from p1 and preferably scattered around the domain boundary
254  for(int k=0; k<1; k++)
255  {
256  Point p(shifted_points.at(shifted_points.size()-1-k));
257  indices.push_back(p.getIndex());
258  Point p2 = p+p1;
259  p2.setIndex(p.getIndex());
260  points.push_back(p2);
261  }
262 
263  // Build and solve linear system
264  int m = points.size();
265 
266  DenseMatrix e; e.setZero(m,1); e(0,0) = 1;
267  DenseMatrix B; B.setZero(m,m);
268  DenseMatrix w; w.setZero(m,1);
269 
270  assert(points.front().getIndex() == i);
271 
272  for(int k1=0; k1<m; k1++)
273  {
274  for(int k2=0; k2<m; k2++)
275  {
276  Point p = points.at(k1) - points.at(k2);
277  B(k1,k2) = fn->eval(p.dist());
278  }
279  }
280 // DenseQR s;
281 // bool success = s.solve(B,e,w);
282 // assert(success);
283 
284  Eigen::JacobiSVD<DenseMatrix> svd(B, Eigen::ComputeThinU | Eigen::ComputeThinV);
285  w = svd.solve(e);
286  //assert(svd.info() == Eigen::Success);
287 
288  for(unsigned int j=0; j<numSamples; j++)
289  {
290  auto it = find(indices.begin(),indices.end(),j);
291  if(it!=indices.end())
292  {
293  int k = it - indices.begin();
294 
295  P(i,j) = w(k,0);
296  //cout << "j/k/g " << j << "/" << k << "/" << points.at(k).getIndex() << endl;
297  assert(points.at(k).getIndex()==j);
298  }
299  }
300  }
301 
302  return P;
303 }

308 double RBFSpline::dist(std::vector<double> x, std::vector<double> y) const
309 {
310  assert(x.size() == y.size());
311  double sum = 0.0;
312  for(unsigned int i=0; i<x.size(); i++)
313  sum += (x.at(i)-y.at(i))*(x.at(i)-y.at(i));
314  return std::sqrt(sum);
315 }
