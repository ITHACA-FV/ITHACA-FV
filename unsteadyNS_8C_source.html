63 void unsteadyNS::truthSolve(List<scalar> mu_now)
64 {
65  #include "initContinuityErrs.H"
66  Time& runTime = _runTime();
67  surfaceScalarField& phi = _phi();
68  fvMesh& mesh = _mesh();
69  fv::options& fvOptions = _fvOptions();
70  pimpleControl& pimple = _pimple();
71  volScalarField p = _p();
72  volVectorField U = _U();
73  IOMRFZoneList& MRF = _MRF();
74  singlePhaseTransportModel& laminarTransport = _laminarTransport();
75 
76  instantList Times = runTime.times();
77  runTime.setEndTime(finalTime);
78  // Perform a TruthSolve
79  bool WRITE;
80  runTime.setTime(Times[1], 1);
81  runTime.setDeltaT(timeStep);
82  nextWrite = startTime;
83 
84  // Start the time loop
85  while (runTime.run())
86  {
87 #include "readTimeControls.H"
88 #include "CourantNo.H"
89 #include "setDeltaT.H"
90  runTime.setEndTime(finalTime);
91  runTime++;
92  Info << "Time = " << runTime.timeName() << nl << endl;
93 
94  // --- Pressure-velocity PIMPLE corrector loop
95  while (pimple.loop())
96  {
97 #include "UEqn.H"
98  // --- Pressure corrector loop
99  while (pimple.correct())
100  {
101 #include "pEqn.H"
102  }
103  if (pimple.turbCorr())
104  {
105  laminarTransport.correct();
106  turbulence->correct();
107  }
108  }
109 
110  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
111  << " ClockTime = " << runTime.elapsedClockTime() << " s"
112  << nl << endl;
113  WRITE = checkWrite(runTime);
114  if (WRITE)
115  {
116  exportSolution(U, name(counter), "./ITHACAoutput/Offline/");
117  exportSolution(p, name(counter), "./ITHACAoutput/Offline/");
118  std::ofstream of("./ITHACAoutput/Offline/"+name(counter)+"/"+runTime.timeName());
119  Ufield.append(U);
120  Pfield.append(p);
121  counter++;
122  nextWrite += writeEvery;
123  writeMu(mu_now);
124  }
125  }
126 }
