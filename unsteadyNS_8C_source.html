96 void unsteadyNS::truthSolve(List<scalar> mu_now, fileName folder)
97 {
98  Time& runTime = _runTime();
99  surfaceScalarField& phi = _phi();
100  fvMesh& mesh = _mesh();
101 #include "initContinuityErrs.H"
102  fv::options& fvOptions = _fvOptions();
103  pimpleControl& pimple = _pimple();
104  volScalarField& p = _p();
105  volVectorField& U = _U();
106  IOMRFZoneList& MRF = _MRF();
107  singlePhaseTransportModel& laminarTransport = _laminarTransport();
108  instantList Times = runTime.times();
109  runTime.setEndTime(finalTime);
110  // Perform a TruthSolve
111  runTime.setTime(Times[1], 1);
112  runTime.setDeltaT(timeStep);
113  nextWrite = startTime;
114 
115  // Set time-dependent velocity BCs for initial condition
116  if (timedepbcMethod == "yes")
117  {
118  for (label i = 0; i < inletPatch.rows(); i++)
119  {
120  Vector<double> inl(0, 0, 0);
121 
122  for (label j = 0; j < inl.size(); j++)
123  {
124  inl[j] = timeBCoff(i * inl.size() + j, 0);
125  }
126 
127  assignBC(U, inletPatch(i, 0), inl);
128  }
129  }
130 
131  // Export and store the initial conditions for velocity and pressure
132  ITHACAstream::exportSolution(U, name(counter), folder);
133  ITHACAstream::exportSolution(p, name(counter), folder);
134  std::ofstream of(folder + name(counter) + "/" +
135  runTime.timeName());
136  Ufield.append(U);
137  Pfield.append(p);
138  counter++;
139  nextWrite += writeEvery;
140 
141  // Start the time loop
142  while (runTime.run())
143  {
144 #include "readTimeControls.H"
145 #include "CourantNo.H"
146 #include "setDeltaT.H"
147  runTime.setEndTime(finalTime);
148  runTime++;
149  Info << "Time = " << runTime.timeName() << nl << endl;
150 
151  // Set time-dependent velocity BCs
152  if (timedepbcMethod == "yes")
153  {
154  for (label i = 0; i < inletPatch.rows(); i++)
155  {
156  Vector<double> inl(0, 0, 0);
157 
158  for (label j = 0; j < inl.size(); j++)
159  {
160  inl[j] = timeBCoff(i * inl.size() + j, counter2);
161  }
162 
163  assignBC(U, inletPatch(i, 0), inl);
164  }
165 
166  counter2 ++;
167  }
168 
169  // --- Pressure-velocity PIMPLE corrector loop
170  while (pimple.loop())
171  {
172 #include "UEqn.H"
173 
174  // --- Pressure corrector loop
175  while (pimple.correct())
176  {
177 #include "pEqn.H"
178  }
179 
180  if (pimple.turbCorr())
181  {
182  laminarTransport.correct();
183  turbulence->correct();
184  }
185  }
186 
187  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
188  << " ClockTime = " << runTime.elapsedClockTime() << " s"
189  << nl << endl;
190 
191  if (checkWrite(runTime))
192  {
193  ITHACAstream::exportSolution(U, name(counter), folder);
194  ITHACAstream::exportSolution(p, name(counter), folder);
195  std::ofstream of(folder + name(counter) + "/" +
196  runTime.timeName());
197  Ufield.append(U);
198  Pfield.append(p);
199  counter++;
200  nextWrite += writeEvery;
201  writeMu(mu_now);
202  // --- Fill in the mu_samples with parameters (time, mu) to be used for the PODI sample points
203  mu_samples.conservativeResize(mu_samples.rows() + 1, mu_now.size() + 1);
204  mu_samples(mu_samples.rows() - 1, 0) = atof(runTime.timeName().c_str());
205 
206  for (label i = 0; i < mu_now.size(); i++)
207  {
208  mu_samples(mu_samples.rows() - 1, i + 1) = mu_now[i];
209  }
210  }
211  }
212 
213  // Resize to Unitary if not initialized by user (i.e. non-parametric problem)
214  if (mu.cols() == 0)
215  {
216  mu.resize(1, 1);
217  }
218 
219  if (mu_samples.rows() == counter * mu.cols())
220  {
221  ITHACAstream::exportMatrix(mu_samples, "mu_samples", "eigen",
222  folder);
223  }
224 }

227 bool unsteadyNS::checkWrite(Time& timeObject)
228 {
229  scalar diffnow = mag(nextWrite - atof(timeObject.timeName().c_str()));
230  scalar diffnext = mag(nextWrite - atof(timeObject.timeName().c_str()) -
231  timeObject.deltaTValue());
232 
233  if ( diffnow < diffnext)
234  {
235  return true;
236  }
237  else
238  {
239  return false;
240  }
241 }
