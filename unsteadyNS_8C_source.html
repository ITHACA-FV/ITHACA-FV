69 void unsteadyNS::truthSolve(List<scalar> mu_now)
70 {
71  #include "initContinuityErrs.H"
72  Time& runTime = _runTime();
73  surfaceScalarField& phi = _phi();
74  fvMesh& mesh = _mesh();
75  fv::options& fvOptions = _fvOptions();
76  pimpleControl& pimple = _pimple();
77  volScalarField p = _p();
78  volVectorField U = _U();
79  IOMRFZoneList& MRF = _MRF();
80  singlePhaseTransportModel& laminarTransport = _laminarTransport();
81 
82  instantList Times = runTime.times();
83  runTime.setEndTime(finalTime);
84  // Perform a TruthSolve
85  bool WRITE;
86  runTime.setTime(Times[1], 1);
87  runTime.setDeltaT(timeStep);
88  nextWrite = startTime;
89 
90  // Start the time loop
91  while (runTime.run())
92  {
93 #include "readTimeControls.H"
94 #include "CourantNo.H"
95 #include "setDeltaT.H"
96  runTime.setEndTime(finalTime);
97  runTime++;
98  Info << "Time = " << runTime.timeName() << nl << endl;
99 
100  // --- Pressure-velocity PIMPLE corrector loop
101  while (pimple.loop())
102  {
103 #include "UEqn.H"
104  // --- Pressure corrector loop
105  while (pimple.correct())
106  {
107 #include "pEqn.H"
108  }
109  if (pimple.turbCorr())
110  {
111  laminarTransport.correct();
112  turbulence->correct();
113  }
114  }
115 
116  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
117  << " ClockTime = " << runTime.elapsedClockTime() << " s"
118  << nl << endl;
119  WRITE = checkWrite(runTime);
120  if (WRITE)
121  {
122  exportSolution(U, name(counter), "./ITHACAoutput/Offline/");
123  exportSolution(p, name(counter), "./ITHACAoutput/Offline/");
124  std::ofstream of("./ITHACAoutput/Offline/"+name(counter)+"/"+runTime.timeName());
125  Ufield.append(U);
126  Pfield.append(p);
127  counter++;
128  nextWrite += writeEvery;
129  writeMu(mu_now);
130  }
131  }
132 }
