74 void unsteadyNS::truthSolve(List<scalar> mu_now)
75 {
76 #include "initContinuityErrs.H"
77  Time& runTime = _runTime();
78  surfaceScalarField& phi = _phi();
79  fvMesh& mesh = _mesh();
80  fv::options& fvOptions = _fvOptions();
81  pimpleControl& pimple = _pimple();
82  volScalarField& p = _p();
83  volVectorField& U = _U();
84  IOMRFZoneList& MRF = _MRF();
85  singlePhaseTransportModel& laminarTransport = _laminarTransport();
86  instantList Times = runTime.times();
87  runTime.setEndTime(finalTime);
88  // Perform a TruthSolve
89  runTime.setTime(Times[1], 1);
90  runTime.setDeltaT(timeStep);
91  nextWrite = startTime;
92  // Initialize Nsnapshots
93  int nsnapshots = 0;
94 
95  // Start the time loop
96  while (runTime.run())
97  {
98 #include "readTimeControls.H"
99 #include "CourantNo.H"
100 #include "setDeltaT.H"
101  runTime.setEndTime(finalTime + timeStep);
102  Info << "Time = " << runTime.timeName() << nl << endl;
103 
104  // --- Pressure-velocity PIMPLE corrector loop
105  while (pimple.loop())
106  {
107 #include "UEqn.H"
108 
109  // --- Pressure corrector loop
110  while (pimple.correct())
111  {
112 #include "pEqn.H"
113  }
114 
115  if (pimple.turbCorr())
116  {
117  laminarTransport.correct();
118  turbulence->correct();
119  }
120  }
121 
122  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
123  << " ClockTime = " << runTime.elapsedClockTime() << " s"
124  << nl << endl;
125 
126  if (checkWrite(runTime))
127  {
128  nsnapshots += 1;
129  ITHACAstream::exportSolution(U, name(counter), "./ITHACAoutput/Offline/");
130  ITHACAstream::exportSolution(p, name(counter), "./ITHACAoutput/Offline/");
131  std::ofstream of("./ITHACAoutput/Offline/" + name(counter) + "/" +
132  runTime.timeName());
133  Ufield.append(U);
134  Pfield.append(p);
135  counter++;
136  nextWrite += writeEvery;
137  writeMu(mu_now);
138  // --- Fill in the mu_samples with parameters (time, mu) to be used for the PODI sample points
139  mu_samples.conservativeResize(mu_samples.rows() + 1, mu_now.size() + 1);
140  mu_samples(mu_samples.rows() - 1, 0) = atof(runTime.timeName().c_str());
141 
142  for (int i = 0; i < mu_now.size(); i++)
143  {
144  mu_samples(mu_samples.rows() - 1, i + 1) = mu_now[i];
145  }
146  }
147 
148  runTime++;
149  }
150 
151  // Resize to Unitary if not initialized by user (i.e. non-parametric problem)
152  if (mu.cols() == 0)
153  {
154  mu.resize(1, 1);
155  }
156 
157  if (mu_samples.rows() == nsnapshots * mu.cols())
158  {
159  ITHACAstream::exportMatrix(mu_samples, "mu_samples", "eigen",
160  "./ITHACAoutput/Offline");
161  }
162 }

165 bool unsteadyNS::checkWrite(Time& timeObject)
166 {
167  scalar diffnow = mag(nextWrite - atof(timeObject.timeName().c_str()));
168  scalar diffnext = mag(nextWrite - atof(timeObject.timeName().c_str()) -
169  timeObject.deltaTValue());
170 
171  if ( diffnow < diffnext)
172  {
173  return true;
174  }
175  else
176  {
177  return false;
178  }
