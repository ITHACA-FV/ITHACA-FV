47 class DEIM_function : public DEIM<fvScalarMatrix>
48 {
49  public:
50  using DEIM::DEIM;
51  static fvScalarMatrix evaluate_expression(volScalarField& T, Eigen::MatrixXd mu)
52  {
53  volScalarField yPos = T.mesh().C().component(vector::Y);
54  volScalarField xPos = T.mesh().C().component(vector::X);
55  volScalarField nu(T);
56 
57  for (auto i = 0; i < nu.size(); i++)
58  {
59  nu[i] = std::exp( - 2 * std::pow(xPos[i] - mu(0) - 1,
60  2) - 2 * std::pow(yPos[i] - mu(1) - 0.5, 2)) + 1;
61  }
62 
63  nu.correctBoundaryConditions();
64  fvScalarMatrix TiEqn22
65  (
66  fvm::laplacian(nu, T, "Gauss linear")
67  );
68  return TiEqn22;
69  }

71  Eigen::MatrixXd onlineCoeffsA(Eigen::MatrixXd mu)
72  {
73  Eigen::MatrixXd theta(fieldsA.size(), 1);
74 
75  for (int i = 0; i < fieldsA.size(); i++)
76  {
77  Eigen::SparseMatrix<double> Mr;
78  Eigen::VectorXd br;
79  fvScalarMatrix Aof = evaluate_expression(fieldsA[i], mu);
80  Foam2Eigen::fvMatrix2Eigen(Aof, Mr, br);
81  int ind_row = localMagicPointsA[i].first() + xyz_A[i].first() *
82  fieldsA[i].size();
83  int ind_col = localMagicPointsA[i].second() + xyz_A[i].second() *
84  fieldsA[i].size();
85  theta(i) = Mr.coeffRef(ind_row, ind_col);
86  }
87 
88  return theta;
89  }
90 
91  Eigen::MatrixXd onlineCoeffsB(Eigen::MatrixXd mu)
92  {
93  Eigen::MatrixXd theta(fieldsB.size(), 1);
94 
95  for (int i = 0; i < fieldsB.size(); i++)
96  {
97  Eigen::SparseMatrix<double> Mr;
98  Eigen::VectorXd br;
99  fvScalarMatrix Aof = evaluate_expression(fieldsB[i], mu);
100  Foam2Eigen::fvMatrix2Eigen(Aof, Mr, br);
101  int ind_row = localMagicPointsB[i] + xyz_B[i] * fieldsB[i].size();
102  theta(i) = br(ind_row);
103  }
104 
105  return theta;
106  }

109  PtrList<volScalarField> fieldsB;

202  void PODDEIM(int NmodesT, int NmodesDEIMA, int NmodesDEIMB)
203  {
204  DEIMmatrice = new DEIM_function(Mlist, NmodesDEIMA, NmodesDEIMB, "T_matrix");
205  fvMesh& mesh = const_cast<fvMesh&>(T.mesh());
206  // Differential Operator
207  DEIMmatrice->fieldsA = DEIMmatrice->generateSubmeshesMatrix(2, mesh, T);
208  // Source Terms
209  DEIMmatrice->fieldsB = DEIMmatrice->generateSubmeshesVector(2, mesh, T);
210  ModesTEig = Foam2Eigen::PtrList2Eigen(Tmodes);
211  ModesTEig.conservativeResize(ModesTEig.rows(), NmodesT);
212  ReducedMatricesA.resize(NmodesDEIMA);
213  ReducedVectorsB.resize(NmodesDEIMB);
214 
215  for (int i = 0; i < NmodesDEIMA; i++)
216  {
217  ReducedMatricesA[i] = ModesTEig.transpose() * DEIMmatrice->MatrixOnlineA[i] *
218  ModesTEig;
219  }
220 
221  for (int i = 0; i < NmodesDEIMB; i++)
222  {
223  ReducedVectorsB[i] = ModesTEig.transpose() * DEIMmatrice->MatrixOnlineB;
224  }
225  };

227  void OnlineSolve(Eigen::MatrixXd par_new, word Folder)
228  {
229  auto t1 = std::chrono::high_resolution_clock::now();
230  auto t2 = std::chrono::high_resolution_clock::now();
231  auto time_span = std::chrono::duration_cast<std::chrono::duration<double>>
232  (t2 - t1);
233  time_rom = 0;
234 
235  for (int i = 0; i < par_new.rows(); i++)
236  {
237  // solve
238  t1 = std::chrono::high_resolution_clock::now();
239  Eigen::MatrixXd thetaonA = DEIMmatrice->onlineCoeffsA(par_new.row(i));
240  Eigen::MatrixXd thetaonB = DEIMmatrice->onlineCoeffsB(par_new.row(i));
241  Eigen::MatrixXd A = EigenFunctions::MVproduct(ReducedMatricesA, thetaonA);
242  Eigen::VectorXd B = EigenFunctions::MVproduct(ReducedVectorsB, thetaonB);
243  Eigen::VectorXd x = A.fullPivLu().solve(B);
244  Eigen::VectorXd full = ModesTEig * x;
245  t2 = std::chrono::high_resolution_clock::now();
246  time_span = std::chrono::duration_cast<std::chrono::duration<double>>(t2 - t1);
247  time_rom += time_span.count();
248  // Export
249  volScalarField Tred("Tred", T);
250  Tred = Foam2Eigen::Eigen2field(Tred, full);
251  ITHACAstream::exportSolution(Tred, name(i + 1), "./ITHACAoutput/" + Folder);
252  Tonline.append(Tred);
253  }
254  }

257 int main(int argc, char* argv[])
258 {
259  // Construct the case
260  DEIMLaplacian example(argc, argv);
261  // Read some parameters from file
262  ITHACAparameters* para = ITHACAparameters::getInstance(example._mesh(),
263  example._runTime());
264  // Create the offline parameters for the solve
265  example.mu = ITHACAutilities::rand(100, 2, -0.5, 0.5);
266  // Solve the offline problem to compute the snapshots for the projections
267  example.OfflineSolve(example.mu, "Offline");
268  // Compute the POD modes
269  ITHACAPOD::getModes(example.Tfield, example.Tmodes, example._T().name(),
270  example.podex, 0, 0, 20);
271  // Compute the offline part of the DEIM procedure
272  example.PODDEIM();
273  // Construct a new set of parameters
274  Eigen::MatrixXd par_new1 = ITHACAutilities::rand(100, 2, -0.5, 0.5);
275  // Solve the online problem with the new parameters
276  example.OnlineSolve(par_new1, "Online_red");
277  // Solve a new full problem with the new parameters (necessary to compute speed up and error)
278  DEIMLaplacian example_new(argc, argv);
279  example_new.OnlineSolveFull(par_new1, "Online_full");
280  // Output some infos
281  std::cout << std::endl << "The FOM Solve took: " << example_new.time_full <<
282  " seconds." << std::endl;
283  std::cout << std::endl << "The ROM Solve took: " << example.time_rom <<
284  " seconds." << std::endl;
285  std::cout << std::endl << "The Speed-up is: " << example_new.time_full /
286  example.time_rom << std::endl << std::endl;
287  Eigen::MatrixXd error = ITHACAutilities::errorL2Abs(example_new.Tfield,
288  example.Tonline);
289  std::cout << "The mean L2 error is: " << error.mean() << std::endl;
290  exit(0);
291 }
