46 reducedUnsteadyNST::reducedUnsteadyNST(unsteadyNST& FOMproblem)
47 //problem(&FOMproblem)
48 {
49  problem = &FOMproblem;
50  N_BC = problem->inletIndex.rows();
51  N_BC_t = problem->inletIndexT.rows();
52  Nphi_u = problem->B_matrix.rows();
53  Nphi_p = problem->K_matrix.cols();
54  Nphi_t = problem->Y_matrix.rows();
55 
56  // Create locally the velocity modes
57  for (label k = 0; k < problem->liftfield.size(); k++)
58  {
59  Umodes.append(problem->liftfield[k]);
60  }
61  for (label k = 0; k < problem->NUmodes; k++)
62  {
63  Umodes.append(problem->Umodes[k]);
64  }
65  for (label k = 0; k < problem->NSUPmodes; k++)
66  {
67  Umodes.append(problem->supmodes[k]);
68  }
69  // Create locally the pressure modes
70  for (label k = 0; k < problem->NPmodes; k++)
71  {
72  Pmodes.append(problem->Pmodes[k]);
73  }
74  for (label k = 0; k < problem->liftfieldT.size(); k++)
75  {
76  LTmodes.append(problem->liftfieldT[k]);
77  }
78  // Create locally the temperature modes
79  for (label k = 0; k < problem->NTmodes; k++)
80  {
81  LTmodes.append(problem->LTmodes[k]);
82  }
83  // Store locally the snapshots for projections
84  for (label k = 0; k < problem->Ufield.size(); k++)
85  {
86  Usnapshots.append(problem->Ufield[k]);
87  Psnapshots.append(problem->Pfield[k]);
88  }
89  for (label k = 0; k < problem->Tfield.size(); k++)
90  {
91  Tsnapshots.append(problem->Tfield[k]);
92  }
93 
94  newton_object_sup = newton_unsteadyNST_sup(Nphi_u + Nphi_p, Nphi_u + Nphi_p, FOMproblem);
95  newton_object_sup_t = newton_unsteadyNST_sup_t(Nphi_t, Nphi_t, FOMproblem);
96 }

101 int newton_unsteadyNST_sup::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
102 {
103  Eigen::VectorXd a_dot(Nphi_u);
104  Eigen::VectorXd a_tmp(Nphi_u);
105  Eigen::VectorXd b_tmp(Nphi_p);
106  a_tmp = x.head(Nphi_u);
107  b_tmp = x.tail(Nphi_p);
108  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
109 
110  // Convective term
111  Eigen::MatrixXd cc(1, 1);
112  // Momentum Term
113  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
114  // Gradient of pressure
115  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
116  // Mass Term Velocity
117  Eigen::VectorXd M5 = problem->M_matrix * a_dot;
118  // Pressure Term
119  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
120 
121  for (label i = 0; i < Nphi_u; i++)
122  {
123  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp;
124  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M2(i);
125  }
126  for (label j = 0; j < Nphi_p; j++)
127  {
128  label k = j + Nphi_u;
129  fvec(k) = M3(j);
130  }
131  for (label j = 0; j < N_BC; j++)
132  {
133  fvec(j) = x(j) - BC(j);
134  }
135  return 0;
136 
137 }

148 int newton_unsteadyNST_sup_t::operator()(const Eigen::VectorXd &t, Eigen::VectorXd &fvect) const
149 {
150  Eigen::VectorXd c_dot(Nphi_t);
151  Eigen::VectorXd c_tmp(Nphi_t);
152  c_tmp = t.head(Nphi_t);
153  c_dot = (t.head(Nphi_t) - z_old.head(Nphi_t)) / dt;
154 
155  // Convective term temperature
156  Eigen::MatrixXd qq(1, 1);
157  // diffusive term temperature
158  Eigen::VectorXd M6 = problem->Y_matrix * c_tmp * DT;
159  // Mass Term Temperature
160  Eigen::VectorXd M8 = problem->MT_matrix * c_dot;
161 
162  for (label i = 0; i < Nphi_t; i++)
163  {
164  qq = a_tmp.transpose() *problem-> Q_matrix[i] * c_tmp;
165  fvect(i) = M8(i) - M6(i) + qq(0, 0);
166  }
167  for (label j = 0; j < N_BC_t; j++)
168  {
169  fvect(j) = t(j) - BC_t(j);
170  }
171  return 0;
172 }

182 void reducedUnsteadyNST::solveOnline_sup(Eigen::MatrixXd& vel_now, Eigen::MatrixXd& temp_now, label startSnap)
183 {
184  // Create and resize the solution vector
185  y.resize(Nphi_u + Nphi_p, 1);
186  y.setZero();
187  z.resize(Nphi_t, 1);
188  z.setZero();
189  // Set Initial Conditions
190  if (this->tstart != 0)
191  {
192  y.head(Nphi_u) = ITHACAutilities::get_coeffs(Usnapshots[startSnap], Umodes);
193  y.tail(Nphi_p) = ITHACAutilities::get_coeffs(Psnapshots[startSnap], Pmodes);
194  z.head(Nphi_t) = ITHACAutilities::get_coeffs(Tsnapshots[startSnap], LTmodes);
195  }
196  // Change initial condition for the lifting function
197  for (label j = 0; j < N_BC; j++)
198  {
199  y(j) = vel_now(j, 0);
200  }
201  for (label j = 0; j < N_BC_t; j++)
202  {
203  z(j) = temp_now(j, 0);
204  }
205  // Set some properties of the newton object
206  newton_object_sup.nu = nu;
207  newton_object_sup.y_old = y;
208  newton_object_sup.dt = dt;
209  newton_object_sup_t.DT = DT;
210  newton_object_sup_t.z_old = z;
211  newton_object_sup_t.dt = dt;
212  newton_object_sup.BC.resize(N_BC);
213  newton_object_sup_t.BC_t.resize(N_BC_t);
214  for (label j = 0; j < N_BC_t; j++)
215  {
216  newton_object_sup_t.BC_t(j) = temp_now(j, 0);
217  }
218  for (label j = 0; j < N_BC; j++)
219  {
220  newton_object_sup.BC(j) = vel_now(j, 0);
221  }
222 
223  // Set number of online solutions
224  int Ntsteps = static_cast<int>((finalTime - tstart) / dt);
225  online_solution.resize(Ntsteps);
226  online_solutiont.resize(Ntsteps);
227  // Set the initial time
228  time = tstart;
229  // Counting variable
230  int counter = 0;
231  // Create vector to store temporal solution and save initial condition as first solution
232  Eigen::MatrixXd tmp_sol(Nphi_u + Nphi_p + 1, 1);
233  tmp_sol(0) = time;
234  tmp_sol.col(0).tail(y.rows()) = y;
235  online_solution[counter] = tmp_sol;
236  Eigen::MatrixXd tmp_solt(Nphi_t + 1, 1);
237  tmp_solt(0) = time;
238  tmp_solt.col(0).tail(z.rows()) = z;
239  online_solutiont[counter] = tmp_solt;
240  counter ++;
241  // Create nonlinear solver object
242  Eigen::HybridNonLinearSolver<newton_unsteadyNST_sup> hnls(newton_object_sup);
243  Eigen::HybridNonLinearSolver<newton_unsteadyNST_sup_t> hnlst(newton_object_sup_t);
244  // Set output colors for fancy output
245  Color::Modifier red(Color::FG_RED);
246  Color::Modifier green(Color::FG_GREEN);
247  Color::Modifier def(Color::FG_DEFAULT);
248  // Start the time loop
249  while (time < finalTime)
250  {
251  time = time + dt;
252  Eigen::VectorXd res(y);
253  Eigen::VectorXd rest(z);
254 
255  res.setZero();
256  rest.setZero();
257  hnls.solve(y);
258  for (label j = 0; j < N_BC; j++)
259  {
260  y(j) = vel_now(j, 0);
261  }
262  // set the a_temp
263  // solve for temperature
264  newton_object_sup_t.a_tmp= y.head(Nphi_u);
265  hnlst.solve(z);
266  for (label j = 0; j < N_BC_t; j++)
267  {
268  z(j) = temp_now(j, 0);
269  }
270  newton_object_sup.operator()(y, res);
271  newton_object_sup.y_old = y;
272  newton_object_sup_t.operator()(z,rest);
273  newton_object_sup_t.z_old = z;
274  std::cout << "################## Online solve N° " << count_online_solve << " ##################" << std::endl;
275  Info << "Time = " << time << endl;
276  std::cout << "Solving for the parameter: " << vel_now << std::endl;
277  std::cout << "Solving for the parameter: " << temp_now << std::endl;
278 
279  if (res.norm() < 1e-5)
280  {
281  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
282  }
283  else
284  {
285  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
286  }
287  if (rest.norm() < 1e-5)
288  {
289  std::cout << green << "|F(x)| = " << rest.norm() << " - Minimun reached in " << hnlst.iter << " iterations " << def << std::endl << std::endl;
290  }
291  else
292  {
293  std::cout << red << "|F(x)| = " << rest.norm() << " - Minimun reached in " << hnlst.iter << " iterations " << def << std::endl << std::endl;
294  }
295  count_online_solve += 1;
296 
297  tmp_sol(0) = time;
298  tmp_sol.col(0).tail(y.rows()) = y;
299 
300  if (counter >= online_solution.size())
301  {
302  online_solution.append(tmp_sol);
303  }
304  else
305  {
306  online_solution[counter] = tmp_sol;
307  }
308 
309  tmp_solt(0) = time;
310  tmp_solt.col(0).tail(z.rows()) = z;
311  if (counter >= online_solutiont.size())
312  {
313  online_solutiont.append(tmp_solt);
314  }
315  else
316  {
317  online_solutiont[counter] = tmp_solt;
318  }
319  counter ++;
320  }
321  // Save the solution
322  ITHACAstream::exportMatrix(online_solution, "red_coeff", "python", "./ITHACAoutput/red_coeff");
323  ITHACAstream::exportMatrix(online_solution, "red_coeff", "matlab", "./ITHACAoutput/red_coeff");
324  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "python", "./ITHACAoutput/red_coeff_t");
325  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "matlab", "./ITHACAoutput/red_coeff_t");
326  count_online_solve += 1;
327 }
