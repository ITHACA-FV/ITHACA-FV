41 reducedSteadyNSturb::reducedSteadyNSturb(steadyNSturb& FOMproblem)

65 int newton_steadyNSturb::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
66 {
67 
68  Eigen::VectorXd a_tmp(Nphi_u);
69  Eigen::VectorXd b_tmp(Nphi_p);
70  a_tmp = x.head(Nphi_u);
71  b_tmp = x.tail(Nphi_p);
72  // Convective term
73  Eigen::MatrixXd cc(1, 1);
74  // Mom Term
75  Eigen::VectorXd M1 = problem->B_total_matrix * a_tmp * nu;
76  // Gradient of pressure
77  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
78  // Pressure Term
79  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
80 
81  for (label i = 0; i < Nphi_u; i++)
82  {
83  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp - nu_c.transpose() * problem->C_total_matrix[i] * a_tmp;
84  fvec(i) = M1(i) - cc(0, 0) - M2(i);
85  }
86  for (label j = 0; j < Nphi_p; j++)
87  {
88  label k = j + Nphi_u;
89  fvec(k) = M3(j);
90  }
91  for (label j = 0; j < N_BC; j++)
92  {
93  fvec(j) = x(j) - BC(j);
94  }
95  return 0;
96 }

109 void reducedSteadyNSturb::solveOnline_sup(Eigen::MatrixXd vel_now)
110 {
111 
112  y.resize(Nphi_u + Nphi_p, 1);
113  y.setZero();
114  for (label j = 0; j < N_BC; j++)
115  {
116  y(j) = vel_now(j, 0);
117  }
118  Color::Modifier red(Color::FG_RED);
119  Color::Modifier green(Color::FG_GREEN);
120  Color::Modifier def(Color::FG_DEFAULT);
121  Eigen::HybridNonLinearSolver<newton_steadyNSturb> hnls(newton_object);
122  newton_object.BC.resize(N_BC);
123  for (label j = 0; j < N_BC; j++)
124  {
125  newton_object.BC(j) = vel_now(j, 0);
126  }
127  for (label i = 0; i < Nphi_nut; i++)
128  {
129  newton_object.nu_c(i) = problem->rbfsplines[i]->eval(vel_now);
130  }
131 
132  volScalarField nut_rec("nut_rec", problem->nuTmodes[0] * 0);
133  for (label j = 0; j < Nphi_nut; j++)
134  {
135  nut_rec += problem->nuTmodes[j] * newton_object.nu_c(j);
136  }
137  nutREC.append(nut_rec);
138  newton_object.nu = nu;
139  hnls.solve(y);
140  Eigen::VectorXd res(y);
141  newton_object.operator()(y, res);
142 
143  std::cout << "################## Online solve N° " << count_online_solve << " ##################" << std::endl;
144  std::cout << "Solving for the parameter: " << vel_now << std::endl;
145  if (res.norm() < 1e-5)
146  {
147  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
148 }
149 else
150 {
151  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
152 }
153 count_online_solve += 1;
154 }

157 void reducedSteadyNSturb::reconstruct_sup(fileName folder, int printevery)
158 {
159  mkDir(folder);
160  ITHACAutilities::createSymLink(folder);
161 
162  int counter = 0;
163  int nextwrite = 0;
164 
165  for (label i = 0; i < online_solution.size(); i++)
166  {
167  if (counter == nextwrite)
168  {
169  volVectorField U_rec("U_rec", Umodes[0] * 0);
170  for (label j = 0; j < Nphi_u; j++)
171  {
172  U_rec += Umodes[j] * online_solution[i](j + 1, 0);
173  }
174  problem->exportSolution(U_rec, name(online_solution[i](0, 0)), folder);
175  volScalarField P_rec("P_rec", problem->Pmodes[0] * 0);
176  for (label j = 0; j < Nphi_p; j++)
177  {
178  P_rec += problem->Pmodes[j] * online_solution[i](j + Nphi_u + 1, 0);
179  }
180  problem->exportSolution(P_rec, name(online_solution[i](0, 0)), folder);
181  nextwrite += printevery;
182 
183  UREC.append(U_rec);
184  PREC.append(P_rec);
185  }
186  counter++;
187  }
188 }
