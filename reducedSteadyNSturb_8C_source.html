68 int newton_steadyNSturb::operator()(const Eigen::VectorXd& x,
69  Eigen::VectorXd& fvec) const
70 {
71  Eigen::VectorXd a_tmp(Nphi_u);
72  Eigen::VectorXd b_tmp(Nphi_p);
73  a_tmp = x.head(Nphi_u);
74  b_tmp = x.tail(Nphi_p);
75  // Convective term
76  Eigen::MatrixXd cc(1, 1);
77  // Mom Term
78  Eigen::VectorXd M1 = problem->B_total_matrix * a_tmp * nu;
79  // Gradient of pressure
80  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
81  // Pressure Term
82  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
83 
84  for (label i = 0; i < Nphi_u; i++)
85  {
86  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp - nu_c.transpose() *
87  problem->C_total_matrix[i] * a_tmp;
88  fvec(i) = M1(i) - cc(0, 0) - M2(i);
89  }
90 
91  for (label j = 0; j < Nphi_p; j++)
92  {
93  label k = j + Nphi_u;
94  fvec(k) = M3(j);
95  }
96 
97  for (label j = 0; j < N_BC; j++)
98  {
99  fvec(j) = x(j) - BC(j);
100  }
101 
102  return 0;
103 }

117 void reducedSteadyNSturb::solveOnline_sup(Eigen::MatrixXd vel_now)
118 {
119  y.resize(Nphi_u + Nphi_p, 1);
120  y.setZero();
121 
122  for (label j = 0; j < N_BC; j++)
123  {
124  y(j) = vel_now(j, 0);
125  }
126 
127  Color::Modifier red(Color::FG_RED);
128  Color::Modifier green(Color::FG_GREEN);
129  Color::Modifier def(Color::FG_DEFAULT);
130  Eigen::HybridNonLinearSolver<newton_steadyNSturb> hnls(newton_object);
131  newton_object.BC.resize(N_BC);
132 
133  for (label j = 0; j < N_BC; j++)
134  {
135  newton_object.BC(j) = vel_now(j, 0);
136  }
137 
138  for (label i = 0; i < Nphi_nut; i++)
139  {
140  newton_object.nu_c(i) = problem->rbfsplines[i]->eval(vel_now);
141  }
142 
143  volScalarField nut_rec("nut_rec", problem->nuTmodes[0] * 0);
144 
145  for (label j = 0; j < Nphi_nut; j++)
146  {
147  nut_rec += problem->nuTmodes[j] * newton_object.nu_c(j);
148  }
149 
150  nutREC.append(nut_rec);
151  newton_object.nu = nu;
152  hnls.solve(y);
153  Eigen::VectorXd res(y);
154  newton_object.operator()(y, res);
155  std::cout << "################## Online solve N° " << count_online_solve <<
156  " ##################" << std::endl;
157  std::cout << "Solving for the parameter: " << vel_now << std::endl;
158 
159  if (res.norm() < 1e-5)
160  {
161  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
162  hnls.iter << " iterations " << def << std::endl << std::endl;
163  }
164  else
165  {
166  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
167  hnls.iter << " iterations " << def << std::endl << std::endl;
168  }
169 
170  count_online_solve += 1;
171 }

174 void reducedSteadyNSturb::reconstruct_sup(fileName folder, int printevery)
175 {
176  mkDir(folder);
177  ITHACAutilities::createSymLink(folder);
178  int counter = 0;
179  int nextwrite = 0;
180 
181  for (label i = 0; i < online_solution.size(); i++)
182  {
183  if (counter == nextwrite)
184  {
185  volVectorField U_rec("U_rec", Umodes[0] * 0);
186 
187  for (label j = 0; j < Nphi_u; j++)
188  {
189  U_rec += Umodes[j] * online_solution[i](j + 1, 0);
190  }
191 
192  problem->exportSolution(U_rec, name(online_solution[i](0, 0)), folder);
193  volScalarField P_rec("P_rec", problem->Pmodes[0] * 0);
194 
195  for (label j = 0; j < Nphi_p; j++)
196  {
197  P_rec += problem->Pmodes[j] * online_solution[i](j + Nphi_u + 1, 0);
198  }
199 
200  problem->exportSolution(P_rec, name(online_solution[i](0, 0)), folder);
201  nextwrite += printevery;
202  UREC.append(U_rec);
203  PREC.append(P_rec);
204  }
205 
206  counter++;
207  }
208 }
