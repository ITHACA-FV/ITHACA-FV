55 /*
56  * Find coefficients of B-spline by solving:
57  * min ||A*x - b||^2 + alpha*||R||^2,
58  * where
59  * A = mxn matrix of n basis functions evaluated at m sample points,
60  * b = vector of m sample points y-values (or x-values when calculating knot averages),
61  * x = B-spline coefficients (or knot averages),
62  * R = Regularization matrix,
63  * alpha = regularization parameter.
64  */
65 DenseVector BSpline::Builder::computeCoefficients(const BSpline& bspline) const
66 {
67  SparseMatrix B = computeBasisFunctionMatrix(bspline);
68  SparseMatrix A = B;
69  DenseVector b = getSamplePointValues();
70 
71  if (_smoothing == Smoothing::IDENTITY)
72  {
73  /*
74  * Computing B-spline coefficients with a regularization term
75  * ||Ax-b||^2 + alpha*x^T*x
76  *
77  * NOTE: This corresponds to a Tikhonov regularization (or ridge regression) with the Identity matrix.
78  * See: https://en.wikipedia.org/wiki/Tikhonov_regularization
79  *
80  * NOTE2: consider changing regularization factor to (alpha/numSample)
81  */
82  SparseMatrix Bt = B.transpose();
83  A = Bt*B;
84  b = Bt*b;
85 
86  auto I = SparseMatrix(A.cols(), A.cols());
87  I.setIdentity();
88  A += _alpha*I;
89  }
90  else if (_smoothing == Smoothing::PSPLINE)
91  {
92  /*
93  * The P-Spline is a smooting B-spline which relaxes the interpolation constraints on the control points to allow
94  * smoother spline curves. It minimizes an objective which penalizes both deviation from sample points (to lower bias)
95  * and the magnitude of second derivatives (to lower variance).
96  *
97  * Setup and solve equations Ax = b,
98  * A = B'*W*B + l*D'*D
99  * b = B'*W*y
100  * x = control coefficients or knot averages.
101  * B = basis functions at sample x-values,
102  * W = weighting matrix for interpolating specific points
103  * D = second-order finite difference matrix
104  * l = penalizing parameter (increase for more smoothing)
105  * y = sample y-values when calculating control coefficients,
106  * y = sample x-values when calculating knot averages
107  */
108 
109  // Assuming regular grid
110  unsigned int numSamples = _data.getNumSamples();
111 
112  SparseMatrix Bt = B.transpose();
113 
114  // Weight matrix
115  SparseMatrix W;
116  W.resize(numSamples, numSamples);
117  W.setIdentity();
118 
119  // Second order finite difference matrix
120  SparseMatrix D = getSecondOrderFiniteDifferenceMatrix(bspline);
121 
122  // Left-hand side matrix
123  A = Bt*W*B + _alpha*D.transpose()*D;
124 
125  // Compute right-hand side matrices
126  b = Bt*W*b;
127  }
128 
129  DenseVector x;
130 
131  int numEquations = A.rows();
132  int maxNumEquations = 100;
133  bool solveAsDense = (numEquations < maxNumEquations);
134 
135  if (!solveAsDense)
136  {
137 #ifndef NDEBUG
138  std::cout << "BSpline::Builder::computeBSplineCoefficients: Computing B-spline control points using sparse solver." << std::endl;
139 #endif // NDEBUG
140 
141  SparseLU<> s;
142  //bool successfulSolve = (s.solve(A,Bx,Cx) && s.solve(A,By,Cy));
143 
144  solveAsDense = !s.solve(A, b, x);
145  }
146 
147  if (solveAsDense)
148  {
149 #ifndef NDEBUG
150  std::cout << "BSpline::Builder::computeBSplineCoefficients: Computing B-spline control points using dense solver." << std::endl;
151 #endif // NDEBUG
152 
153  DenseMatrix Ad = A.toDense();
154  DenseQR<DenseVector> s;
155  // DenseSVD<DenseVector> s;
156  //bool successfulSolve = (s.solve(Ad,Bx,Cx) && s.solve(Ad,By,Cy));
157  if (!s.solve(Ad, b, x))
158  {
159  throw Exception("BSpline::Builder::computeBSplineCoefficients: Failed to solve for B-spline coefficients.");
160  }
161  }
162 
163  return x;
164 }

166 SparseMatrix BSpline::Builder::computeBasisFunctionMatrix(const BSpline &bspline) const
167 {
168  unsigned int numVariables = _data.getNumVariables();
169  unsigned int numSamples = _data.getNumSamples();
170 
171  // TODO: Reserve nnz per row (degree+1)
172  //int nnzPrCol = bspline.basis.supportedPrInterval();
173 
174  SparseMatrix A(numSamples, bspline.getNumBasisFunctions());
175  //A.reserve(DenseVector::Constant(numSamples, nnzPrCol)); // TODO: should reserve nnz per row!
176 
177  int i = 0;
178  for (auto it = _data.cbegin(); it != _data.cend(); ++it, ++i)
179  {
180  DenseVector xi(numVariables);
181  xi.setZero();
182  std::vector<double> xv = it->getX();
183  for (unsigned int j = 0; j < numVariables; ++j)
184  {
185  xi(j) = xv.at(j);
186  }
187 
188  SparseVector basisValues = bspline.evalBasis(xi);
189 
190  for (SparseVector::InnerIterator it2(basisValues); it2; ++it2)
191  {
192  A.insert(i,it2.index()) = it2.value();
193  }
194  }
195 
196  A.makeCompressed();
197 
198  return A;

212 /*
213 * Function for generating second order finite-difference matrix, which is used for penalizing the
214 * (approximate) second derivative in control point calculation for P-splines.
215 */
216 SparseMatrix BSpline::Builder::getSecondOrderFiniteDifferenceMatrix(const BSpline &bspline) const
217 {
218  unsigned int numVariables = bspline.getNumVariables();
219 
220  // Number of (total) basis functions - defines the number of columns in D
221  unsigned int numCols = bspline.getNumBasisFunctions();
222  std::vector<unsigned int> numBasisFunctions = bspline.getNumBasisFunctionsPerVariable();
223 
224  // Number of basis functions (and coefficients) in each variable
225  std::vector<unsigned int> dims;
226  for (unsigned int i = 0; i < numVariables; i++)
227  dims.push_back(numBasisFunctions.at(i));
228 
229  std::reverse(dims.begin(), dims.end());
230 
231  for (unsigned int i=0; i < numVariables; ++i)
232  if (numBasisFunctions.at(i) < 3)
233  throw Exception("BSpline::Builder::getSecondOrderDifferenceMatrix: Need at least three coefficients/basis function per variable.");
234 
235  // Number of rows in D and in each block
236  int numRows = 0;
237  std::vector< int > numBlkRows;
238  for (unsigned int i = 0; i < numVariables; i++)
239  {
240  int prod = 1;
241  for (unsigned int j = 0; j < numVariables; j++)
242  {
243  if (i == j)
244  prod *= (dims[j] - 2);
245  else
246  prod *= dims[j];
247  }
248  numRows += prod;
249  numBlkRows.push_back(prod);
250  }
251 
252  // Resize and initialize D
253  SparseMatrix D(numRows, numCols);
254  D.reserve(DenseVector::Constant(numCols,2*numVariables)); // D has no more than two elems per col per dim
255 
256  int i = 0; // Row index
257  // Loop though each dimension (each dimension has its own block)
258  for (unsigned int d = 0; d < numVariables; d++)
259  {
260  // Calculate left and right products
261  int leftProd = 1;
262  int rightProd = 1;
263  for (unsigned int k = 0; k < d; k++)
264  {
265  leftProd *= dims[k];
266  }
267  for (unsigned int k = d+1; k < numVariables; k++)
268  {
269  rightProd *= dims[k];
270  }
271 
272  // Loop through subblocks on the block diagonal
273  for (int j = 0; j < rightProd; j++)
274  {
275  // Start column of current subblock
276  int blkBaseCol = j*leftProd*dims[d];
277  // Block rows [I -2I I] of subblock
278  for (unsigned int l = 0; l < (dims[d] - 2); l++)
279  {
280  // Special case for first dimension
281  if (d == 0)
282  {
283  int k = j*leftProd*dims[d] + l;
284  D.insert(i,k) = 1;
285  k += leftProd;
286  D.insert(i,k) = -2;
287  k += leftProd;
288  D.insert(i,k) = 1;
289  i++;
290  }
291  else
292  {
293  // Loop for identity matrix
294  for (int n = 0; n < leftProd; n++)
295  {
296  int k = blkBaseCol + l*leftProd + n;
297  D.insert(i,k) = 1;
298  k += leftProd;
299  D.insert(i,k) = -2;
300  k += leftProd;
301  D.insert(i,k) = 1;
302  i++;
303  }
304  }
305  }
306  }
307  }
308 
309  D.makeCompressed();
310 
311  return D;
312 }

353 /*
354 * Automatic construction of (p+1)-regular knot vector
355 * using moving average.
356 *
357 * Requirement:
358 * Knot vector should be of size n+p+1.
359 * End knots are should be repeated p+1 times.
360 *
361 * Computed sizes:
362 * n+2*(p) = n + p + 1 + (p - 1)
363 * k = (p - 1) values must be removed from sample vector.
364 * w = k + 3 window size in moving average
365 *
366 * Algorithm:
367 * 1) compute n - k values using moving average with window size w
368 * 2) repeat first and last value p + 1 times
369 *
370 * The resulting knot vector has n - k + 2*p = n + p + 1 knots.
371 *
372 * NOTE:
373 * For equidistant samples, the resulting knot vector is identically to
374 * the free end conditions knot vector used in cubic interpolation.
375 * That is, samples (a,b,c,d,e,f) produces the knot vector (a,a,a,a,c,d,f,f,f,f) for p = 3.
376 * For p = 1, (a,b,c,d,e,f) becomes (a,a,b,c,d,e,f,f).
377 *
378 * TODO:
379 * Does not work well when number of knots is << number of samples! For such cases
380 * almost all knots will lie close to the left samples. Try a bucket approach, where the
381 * samples are added to buckets and the knots computed as the average of these.
382 */
383 std::vector<double> BSpline::Builder::knotVectorMovingAverage(const std::vector<double> &values,
384  unsigned int degree) const
385 {
386  // Sort and remove duplicates
387  std::vector<double> unique = extractUniqueSorted(values);
388 
389  // Compute sizes
390  unsigned int n = unique.size();
391  unsigned int k = degree-1; // knots to remove
392  unsigned int w = k + 3; // Window size
393 
394  // The minimum number of samples from which a free knot vector can be created
395  if (n < degree+1)
396  {
397  std::ostringstream e;
398  e << "knotVectorMovingAverage: Only " << n
399  << " unique interpolation points are given. A minimum of degree+1 = " << degree+1
400  << " unique points are required to build a B-spline basis of degree " << degree << ".";
401  throw Exception(e.str());
402  }
403 
404  std::vector<double> knots(n-k-2, 0);
405 
406  // Compute (n-k-2) interior knots using moving average
407  for (unsigned int i = 0; i < n-k-2; ++i)
408  {
409  double ma = 0;
410  for (unsigned int j = 0; j < w; ++j)
411  ma += unique.at(i+j);
412 
413  knots.at(i) = ma/w;
414  }
415 
416  // Repeat first knot p + 1 times (for interpolation of start point)
417  for (unsigned int i = 0; i < degree + 1; ++i)
418  knots.insert(knots.begin(), unique.front());
419 
420  // Repeat last knot p + 1 times (for interpolation of end point)
421  for (unsigned int i = 0; i < degree + 1; ++i)
422  knots.insert(knots.end(), unique.back());
423 
424  // Number of knots in a (p+1)-regular knot vector
425  //assert(knots.size() == uniqueX.size() + degree + 1);
426 
427  return knots;
428 }

472 std::vector<double> BSpline::Builder::knotVectorBuckets(const std::vector<double> &values, unsigned int degree, unsigned int maxSegments) const
473 {
474  // Sort and remove duplicates
475  std::vector<double> unique = extractUniqueSorted(values);
476 
477  // The minimum number of samples from which a free knot vector can be created
478  if (unique.size() < degree+1)
479  {
480  std::ostringstream e;
481  e << "BSpline::Builder::knotVectorBuckets: Only " << unique.size()
482  << " unique sample points are given. A minimum of degree+1 = " << degree+1
483  << " unique points are required to build a B-spline basis of degree " << degree << ".";
484  throw Exception(e.str());
485  }
486 
487  // Num internal knots (0 <= ni <= unique.size() - degree - 1)
488  unsigned int ni = unique.size() - degree - 1;
489 
490  // Num segments
491  unsigned int ns = ni + degree + 1;
492 
493  // Limit number of segments
494  if (ns > maxSegments && maxSegments >= degree + 1)
495  {
496  ns = maxSegments;
497  ni = ns - degree - 1;
498  }
499 
500  // Num knots
501 // unsigned int nk = ns + degree + 1;
502 
503  // Check numbers
504  if (ni > unique.size() - degree - 1)
505  throw Exception("BSpline::Builder::knotVectorBuckets: Invalid number of internal knots!");
506 
507  // Compute window sizes
508  unsigned int w = 0;
509  if (ni > 0)
510  w = std::floor(unique.size()/ni);
511 
512  // Residual
513  unsigned int res = unique.size() - w*ni;
514 
515  // Create array with window sizes
516  std::vector<unsigned int> windows(ni, w);
517 
518  // Add residual
519  for (unsigned int i = 0; i < res; ++i)
520  windows.at(i) += 1;
521 
522  // Compute internal knots
523  std::vector<double> knots(ni, 0);
524 
525  // Compute (n-k-2) interior knots using moving average
526  unsigned int index = 0;
527  for (unsigned int i = 0; i < ni; ++i)
528  {
529  for (unsigned int j = 0; j < windows.at(i); ++j)
530  {
531  knots.at(i) += unique.at(index+j);
532  }
533  knots.at(i) /= windows.at(i);
534  index += windows.at(i);
535  }
536 
537  // Repeat first knot p + 1 times (for interpolation of start point)
538  for (unsigned int i = 0; i < degree + 1; ++i)
539  knots.insert(knots.begin(), unique.front());
540 
541  // Repeat last knot p + 1 times (for interpolation of end point)
542  for (unsigned int i = 0; i < degree + 1; ++i)
543  knots.insert(knots.end(), unique.back());
544 
545  return knots;
546 }
