160 int main(int argc, char *argv[])
161 {
162  // Create the example object of the tutorial02 type
163  tutorial02 example(argc, argv);
164 
165  // Read some parameters from file
166  ITHACAparameters para;
167  int NmodesTout = para.ITHACAdict->lookupOrDefault<int>("NmodesTout", 15);
168  int NmodesTproj = para.ITHACAdict->lookupOrDefault<int>("NmodesTproj", 10);
169 
170  // Set the number of parameters
171  example.Pnumber = 9;
172  // Set the parameters
173  example.setParameters();
174 
175  // Set the parameter ranges, in all the subdomains the diffusivity varies between
176  // 0.001 and 0.1
177  example.mu_range.col(0) = Eigen::MatrixXd::Ones(9, 1) * 0.001;
178  example.mu_range.col(1) = Eigen::MatrixXd::Ones(9, 1) * 0.1;
179 
180  // Generate the Parameters
181  example.genRandPar(500);
182  // Set the size of the list of values that are multiplying the affine forms
183  example.theta.resize(9);
184 
185  // Set the source term
186  example.SetSource();
187  // Compute the diffusivity field for each subdomain
188  example.compute_nu();
189  // Assemble all the operators of the affine decomposition
190  example.assemble_operator();
191  // Perform an Offline Solve
192  example.offlineSolve();
193 
194  // Perform a POD decomposition and get the modes
195  ITHACAPOD::getModes(example.Tfield, example.Tmodes, example.podex, 0, 0, 15);
196 
197  // Perform the Galerkin projection onto the space spanned by the POD modes
199  example.project(10);
201 
202  // Create a reduced object
203  reducedLaplacian ridotto(example);
204 
205  // Solve the online reduced problem some new values of the parameters
206  for (int i = 0; i < 10; i++)
207  {
208  ridotto.solveOnline(example.mu.row(i));
209  }
210 
211  // Reconstruct the solution and store it into Reconstruction folder
212  ridotto.reconstruct("./ITHACAoutput/Reconstruction");
213  // Exit the code
214  exit(0);
215 }
