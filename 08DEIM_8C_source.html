42 class DEIM_function : public DEIM<volScalarField>
43 {
44  public:
45  using DEIM::DEIM;
46  static volScalarField evaluate_expression(volScalarField& S, Eigen::MatrixXd mu)
47  {
48  volScalarField yPos = S.mesh().C().component(vector::Y);
49  volScalarField xPos = S.mesh().C().component(vector::X);
50 
51  for (auto i = 0; i < S.size(); i++)
52  {
53  S[i] = std::exp( - 2 * std::pow(xPos[i] - mu(0) - 1,
54  2) - 2 * std::pow(yPos[i] - mu(1) - 0.5, 2));
55  }
56 
57  return S;
58  }
59  Eigen::VectorXd onlineCoeffs(Eigen::MatrixXd mu)
60  {
61  theta.resize(fields.size());
62 
63  for (int i = 0; i < fields.size(); i++)
64  {
65  double on_coeff = evaluate_expression(fields[i], mu)[localMagicPoints[i]];
66  theta(i) = on_coeff;
67  }
68 
69  return theta;
70  }
71 
72  PtrList<volScalarField> fields;
73 };

94  // List of volScalarField where the snapshots are stored
95  PtrList<volScalarField> Sp;
96  // Non linear function
97  volScalarField S
98  (
99  IOobject
100  (
101  "S",
102  runTime.timeName(),
103  mesh,
104  IOobject::NO_READ,
105  IOobject::AUTO_WRITE
106  ),
107  T.mesh(),
108  dimensionedScalar("zero", dimensionSet(0, 0, -1, 1, 0, 0, 0), 0)
109  );
110  // Parameters used to train the non-linear function
111  Eigen::MatrixXd pars = ITHACAutilities::rand(100, 2, -0.5, 0.5);
112 
113  // Perform the offline phase
114  for (int i = 0; i < 100; i++)
115  {
116  DEIM_function::evaluate_expression(S, pars.row(i));
117  Sp.append(S);
118  ITHACAstream::exportSolution(S, name(i + 1), "./ITHACAoutput/Offline/");
119  }
120 
121  // Create DEIM object with given number of basis functions
122  DEIM_function c(Sp, NDEIM, "Gaussian_function");
123  // Generate the submeshes with the depth of the layer
124  c.fields = c.generateSubmeshes(2, mesh, S);
125  // Define a new online parameter
126  Eigen::MatrixXd par_new(2, 1);
127  par_new(0, 0) = 0;
128  par_new(1, 0) = 0;
129  // Online evaluation of the non linear function
130  Eigen::VectorXd aprfield = c.MatrixOnline * c.onlineCoeffs(par_new);
131  // Transform to an OpenFOAM field and export
132  volScalarField S2("S_online", Foam2Eigen::Eigen2field(S, aprfield));
133  ITHACAstream::exportSolution(S2, name(1), "./ITHACAoutput/Online/");
134  // Evaluate the full order function and export it
135  DEIM_function::evaluate_expression(S, par_new);
136  ITHACAstream::exportSolution(S, name(1), "./ITHACAoutput/Online/");
137  // Compute the L2 error and print it
138  Info << ITHACAutilities::errorL2Rel(S2, S) << endl;
139  return 0;
140 }
