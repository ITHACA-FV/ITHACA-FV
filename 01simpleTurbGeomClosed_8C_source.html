75  // This function creates the NN, trains it and saves the weights
76  void getTurbNN()
77  {
78  Eigen::MatrixXd NNdata(2, 1);
79 
80  if (ITHACAutilities::check_file("./ITHACAoutput/Offline/NN/NNdata.txt"))
81  {
82  NNdata = ITHACAstream::readMatrix("./ITHACAoutput/Offline/NN/NNdata.txt");
83  }

85  // If the NN does not exist or input/output dimensions are changed, a new NN is constructed
86  if (!ITHACAutilities::check_file("./ITHACAoutput/Offline/NN/Net.pt") ||
87  NNdata(0, 0) != NUmodes || NNdata(1, 0) != NNutModes)
88  {
89  // Since offline velocity and eddy viscosity are used to train the NN, they are read jyst in case
90  // it has not been done during the offline phase
91  if (nutFields.size() == 0 || Ufield.size() == 0)
92  {
93  ITHACAstream::readMiddleFields(Ufield, _U(), "./ITHACAoutput/Offline/");
94  auto nut = _mesh().lookupObject<volScalarField>("nut");
95  ITHACAstream::readMiddleFields(nutFields, nut, "./ITHACAoutput/Offline/");
96  }
97 
98  coeffL2Nut = ITHACAutilities::getCoeffs(nutFields, nutModes,
99  NNutModes, true);
100  coeffL2Nut.transposeInPlace();
101  coeffL2U = ITHACAutilities::getCoeffs(Ufield, Umodes,
102  NUmodes, true);
103  coeffL2U.transposeInPlace();
104  bias_inp = coeffL2U.colwise().minCoeff();
105  scale_inp = coeffL2U.colwise().maxCoeff() - coeffL2U.colwise().minCoeff();
106  bias_out = coeffL2Nut.colwise().minCoeff();
107  scale_out = coeffL2Nut.colwise().maxCoeff() - coeffL2Nut.colwise().minCoeff();
108 
109  for (unsigned int i = 0; i < coeffL2U.rows(); i++)
110  {
111  coeffL2U.row(i) = (coeffL2U.row(i).array() - bias_inp.array()).array()
112  / scale_inp.array();
113  }
114 
115  for (unsigned int i = 0; i < coeffL2Nut.rows(); i++)
116  {
117  coeffL2Nut.row(i) = (coeffL2Nut.row(i).array() - bias_out.array()).array()
118  / scale_out.array();
119  }
120 
121  coeffL2U_tensor = eigenMatrix2torchTensor(coeffL2U);
122  coeffL2Nut_tensor = eigenMatrix2torchTensor(coeffL2Nut);
123  ITHACAparameters* para = ITHACAparameters::getInstance();
124  int epochs = para->ITHACAdict->lookupOrDefault<int>("epochs", 20000);
125 
126  for (int64_t epoch = 1; epoch <= epochs; ++epoch)
127  {
128  std::cerr << epoch << std::endl;
129  Net->zero_grad();
130  torch::Tensor y_r = Net->forward(coeffL2U_tensor);
131  torch::Tensor d_loss_real = torch::nn::functional::mse_loss(y_r,
132  coeffL2Nut_tensor);
133  std::cout << d_loss_real.item<float>() << std::endl;
134  d_loss_real.backward();
135  optimizer->step();
136  }
137

140  torch::save(Net, "./ITHACAoutput/Offline/NN/Net.pt");

155  // If the NN already exists, all the weights are just read
156  else
157  {
158  torch::load(Net, "./ITHACAoutput/Offline/NN/Net.pt");
159  ITHACAstream::ReadDenseMatrix(bias_inp, "./ITHACAoutput/Offline/NN/",

161  ITHACAstream::ReadDenseMatrix(scale_inp, "./ITHACAoutput/Offline/NN/",

170  // Function to eval the NN once the input is provided
171  Eigen::MatrixXd evalNet(Eigen::MatrixXd a)
172  {
173  a.transposeInPlace();
174  a = (a.array() - bias_inp.array()) / scale_inp.array();
175  torch::Tensor aTensor = eigenMatrix2torchTensor(a);
176  torch::Tensor out = Net->forward(aTensor);
177  Eigen::MatrixXd g = torchTensor2eigenMatrix<double>(out);
178  g = g.array() * scale_out.array() + bias_out.array();
179  g.transposeInPlace();
180  return g;

197  void solveOnline_Simple(scalar mu_now,

199  word Folder = "./ITHACAoutput/Reconstruct/")

201  counter++;

204  // the requested number into the ITHACAdict to zero
205  if (NmodesUproj == 0)
206  {
207  NmodesUproj = problem->Umodes.size();
208  }
209 
210  if (NmodesPproj == 0)
211  {
212  NmodesPproj = problem->Pmodes.size();
213  }
214 
215  if (NmodesNut == 0)
216  {
217  NmodesNut = problem->nutModes.size();
218  }
219 
220  // Initializations
221  Eigen::VectorXd uresidualOld = Eigen::VectorXd::Zero(NmodesUproj);
222  Eigen::VectorXd presidualOld = Eigen::VectorXd::Zero(NmodesPproj);
223  Eigen::VectorXd uresidual = Eigen::VectorXd::Zero(NmodesUproj);
224  Eigen::VectorXd presidual = Eigen::VectorXd::Zero(NmodesPproj);
225  scalar U_norm_res(1);
226  scalar P_norm_res(1);
227  Eigen::MatrixXd a = Eigen::VectorXd::Zero(NmodesUproj);
228  Eigen::MatrixXd a0 = Eigen::VectorXd::Zero(NmodesUproj);
229  Eigen::MatrixXd b = Eigen::VectorXd::Zero(NmodesPproj);
230  Eigen::MatrixXd b0 = Eigen::VectorXd::Zero(NmodesPproj);
231  Eigen::MatrixXd nutCoeff = Eigen::VectorXd::Zero(NmodesNut);
232  float residualJumpLim =
233  problem->para->ITHACAdict->lookupOrDefault<float>("residualJumpLim", 1e-5);
234  float normalizedResidualLim =
235  problem->para->ITHACAdict->lookupOrDefault<float>("normalizedResidualLim",
236  1e-5);
237  scalar residual_jump(1 + residualJumpLim);
238  volScalarField& P = problem->_p();
239  volVectorField& U = problem->_U();
240  volScalarField& nut = const_cast<volScalarField&>
241  (problem->_mesh().lookupObject<volScalarField>("nut"));
242  volVectorField u2 = U;
243  a0 = ITHACAutilities::getCoeffs(U, problem->Umodes, NmodesUproj, true);
244  b = ITHACAutilities::getCoeffs(P, problem->Pmodes, NmodesPproj, true);
245  nutCoeff = ITHACAutilities::getCoeffs(nut, problem->nutModes, NmodesNut, true);
246  a(0) = a0(0); // Do not remove: it is not working without this condition
247  b = b0;
248  fvMesh& mesh = problem->_mesh();
249  Time& runTime = problem->_runTime();
250  P.rename("p");
251  surfaceScalarField& phi(problem->_phi());
252  problem->Umodes.reconstruct(U, a, "U");
253  Info << ITHACAutilities::errorL2Rel(U, u2) << endl;
254  problem->Pmodes.reconstruct(P, b, "p");
255  problem->nutModes.reconstruct(nut, nutCoeff, "nut");
256  phi = fvc::flux(U);
257  int iter = 0;
258  simpleControl& simple = problem->_simple();
259 
260  if (ITHACAutilities::isTurbulent())
261  {
262  Eigen::MatrixXd nutCoeff = problem->evalNet(a);
263  problem->nutModes.reconstruct(nut, nutCoeff, "nut");
264  ITHACAstream::exportSolution(nut, name(counter), Folder);
265  }
266 
267  PtrList<volVectorField> gradModP;
268 
269  for (label i = 0; i < NmodesPproj; i++)
270  {
271  gradModP.append(fvc::grad(problem->Pmodes[i]));
272  }
273 
274  projGradModP = (problem->Umodes).project(gradModP, NmodesUproj);
275 
276  // SIMPLE algorithm starts here
277  while ((residual_jump > residualJumpLim
278  || std::max(U_norm_res, P_norm_res) > normalizedResidualLim) &&
279  iter < maxIterOn)
280  {
281  iter++;
282  std::cout << iter << std::endl;
283 #if OFVER == 6
284  simple.loop(runTime);
285 #else
286  simple.loop();
287 #endif
288  volScalarField nueff = problem->turbulence->nuEff();
289  vector v(1, 0, 0);
290  ITHACAutilities::assignBC(U, 0, v);
291 
292  // If the case is turbulent, then the network is evaluated
293  if (ITHACAutilities::isTurbulent())
294  {
295  Eigen::MatrixXd nutCoeff = problem->evalNet(a);
296  volScalarField& nut = const_cast<volScalarField&>
297  (problem->_mesh().lookupObject<volScalarField>("nut"));
298  problem->nutModes.reconstruct(nut, nutCoeff, "nut");
299  ITHACAstream::exportSolution(nut, name(counter), Folder);
300  }
301 
302  // Momentum equation
303  fvVectorMatrix UEqn
304  (
305  fvm::div(phi, U)
306  - fvm::laplacian(nueff, U)
307  - fvc::div(nueff * dev2(T(fvc::grad(U))))
308  );
309  UEqn.relax();
310  List<Eigen::MatrixXd> RedLinSysU = problem->Umodes.project(UEqn, NmodesUproj);
311  RedLinSysU[1] = RedLinSysU[1] - projGradModP * b;
312  a = reducedProblem::solveLinearSys(RedLinSysU, a, uresidual);
313  problem->Umodes.reconstruct(U, a, "U");
314  ITHACAutilities::assignBC(U, 0, v);
315  volScalarField rAU(1.0 / UEqn.A());
316  volVectorField HbyA(constrainHbyA(1.0 / UEqn.A() * UEqn.H(), U, P));
317  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
318  adjustPhi(phiHbyA, U, P);
319  tmp<volScalarField> rAtU(rAU);
320 
321  if (simple.consistent())
322  {
323  rAtU = 1.0 / (1.0 / rAU - UEqn.H1());
324  phiHbyA +=
325  fvc::interpolate(rAtU() - rAU) * fvc::snGrad(P) * mesh.magSf();
326  HbyA -= (rAU - rAtU()) * fvc::grad(P);
327  }
328 
329  List<Eigen::MatrixXd> RedLinSysP;
330 
331  while (simple.correctNonOrthogonal())
332  {
333  // Continuity equation
334  fvScalarMatrix pEqn
335  (
336  fvm::laplacian(rAtU(), P) == fvc::div(phiHbyA)
337  );
338  RedLinSysP = problem->Pmodes.project(pEqn, NmodesPproj);
339  b = reducedProblem::solveLinearSys(RedLinSysP, b, presidual);
340  problem->Pmodes.reconstruct(P, b, "p");
341 
342  if (simple.finalNonOrthogonalIter())
343  {
344  phi = phiHbyA - pEqn.flux();
345  }
346  }
347 
348  P.relax();
349  U = HbyA - rAtU() * fvc::grad(P);
350  U.correctBoundaryConditions();
351  uresidualOld = uresidualOld - uresidual;
352  presidualOld = presidualOld - presidual;
353  uresidualOld = uresidualOld.cwiseAbs();
354  presidualOld = presidualOld.cwiseAbs();
355  residual_jump = std::max(uresidualOld.sum(), presidualOld.sum());
356  uresidualOld = uresidual;
357  presidualOld = presidual;
358  uresidual = uresidual.cwiseAbs();
359  presidual = presidual.cwiseAbs();
360  U_norm_res = uresidual.sum() / (RedLinSysU[1].cwiseAbs()).sum();
361  P_norm_res = presidual.sum() / (RedLinSysP[1].cwiseAbs()).sum();
362 
363  if (problem->para->debug)
364  {
365  std::cout << "Residual jump = " << residual_jump << std::endl;
366  std::cout << "Normalized residual = " << std::max(U_norm_res,
367  P_norm_res) << std::endl;
368  std::cout << "Final normalized residual for velocity: " << U_norm_res <<
369  std::endl;
370  std::cout << "Final normalized residual for pressure: " << P_norm_res <<
371  std::endl;
372  }
373  }
374 
375  std::cout << "Solution " << counter << " converged in " << iter <<
376  " iterations." << std::endl;
377  std::cout << "Final normalized residual for velocity: " << U_norm_res <<
378  std::endl;
379  std::cout << "Final normalized residual for pressure: " << P_norm_res <<
380  std::endl;
381  problem->Umodes.reconstruct(U, a, "Uaux");
382  problem->Pmodes.reconstruct(P, b, "Paux");
383  ITHACAstream::exportSolution(U, name(counter), Folder);
384  ITHACAstream::exportSolution(P, name(counter), Folder);
385  runTime.setTime(runTime.startTime(), 0);
386  }

471  void linearMovePts(double angle, List<vector>& points2Move)
472  {
473  double sMax;
474  scalarList x;
475  scalarList y;
476 
477  for (label i = 0; i < points2Move.size(); i++)
478  {
479  x.append(points2Move[i].component(0));
480  y.append(points2Move[i].component(1));
481  }
482 
483  double xMin = min(x);
484  double xMax = max(x);
485  double yMin = min(y);
486  double yMax = max(y);
487  sMax = (yMax - yMin) * std::tan(M_PI * angle / 180);
488 
489  for (label i = 0; i < points2Move.size(); i++)
490  {
491  points2Move[i].component(0) = points2Move[i].component(0) +
492  (yMax - points2Move[i].component(1)) / (yMax - yMin) * (xMax -
493  points2Move[i].component(0)) / (xMax - xMin) * sMax;
494  }
495  }
