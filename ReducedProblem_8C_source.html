64 Eigen::MatrixXd reducedProblem::solveLinearSys(List<Eigen::MatrixXd> LinSys,
65  Eigen::MatrixXd x, Eigen::VectorXd& residual, const Eigen::MatrixXd& bc,
66  const std::string solverType)
67 
68 {
69  M_Assert(solverType == "fullPivLu" || solverType == "partialPivLu"
70  || solverType == "householderQR" || solverType == "colPivHouseholderQR"
71  || solverType == "fullPivHouseholderQR"
72  || solverType == "CompleteOrthogonalDecomposition" || solverType == "llt"
73  || solverType == "ldlt" || solverType == "bdcSvd"
74  || solverType == "jacobiSvd", "solver not defined");
75  Eigen::MatrixXd y;
76 
77  for (int i = 0; i < bc.size(); i++)
78  {
79  LinSys[0].row(i) *= 0;
80  LinSys[0](i, i) = 1;
81  LinSys[1](i, 0) = bc(i);
82  }
83 
84  residual = LinSys[0] * x - LinSys[1];
85 
86  if (solverType == "fullPivLu")
87  {
88  y = LinSys[0].fullPivLu().solve(LinSys[1]);
89  }
90  else if (solverType == "partialPivLu")
91  {
92  y = LinSys[0].partialPivLu().solve(LinSys[1]);
93  }
94  else if (solverType == "householderQr")
95  {
96  y = LinSys[0].householderQr().solve(LinSys[1]);
97  }
98  else if (solverType == "colPivHouseholderQr")
99  {
100  y = LinSys[0].colPivHouseholderQr().solve(LinSys[1]);
101  }
102  else if (solverType == "fullPivHouseholderQr")
103  {
104  y = LinSys[0].fullPivHouseholderQr().solve(LinSys[1]);
105  }
106  else if (solverType == "completeOrthogonalDecomposition")
107  {
108  y = LinSys[0].completeOrthogonalDecomposition().solve(LinSys[1]);
109  }
110  else if (solverType == "llt")
111  {
112  y = LinSys[0].llt().solve(LinSys[1]);
113  }
114  else if (solverType == "ldlt")
115  {
116  y = LinSys[0].ldlt().solve(LinSys[1]);
117  }
118  else if (solverType == "bdcSvd")
119  {
120  y = LinSys[0].bdcSvd(Eigen::ComputeThinU | Eigen::ComputeThinV).solve(
121  LinSys[1]);
122  }
123  else if (solverType == "jacobiSvd")
124  {
125  y = LinSys[0].jacobiSvd().solve(LinSys[1]);
126  }
127 
128  return y;
129 }

154 Eigen::MatrixXd onlineInterp::getInterpCoeffRBF(std::vector<SPLINTER::RBFSpline>
155  rbfVec, Eigen::MatrixXd mu_interp)
156 {
157  M_Assert(mu_interp.cols() == Nmu_samples,
158  "Matrix 'mu_interp' must have same number and order of columns (i.e. parameters) as the matrix 'mu_samples'.");
159  int Nmodes = rbfVec.size();
160  // Nsamples (snapshots) to evaluate the interpolator at
161  int Nsamples = mu_interp.rows();
162  Eigen::MatrixXd muInterpRefined(mu_interp.rows(), 0);
163 
164  // Discard columns of mu_interp that have constant value for all elements, and use a refined matrix muInterpRefined
165  for (int i = 0; i < mu_interp.cols(); i++)
166  {
167  double firstVal = mu_interp(0, i);
168  bool isConst = (mu_interp.col(i).array() == firstVal).all();
169 
170  // Consider only columns with non-constant values
171  if (isConst == 0)
172  {
173  muInterpRefined.conservativeResize(muInterpRefined.rows(),
174  muInterpRefined.cols() + 1);
175  muInterpRefined.col(muInterpRefined.cols() - 1) = mu_interp.col(i);
176  }
177  }
178 
179  int Nmu = muInterpRefined.cols();
180  Eigen::MatrixXd coeff_interp(Nmodes, Nsamples);
181 
182  for (int i = 0; i < Nmodes; i++)
183  {
184  for (int j = 0; j < Nsamples; j++)
185  {
186  SPLINTER::DenseVector x(Nmu);
187 
188  for (int k = 0; k < Nmu; k++)
189  {
190  x(k) = muInterpRefined(j, k);
191  }
192 
193  coeff_interp(i, j) = rbfVec[i].eval(x);
194  }
195  }
196 
197  return coeff_interp;
198 }
