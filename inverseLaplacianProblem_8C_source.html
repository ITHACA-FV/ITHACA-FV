121 void inverseLaplacianProblem::set_valueFraction()
122 {
123  fvMesh& mesh = _mesh();
124  valueFraction.resize(mesh.boundaryMesh()["coldSide"].size());
125  homogeneousBCcoldSide.resize(mesh.boundaryMesh()["coldSide"].size());
126  coldSide_ind = mesh.boundaryMesh().findPatchID("coldSide");
127  Eigen::VectorXd faceCellDist =
128  ITHACAutilities::boudaryFaceToCellDistance(mesh, coldSide_ind);
129  forAll (valueFraction, faceI)
130  {
131  scalar faceDist = faceCellDist(faceI);
132  valueFraction[faceI] = 1.0 / (1.0 + (k / H / faceDist));
133  homogeneousBCcoldSide[faceI] = 0;
134  }
135  refGrad = homogeneousBCcoldSide;
136 }

139 void inverseLaplacianProblem::assignDirectBC()
140 {
141  fvMesh& mesh = _mesh();
142  volScalarField& T = _T();
143  set_valueFraction();
144  forAll(mesh.boundaryMesh(), patchI)
145  {
146  if (patchI == mesh.boundaryMesh().findPatchID("coldSide"))
147  {
148  ITHACAutilities::assignMixedBC(T, patchI, Tf, refGrad, valueFraction);
149  }
150  else if (patchI == mesh.boundaryMesh().findPatchID("hotSide"))
151  {
152  ITHACAutilities::assignBC(T, patchI, - g / k);
153  }
154  else
155  {
156  ITHACAutilities::assignBC(T, patchI, homogeneousBC);
157  }
158  }
159 }

168 void inverseLaplacianProblem::solve(const char* problemID)
169 {
170  volScalarField& T = _T();
171  simpleControl& simple = _simple();
172  Foam::Time& runTime = _runTime();
173 
174  if (strcmp( problemID, "direct") == 0)
175  {
176  ITHACAutilities::assignIF(T, homogeneousBC);
177  }
178  else
179  {
180  Info << "Problem name should be direct or sensitivity" << endl;
181  exit(10);
182  }
183 
184 #if OFVER == 6
185 
186  while (simple.loop(runTime))
187 #else
188  while (simple.loop())
189 #endif
190  {
191  while (simple.correctNonOrthogonal())
192  {
193  if (strcmp( problemID, "direct") == 0)
194  {
195  fvScalarMatrix TEqn
196  (
197  fvm::laplacian(DT, T)
198  );
199  TEqn.solve();
200  }
201  }
202  }

205 void inverseLaplacianProblem::readThermocouples()
206 {
207  if (!thermocouplesRead)
208  {
209  word fileName = "./thermocouplesCellsID";
210 
211  if (ITHACAutilities::check_file(fileName + "_mat.txt"))
212  {
213  Info << "Reading thermocouples cells from file" << endl;
214  Eigen::MatrixXi TCmatrix = ITHACAstream::readMatrix(fileName + "_mat.txt").cast
215  <int> ();
216  thermocouplesCellID = Foam2Eigen::EigenMatrix2List(TCmatrix);
217  }
218  else
219  {
220  Info << "Defining positions of thermocouples" << endl;
221  fvMesh& mesh = _mesh();
222  volScalarField& T = _T();
223  thermocouplesCellID.resize(thermocouplesPos.size());
224  forAll(thermocouplesPos, tcI)
225  {
226  thermocouplesCellID[tcI] = mesh.findCell(thermocouplesPos[tcI]);
227  }
228  volScalarField thermocouplesField(T);
229  ITHACAutilities::assignIF(thermocouplesField, homogeneousBC);
230  forAll(thermocouplesCellID, tcI)
231  {
232  thermocouplesField.ref()[thermocouplesCellID[tcI]] = 1;
233  }
234  ITHACAstream::exportSolution(thermocouplesField, "1", "./ITHACAoutput/debug/",
235  "thermocouplesField,");
236  Eigen::MatrixXi thermocouplesCellID_eigen = Foam2Eigen::List2EigenMatrix(
237  thermocouplesCellID);
238  ITHACAstream::exportMatrix(thermocouplesCellID_eigen, fileName,
239  "eigen", "./");
240  }
241 
242  thermocouplesRead = 1;
243  thermocouplesNum = thermocouplesPos.size();
244  }
245  else
246  {
247  WarningInFunction << "readThermocouples function called twice." << endl;
248  WarningInFunction << "I am not doing the second reading." << endl;
249  }
