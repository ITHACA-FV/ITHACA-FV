115  // Method to compute the lifting function for temperature
116  void liftSolveT()
117  {
118  for (label k = 0; k < inletIndexT.rows(); k++)
119  {
120  Time& runTime = _runTime();
121  fvMesh& mesh = _mesh();
122  volScalarField T = _T();
123  simpleControl simple(mesh);
124  volScalarField& alphat = _alphat();
125  dimensionedScalar& Pr = _Pr();
126  dimensionedScalar& Prt = _Prt();
127  label BCind = inletIndexT(k, 0);
128  volScalarField Tlift("Tlift" + name(k), T);
129  instantList Times = runTime.times();
130  runTime.setTime(Times[1], 1);
131  Info << "Solving a lifting Problem" << endl;
132  scalar t1 = 1;
133  scalar t0 = 0;
134 
135  for (label j = 0; j < T.boundaryField().size(); j++)
136  {
137  assignIF(Tlift, t0);
138 
139  if (j == BCind)
140  {
141  assignBC(Tlift, j, t1);
142  }
143  else if (T.boundaryField()[BCind].type() == "fixedValue")
144  {
145  assignBC(Tlift, j, t0);
146  }
147  else
148  {
149  }
150  }
151 
152  while (simple.correctNonOrthogonal())
153  {
154  alphat = turbulence->nut() / Prt;
155  alphat.correctBoundaryConditions();
156  volScalarField alphaEff("alphaEff", turbulence->nu() / Pr + alphat);
157  fvScalarMatrix TEqn
158  (
159  fvm::laplacian(alphaEff, Tlift)
160  );
161  TEqn.solve();
162  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
163  << " ClockTime = " << runTime.elapsedClockTime() << " s"
164  << nl << endl;
165  }
166 
167  Tlift.write();
168  liftfieldT.append(Tlift);
169  }
170  }
171 };

177 int main(int argc, char* argv[])
178 {
179  // Construct the tutorial object
180  tutorial10 example(argc, argv);
181  // the offline samples for the boundary conditions
182  word par_offline_BC("./par_offline_BC");
183  Eigen::MatrixXd par_off_BC = ITHACAstream::readMatrix(par_offline_BC);
184  // the samples which will be used for setting the boundary condition in the online stage
185  word par_online_BC("./par_online_BC");
186  Eigen::MatrixXd par_on_BC = ITHACAstream::readMatrix(par_online_BC);
187  // Read some parameters from file
188  ITHACAparameters para;
189  int NmodesUproj = para.ITHACAdict->lookupOrDefault<int>("NmodesUproj", 5);
190  int NmodesPproj = para.ITHACAdict->lookupOrDefault<int>("NmodesPproj", 5);
191  int NmodesTproj = para.ITHACAdict->lookupOrDefault<int>("NmodesTproj", 5);
192  int NmodesSUPproj = para.ITHACAdict->lookupOrDefault<int>("NmodesSUPproj", 5);
193  int NmodesOut = para.ITHACAdict->lookupOrDefault<int>("NmodesOut", 15);
195  example.Pnumber = 1;
197  example.Tnumber = 1;
199  example.setParameters();
201  example.mu_range(0, 0) = 0.00001;
202  example.mu_range(0, 1) = 0.00001;
203  // Generate equispaced samples inside the parameter range
204  example.genEquiPar();
206  example.inletIndexT.resize(2, 1);
207  example.inletIndexT << 1, 2;
209  example.startTime = 0.0;
210  example.finalTime = 10.0;
211  example.timeStep = 0.005;
212  example.writeEvery = 0.01;
213  // Perform the Offline Solve;
214  example.offlineSolve(par_off_BC);
215  // Search the lift function for the temperature
216  example.liftSolveT();
217  // Create homogeneous basis functions for temperature
218  example.computeLiftT(example.Tfield, example.liftfieldT, example.Tomfield);
219  // Perform a POD decomposition for velocity temperature and pressure fields
220  ITHACAPOD::getModes(example.Ufield, example.Umodes, example.podex, 0, 0,
221  NmodesOut);
222  ITHACAPOD::getModes(example.Tomfield, example.Tmodes, example.podex, 0, 0,
223  NmodesOut);
224  // Create a list with number of modes for which the projection needs to be performed
225  Eigen::MatrixXd List_of_modes(NmodesOut - 5, 1);
226 
227  for (int i = 0; i < List_of_modes.rows(); i++)
228  {
229  List_of_modes(i, 0) = i + 1;
230  }
231 
232  // Export with number of modes for which the projection needs to be performed
233  ITHACAstream::exportMatrix(List_of_modes, "List_of_modes", "eigen",
234  "./ITHACAoutput/l2error");
235  // Create locally the temperature modes
236  PtrList<volScalarField> TLmodes;
237 
238  for (label k = 0; k < example.liftfieldT.size(); k++)
239  {
240  TLmodes.append(example.liftfieldT[k]);
241  }
242 
243  for (label k = 0; k < List_of_modes.size(); k++)
244  {
245  TLmodes.append(example.Tmodes[k]);
246  }
247 
248  // Perform the projection for all number of modes in list List_of_modes
249  Eigen::MatrixXd L2errorProjMatrixU(example.Ufield.size(), List_of_modes.rows());
250  Eigen::MatrixXd L2errorProjMatrixT(example.Tfield.size(), List_of_modes.rows());
251 
252  // Calculate the coefficients and L2 error and store the error in a matrix for each number of modes
253  for (int i = 0; i < List_of_modes.rows(); i++)
254  {
255  Eigen::MatrixXd coeffU = ITHACAutilities::get_coeffs(example.Ufield,
256  example.Umodes,
257  List_of_modes(i, 0) + example.liftfield.size() + NmodesSUPproj);
258  Eigen::MatrixXd coeffT = ITHACAutilities::get_coeffs(example.Tfield, TLmodes,
259  List_of_modes(i, 0) + example.liftfieldT.size());
260  PtrList<volVectorField> rec_fieldU = ITHACAutilities::reconstruct_from_coeff(
261  example.Umodes, coeffU, List_of_modes(i, 0));
262  PtrList<volScalarField> rec_fieldT = ITHACAutilities::reconstruct_from_coeff(
263  TLmodes, coeffT, List_of_modes(i, 0) + example.liftfieldT.size());
264  Eigen::MatrixXd L2errorProjU = ITHACAutilities::error_listfields(example.Ufield,
265  rec_fieldU);
266  Eigen::MatrixXd L2errorProjT = ITHACAutilities::error_listfields(example.Tfield,
267  rec_fieldT);
268  L2errorProjMatrixU.col(i) = L2errorProjU;
269  L2errorProjMatrixT.col(i) = L2errorProjT;
270  }
271 
272  // Export the matrix containing the error
273  ITHACAstream::exportMatrix(L2errorProjMatrixU, "L2errorProjMatrixU", "eigen",
274  "./ITHACAoutput/l2error");
275  ITHACAstream::exportMatrix(L2errorProjMatrixT, "L2errorProjMatrixT", "eigen",
276  "./ITHACAoutput/l2error");
277  // Get reduced matrices
278  example.projectSUP("./Matrices", NmodesUproj, NmodesPproj, NmodesTproj,
279  NmodesSUPproj);
280  // Resize the modes for projection
281  example.Tmodes.resize(NmodesTproj);
282  example.Umodes.resize(NmodesUproj);
283  // Online part
284  reducedUnsteadyBB reduced(example);
285  // Set values of the online solve
286  reduced.nu = 0.00001;
287  reduced.Pr = 0.71;
288  reduced.tstart = 0.0;
289  reduced.finalTime = 10;
290  reduced.dt = 0.005;
291  // No parametrization of velocity on boundary
292  Eigen::MatrixXd vel_now_BC(0, 0);
293 
294  // Set the online temperature BC and solve reduced model
295  for (label k = 0; k < (par_on_BC.rows()); k++)
296  {
297  Eigen::MatrixXd temp_now_BC(2, 1);
298  temp_now_BC(0, 0) = par_on_BC(k, 0);
299  temp_now_BC(1, 0) = par_on_BC(k, 1);
300  reduced.solveOnline_sup(temp_now_BC, vel_now_BC, k, par_on_BC.rows());
301  reduced.reconstruct_sup("./ITHACAoutput/ReconstructionSUP", 2);
302  }
303 
304  // Performing full order simulation for second parameter set - temp_BC
305  tutorial10 HFonline2(argc, argv);
306  HFonline2.Pnumber = 1;
307  HFonline2.Tnumber = 1;
308  HFonline2.setParameters();
309  HFonline2.mu_range(0, 0) = 0.00001;
310  HFonline2.mu_range(0, 1) = 0.00001;
311  HFonline2.genEquiPar();
312  HFonline2.inletIndexT.resize(2, 1);
313  HFonline2.inletIndexT << 1, 2;
314  HFonline2.startTime = 0.0;
315  HFonline2.finalTime = 10;
316  HFonline2.timeStep = 0.005;
317  HFonline2.writeEvery = 0.01;
318  // Reconstruct the online solution
319  HFonline2.onlineSolveFull(par_on_BC, 1,
320  "./ITHACAoutput/HFonline2");
321  // Performing full order simulation for third parameter set - temp_BC
322  tutorial10 HFonline3(argc, argv);
323  HFonline3.Pnumber = 1;
324  HFonline3.Tnumber = 1;
325  HFonline3.setParameters();
326  HFonline3.mu_range(0, 0) = 0.00001;
327  HFonline3.mu_range(0, 1) = 0.00001;
328  HFonline3.genEquiPar();
329  HFonline3.inletIndexT.resize(2, 1);
330  HFonline3.inletIndexT << 1, 2;
331  HFonline3.startTime = 0.0;
332  HFonline3.finalTime = 10.0;
333  HFonline3.timeStep = 0.005;
334  HFonline3.writeEvery = 0.01;
335  // Reconstruct the online solution
336  HFonline3.onlineSolveFull(par_on_BC, 2,
337  "./ITHACAoutput/HFonline3");
338  // Reading in the high-fidelity solutions for the parameter set
339  // for which the offline solve has been performed
340  example.onlineSolveRead("./ITHACAoutput/Offline/");
341  // Reading in the high-fidelity solutions for the second parameter set
342  example.onlineSolveRead("./ITHACAoutput/HFonline2/");
343  // Reading in the high-fidelity solutions for the second parameter set
344  example.onlineSolveRead("./ITHACAoutput/HFonline3/");
345  // Calculate error between online- and corresponding full order solution
346  Eigen::MatrixXd L2errorMatrixU = ITHACAutilities::error_listfields(
347  example.Ufield_on, reduced.UREC);
348  Eigen::MatrixXd L2errorMatrixT = ITHACAutilities::error_listfields(
349  example.Tfield_on, reduced.TREC);
350  //Export the matrix containing the error
351  ITHACAstream::exportMatrix(L2errorMatrixU, "L2errorMatrixU", "eigen",
352  "./ITHACAoutput/l2error");
353  ITHACAstream::exportMatrix(L2errorMatrixT, "L2errorMatrixT", "eigen",
354  "./ITHACAoutput/l2error");
355  exit(0);
356 }
