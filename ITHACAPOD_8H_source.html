266 std::tuple<List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > ITHACAPOD::DEIMmodes(PtrList<type_matrix>& MatrixList, int nmodesA, int nmodesB, word MatrixName)
267 {
268  List<Eigen::SparseMatrix<double> > ModesA(nmodesA);
269  List<Eigen::VectorXd> ModesB(nmodesB);
270  if (!ITHACAutilities::check_folder("./ITHACAoutput/DEIM/" + MatrixName))
271  {
272  M_Assert(nmodesA <= MatrixList.size() - 2 && nmodesB <= MatrixList.size() - 2, "The number of requested modes cannot be bigger than the number of Snapshots - 2");
273  std::tuple<List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > snapshots = Foam2Eigen::LFvMatrix2LSM(MatrixList);
274  Eigen::MatrixXd corMatrixA = ITHACAPOD::corMatrix(std::get<0>(snapshots));
275  Eigen::MatrixXd corMatrixB = ITHACAPOD::corMatrix(std::get<1>(snapshots));
276  Info << "####### Performing the POD decomposition for the Matrix List #######" << endl;
277  Spectra::DenseSymMatProd<double> opA(corMatrixA);
278  Spectra::DenseSymMatProd<double> opB(corMatrixB);
279  int ncvA = MatrixList.size();
280  int ncvB = MatrixList.size();
281  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double> > esA(&opA, nmodesA, ncvA);
282  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double> > esB(&opB, nmodesB, ncvB);
283  esA.init();
284  esB.init();
285  esA.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
286  M_Assert(esA.info() == Spectra::SUCCESSFUL, "The Eigenvalue Decomposition did not succeed");
287  if (nmodesB != 1)
288  {
289  esB.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
290  M_Assert(esB.info() == Spectra::SUCCESSFUL, "The Eigenvalue Decomposition did not succeed");
291  }
292  Info << "####### End of the POD decomposition for the Matrix List #######" << endl;
293  Eigen::VectorXd eigenValueseigA;
294  Eigen::MatrixXd eigenVectorseigA;
295  Eigen::VectorXd eigenValueseigB;
296  Eigen::MatrixXd eigenVectorseigB;
297  eigenValueseigA = esA.eigenvalues().real();
298  eigenVectorseigA = esA.eigenvectors().real();
299  if (nmodesB != 1)
300  {
301  eigenValueseigB = esB.eigenvalues().real();
302  eigenVectorseigB = esB.eigenvectors().real();
303  }
304  else if (nmodesB == 1)
305  {
306  eigenValueseigB.resize(1);
307  eigenVectorseigB.resize(MatrixList.size(), nmodesB);
308  eigenValueseigB(0) = 1;
309  eigenVectorseigB = eigenVectorseigB * 0;
310  eigenVectorseigB(0, 0) = 1;
311  }
312  Eigen::SparseMatrix<double> tmp_A;
313  Eigen::VectorXd tmp_B;
314  for (label i = 0; i < nmodesA; i++)
315  {
316  tmp_A = eigenVectorseigA(0, i) * std::get<0>(snapshots)[0];
317  for (label k = 1; k < MatrixList.size(); k++)
318  {
319  tmp_A += eigenVectorseigA(k, i) * std::get<0>(snapshots)[k];
320  }
321  ModesA[i] = tmp_A;
322  }
323  for (label i = 0; i < nmodesB; i++)
324  {
325  tmp_B = eigenVectorseigB(0, i) * std::get<1>(snapshots)[0];
326  for (label k = 1; k < MatrixList.size(); k++)
327  {
328  tmp_B += eigenVectorseigB(k, i) * std::get<1>(snapshots)[k];
329  }
330  ModesB[i] = tmp_B;
331  }
332  eigenValueseigA = eigenValueseigA/eigenValueseigA.sum();
333  eigenValueseigB = eigenValueseigB/eigenValueseigB.sum();
334  Eigen::VectorXd cumEigenValuesA(eigenValueseigA);
335  Eigen::VectorXd cumEigenValuesB(eigenValueseigB);
336  for (int j = 1; j < cumEigenValuesA.size(); ++j)
337  {
338  cumEigenValuesA(j) += cumEigenValuesA(j - 1);
339  }
340  for (int j = 1; j < cumEigenValuesB.size(); ++j)
341  {
342  cumEigenValuesB(j) += cumEigenValuesB(j - 1);
343  }
344  for (int i = 0; i < ModesA.size(); i++)
345  {
346  ITHACAstream::SaveSparseMatrix(ModesA[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "A_" + MatrixName + name(i));
347  }
348  for (int i = 0; i < ModesB.size(); i++)
349  {
350  ITHACAstream::SaveDenseMatrix(ModesB[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "B_" + MatrixName + name(i));
351  }
352  ITHACAparameters para;
353  Eigen::saveMarketVector(eigenValueseigA, "./ITHACAoutput/DEIM/" + MatrixName + "/eigenValuesA",para.precision, para.outytpe);
354  Eigen::saveMarketVector(eigenValueseigB, "./ITHACAoutput/DEIM/" + MatrixName + "/eigenValuesB",para.precision, para.outytpe);
355  Eigen::saveMarketVector(cumEigenValuesA, "./ITHACAoutput/DEIM/" + MatrixName + "/cumEigenValuesA",para.precision, para.outytpe);
356  Eigen::saveMarketVector(cumEigenValuesB, "./ITHACAoutput/DEIM/" + MatrixName + "/cumEigenValuesB",para.precision, para.outytpe);
357  }
358  else
359  {
360  for (label i = 0; i < nmodesA; i++)
361  {
362  ITHACAstream::ReadSparseMatrix(ModesA[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "A_" + MatrixName + name(i));
363  }
364  for (label i = 0; i < nmodesB; i++)
365  {
366  ITHACAstream::ReadDenseMatrix(ModesB[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "B_" + MatrixName + name(i));
367  }
368 
369  }
370  std::tuple <List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > tupla;
371  tupla = std::make_tuple(ModesA, ModesB);
372  return tupla;
373 }

377 PtrList<GeometricField<T, fvPatchField, volMesh> > ITHACAPOD::DEIMmodes( PtrList<GeometricField<T, fvPatchField, volMesh> >& SnapShotsMatrix, int nmodes, word FunctionName)
378 {
379  if (nmodes == 0)
380  {
381  nmodes = SnapShotsMatrix.size() - 2;
382  }
383  M_Assert(nmodes <= SnapShotsMatrix.size() - 2, "The number of requested modes cannot be bigger than the number of Snapshots - 2");
384 
385  if (nmodes > SnapShotsMatrix.size() - 2)
386  {
387  std::cout << "The number of requested modes cannot be bigger than the number of Snapshots - 2" << std::endl;
388  exit(0);
389  }
390  PtrList<GeometricField<T, fvPatchField, volMesh> > Bases;
391  PtrList<GeometricField<T, fvPatchField, volMesh> > modes;
392  Bases.resize(nmodes);
393  modes.resize(nmodes);
394  Eigen::MatrixXd _corMatrix;
395  scalarField eigenValues(nmodes);
396  scalarField cumEigenValues(nmodes);
397  List<scalarField> eigenVector(nmodes);
398 
399  for (label i = 0; i < nmodes; i++)
400  {
401  eigenVector[i].setSize(SnapShotsMatrix.size());
402  }
403  _corMatrix = ITHACAPOD::corMatrix(SnapShotsMatrix);
404  Info << "####### Performing the POD decomposition for " << SnapShotsMatrix[0].name() << " #######" << endl;
405  int ncv = SnapShotsMatrix.size();
406  Spectra::DenseGenMatProd<double> op(_corMatrix);
407  Spectra::GenEigsSolver< double, Spectra::LARGEST_REAL, Spectra::DenseGenMatProd<double> > es(&op, nmodes, ncv);
408  es.init();
409  int nconv = es.compute(1000, 1e-10, Spectra::LARGEST_REAL);
410  Info << "####### End of the POD decomposition for " << SnapShotsMatrix[0].name() << " #######" << endl;
411  Eigen::VectorXd eigenValueseig;
412  Eigen::MatrixXd eigenVectoreig;
413  if (es.info() == Spectra::SUCCESSFUL)
414  {
415  eigenValueseig = es.eigenvalues().real();
416  eigenVectoreig = es.eigenvectors().real();
417  }
418  cumEigenValues[0] = eigenValueseig(0) / eigenValueseig.sum();
419  eigenValues[0] = eigenValueseig(0) / eigenValueseig.sum();
420  for (label i = 1; i < nmodes; i++)
421  {
422  cumEigenValues[i] = cumEigenValues[i - 1] + eigenValueseig(i) / eigenValueseig.sum();
423  eigenValues[i] = eigenValueseig(i) / eigenValueseig.sum();
424  }
425 
426  for (label i = 0; i < nmodes; i++)
427  {
428  for (label k = 0; k < SnapShotsMatrix.size(); k++)
429  {
430  eigenVector[i][k] = eigenVectoreig(k, i);
431  }
432  }
433  auto tmb_bu = SnapShotsMatrix[0];
434  tmb_bu.rename(SnapShotsMatrix[0].name());
435 
436  for (label i = 0; i < nmodes; i++)
437  {
438  tmb_bu = eigenVector[i][0] * SnapShotsMatrix[0];
439  for (label k = 1; k < SnapShotsMatrix.size(); k++)
440  {
441  tmb_bu += eigenVector[i][k] * SnapShotsMatrix[k];
442  }
443  Bases.set(i, tmb_bu);
444  Info << "creating the bases " << i << " for " << SnapShotsMatrix[0].name() << endl;
445  }
446  ITHACAPOD::normalizeBases(Bases);
447  for (label i = 0; i < Bases.size(); i++)
448  {
449  auto tmp(Bases[i]);
450  tmp.rename(SnapShotsMatrix[0].name());
451  modes.set(i, tmp);
452  }
453  Info << "####### Saving the POD bases for " << SnapShotsMatrix[0].name() << " #######" << endl;
454  ITHACAutilities::createSymLink("./ITHACAoutput/DEIM");
455 
456  for (int i = 0; i < modes.size(); i++)
457  {
458  ITHACAstream::exportSolution(modes[i], name(i + 1), "./ITHACAoutput/DEIM", modes[i].name());
459  }
460  ITHACAstream::exportList(eigenValues, "./ITHACAoutput/DEIM/", "eigenValues_" + SnapShotsMatrix[0].name());
461  ITHACAstream::exportList(cumEigenValues, "./ITHACAoutput/DEIM/", "cumEigenValues_" + SnapShotsMatrix[0].name());
462  return modes;
463 }
