252 std::tuple<List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > ITHACAPOD::DEIMmodes(PtrList<type_matrix>& MatrixList, int nmodesA, int nmodesB, word MatrixName)
253 {
254  List<Eigen::SparseMatrix<double> > ModesA(nmodesA);
255  List<Eigen::VectorXd> ModesB(nmodesB);
256  if (!ITHACAutilities::check_folder("./ITHACAoutput/DEIM/" + MatrixName))
257  {
258  M_Assert(nmodesA <= MatrixList.size() - 2 && nmodesB <= MatrixList.size() - 2, "The number of requested modes cannot be bigger than the number of Snapshots - 2");
259  std::tuple<List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > snapshots = Foam2Eigen::LFvMatrix2LSM(MatrixList);
260  Eigen::MatrixXd corMatrixA = ITHACAPOD::corMatrix(std::get<0>(snapshots));
261  Eigen::MatrixXd corMatrixB = ITHACAPOD::corMatrix(std::get<1>(snapshots));
262  Info << "####### Performing the POD decomposition for the Matrix List #######" << endl;
263  Spectra::DenseSymMatProd<double> opA(corMatrixA);
264  Spectra::DenseSymMatProd<double> opB(corMatrixB);
265  int ncvA = MatrixList.size();
266  int ncvB = MatrixList.size();
267  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double> > esA(&opA, nmodesA, ncvA);
268  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double> > esB(&opB, nmodesB, ncvB);
269  esA.init();
270  esB.init();
271  esA.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
272  M_Assert(esA.info() == Spectra::SUCCESSFUL, "The Eigenvalue Decomposition did not succeed");
273  if (nmodesB != 1)
274  {
275  esB.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
276  M_Assert(esB.info() == Spectra::SUCCESSFUL, "The Eigenvalue Decomposition did not succeed");
277  }
278  Info << "####### End of the POD decomposition for the Matrix List #######" << endl;
279  Eigen::VectorXd eigenValueseigA;
280  Eigen::MatrixXd eigenVectorseigA;
281  Eigen::VectorXd eigenValueseigB;
282  Eigen::MatrixXd eigenVectorseigB;
283  eigenValueseigA = esA.eigenvalues().real();
284  eigenVectorseigA = esA.eigenvectors().real();
285  if (nmodesB != 1)
286  {
287  eigenValueseigB = esB.eigenvalues().real();
288  eigenVectorseigB = esB.eigenvectors().real();
289  }
290  else if (nmodesB == 1)
291  {
292  eigenValueseigB.resize(1);
293  eigenVectorseigB.resize(MatrixList.size(), nmodesB);
294  eigenValueseigB(0) = 1;
295  eigenVectorseigB = eigenVectorseigB * 0;
296  eigenVectorseigB(0, 0) = 1;
297  }
298  Eigen::SparseMatrix<double> tmp_A;
299  Eigen::VectorXd tmp_B;
300  for (label i = 0; i < nmodesA; i++)
301  {
302  tmp_A = eigenVectorseigA(0, i) * std::get<0>(snapshots)[0];
303  for (label k = 1; k < MatrixList.size(); k++)
304  {
305  tmp_A += eigenVectorseigA(k, i) * std::get<0>(snapshots)[k];
306  }
307  ModesA[i] = tmp_A;
308  }
309  for (label i = 0; i < nmodesB; i++)
310  {
311  tmp_B = eigenVectorseigB(0, i) * std::get<1>(snapshots)[0];
312  for (label k = 1; k < MatrixList.size(); k++)
313  {
314  tmp_B += eigenVectorseigB(k, i) * std::get<1>(snapshots)[k];
315  }
316  ModesB[i] = tmp_B;
317  }
318  eigenValueseigA = eigenValueseigA/eigenValueseigA.sum();
319  eigenValueseigB = eigenValueseigB/eigenValueseigB.sum();
320  Eigen::VectorXd cumEigenValuesA(eigenValueseigA);
321  Eigen::VectorXd cumEigenValuesB(eigenValueseigB);
322  for (int j = 1; j < cumEigenValuesA.size(); ++j)
323  {
324  cumEigenValuesA(j) += cumEigenValuesA(j - 1);
325  }
326  for (int j = 1; j < cumEigenValuesB.size(); ++j)
327  {
328  cumEigenValuesB(j) += cumEigenValuesB(j - 1);
329  }
330  for (int i = 0; i < ModesA.size(); i++)
331  {
332  ITHACAstream::SaveSparseMatrix(ModesA[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "A_" + MatrixName + name(i));
333  }
334  for (int i = 0; i < ModesB.size(); i++)
335  {
336  ITHACAstream::SaveDenseMatrix(ModesB[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "B_" + MatrixName + name(i));
337  }
338  ITHACAparameters para;
339  Eigen::saveMarketVector(eigenValueseigA, "./ITHACAoutput/DEIM/" + MatrixName + "/eigenValuesA",para.precision, para.outytpe);
340  Eigen::saveMarketVector(eigenValueseigB, "./ITHACAoutput/DEIM/" + MatrixName + "/eigenValuesB",para.precision, para.outytpe);
341  Eigen::saveMarketVector(cumEigenValuesA, "./ITHACAoutput/DEIM/" + MatrixName + "/cumEigenValuesA",para.precision, para.outytpe);
342  Eigen::saveMarketVector(cumEigenValuesB, "./ITHACAoutput/DEIM/" + MatrixName + "/cumEigenValuesB",para.precision, para.outytpe);
343  }
344  else
345  {
346  for (label i = 0; i < nmodesA; i++)
347  {
348  ITHACAstream::ReadSparseMatrix(ModesA[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "A_" + MatrixName + name(i));
349  }
350  for (label i = 0; i < nmodesB; i++)
351  {
352  ITHACAstream::ReadDenseMatrix(ModesB[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "B_" + MatrixName + name(i));
353  }
354 
355  }
356  std::tuple <List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > tupla;
357  tupla = std::make_tuple(ModesA, ModesB);
358  return tupla;
359 }

363 PtrList<GeometricField<T, fvPatchField, volMesh> > ITHACAPOD::DEIMmodes( PtrList<GeometricField<T, fvPatchField, volMesh> >& SnapShotsMatrix, int nmodes, word FunctionName)
364 {
365  if (nmodes == 0)
366  {
367  nmodes = SnapShotsMatrix.size() - 2;
368  }
369  M_Assert(nmodes <= SnapShotsMatrix.size() - 2, "The number of requested modes cannot be bigger than the number of Snapshots - 2");
370 
371  if (nmodes > SnapShotsMatrix.size() - 2)
372  {
373  std::cout << "The number of requested modes cannot be bigger than the number of Snapshots - 2" << std::endl;
374  exit(0);
375  }
376  PtrList<GeometricField<T, fvPatchField, volMesh> > Bases;
377  PtrList<GeometricField<T, fvPatchField, volMesh> > modes;
378  Bases.resize(nmodes);
379  modes.resize(nmodes);
380  Eigen::MatrixXd _corMatrix;
381  scalarField eigenValues(nmodes);
382  scalarField cumEigenValues(nmodes);
383  List<scalarField> eigenVector(nmodes);
384 
385  for (label i = 0; i < nmodes; i++)
386  {
387  eigenVector[i].setSize(SnapShotsMatrix.size());
388  }
389  _corMatrix = ITHACAPOD::corMatrix(SnapShotsMatrix);
390  Info << "####### Performing the POD decomposition for " << SnapShotsMatrix[0].name() << " #######" << endl;
391  int ncv = SnapShotsMatrix.size();
392  Spectra::DenseGenMatProd<double> op(_corMatrix);
393  Spectra::GenEigsSolver< double, Spectra::LARGEST_REAL, Spectra::DenseGenMatProd<double> > es(&op, nmodes, ncv);
394  es.init();
395  int nconv = es.compute(1000, 1e-10, Spectra::LARGEST_REAL);
396  Info << "####### End of the POD decomposition for " << SnapShotsMatrix[0].name() << " #######" << endl;
397  Eigen::VectorXd eigenValueseig;
398  Eigen::MatrixXd eigenVectoreig;
399  if (es.info() == Spectra::SUCCESSFUL)
400  {
401  eigenValueseig = es.eigenvalues().real();
402  eigenVectoreig = es.eigenvectors().real();
403  }
404  cumEigenValues[0] = eigenValueseig(0) / eigenValueseig.sum();
405  eigenValues[0] = eigenValueseig(0) / eigenValueseig.sum();
406  for (label i = 1; i < nmodes; i++)
407  {
408  cumEigenValues[i] = cumEigenValues[i - 1] + eigenValueseig(i) / eigenValueseig.sum();
409  eigenValues[i] = eigenValueseig(i) / eigenValueseig.sum();
410  }
411 
412  for (label i = 0; i < nmodes; i++)
413  {
414  for (label k = 0; k < SnapShotsMatrix.size(); k++)
415  {
416  eigenVector[i][k] = eigenVectoreig(k, i);
417  }
418  }
419  auto tmb_bu = SnapShotsMatrix[0];
420  tmb_bu.rename(SnapShotsMatrix[0].name());
421 
422  for (label i = 0; i < nmodes; i++)
423  {
424  tmb_bu = eigenVector[i][0] * SnapShotsMatrix[0];
425  for (label k = 1; k < SnapShotsMatrix.size(); k++)
426  {
427  tmb_bu += eigenVector[i][k] * SnapShotsMatrix[k];
428  }
429  Bases.set(i, tmb_bu);
430  Info << "creating the bases " << i << " for " << SnapShotsMatrix[0].name() << endl;
431  }
432  ITHACAPOD::normalizeBases(Bases);
433  for (label i = 0; i < Bases.size(); i++)
434  {
435  auto tmp(Bases[i]);
436  tmp.rename(SnapShotsMatrix[0].name());
437  modes.set(i, tmp);
438  }
439  Info << "####### Saving the POD bases for " << SnapShotsMatrix[0].name() << " #######" << endl;
440  ITHACAutilities::createSymLink("./ITHACAoutput/DEIM");
441 
442  for (int i = 0; i < modes.size(); i++)
443  {
444  ITHACAstream::exportSolution(modes[i], name(i + 1), "./ITHACAoutput/DEIM", modes[i].name());
445  }
446  ITHACAstream::exportList(eigenValues, "./ITHACAoutput/DEIM/", "eigenValues_" + SnapShotsMatrix[0].name());
447  ITHACAstream::exportList(cumEigenValues, "./ITHACAoutput/DEIM/", "cumEigenValues_" + SnapShotsMatrix[0].name());
448  return modes;
449 }
