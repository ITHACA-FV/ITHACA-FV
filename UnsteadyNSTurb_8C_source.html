109 void UnsteadyNSTurb::truthSolve(List<scalar> mu_now)
110 {
111  Time& runTime = _runTime();
112  surfaceScalarField& phi = _phi();
113  fvMesh& mesh = _mesh();
114 #include "initContinuityErrs.H"
115  fv::options& fvOptions = _fvOptions();
116  pimpleControl& pimple = _pimple();
117  volScalarField p = _p();
118  volVectorField U = _U();
119  volScalarField nut = _nut();
120  IOMRFZoneList& MRF = _MRF();
121  singlePhaseTransportModel& laminarTransport = _laminarTransport();
122  instantList Times = runTime.times();
123  runTime.setEndTime(finalTime);
124  // Perform a TruthSolve
125  runTime.setTime(Times[1], 1);
126  runTime.setDeltaT(timeStep);
127  nextWrite = startTime;
128  // Initialize Nsnapshots
129  label nsnapshots = 0;
130 
131  // Start the time loop
132  while (runTime.run())
133  {
134 #include "readTimeControls.H"
135 #include "CourantNo.H"
136 #include "setDeltaT.H"
137  runTime.setEndTime(finalTime + timeStep);
138  Info << "Time = " << runTime.timeName() << nl << endl;
139 
140  // --- Pressure-velocity PIMPLE corrector loop
141  while (pimple.loop())
142  {
143 #include "UEqn.H"
144 
145  // --- Pressure corrector loop
146  while (pimple.correct())
147  {
148 #include "pEqn.H"
149  }
150 
151  if (pimple.turbCorr())
152  {
153  laminarTransport.correct();
154  turbulence->correct();
155  }
156  }
157 
158  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
159  << " ClockTime = " << runTime.elapsedClockTime() << " s"
160  << nl << endl;
161 
162  if (checkWrite(runTime))
163  {
164  nsnapshots += 1;
165  volScalarField nut = turbulence->nut();
166  ITHACAstream::exportSolution(U, name(counter), "./ITHACAoutput/Offline/");
167  ITHACAstream::exportSolution(p, name(counter), "./ITHACAoutput/Offline/");
168  ITHACAstream::exportSolution(nut, name(counter), "./ITHACAoutput/Offline/");
169  std::ofstream of("./ITHACAoutput/Offline/" + name(counter) + "/" +
170  runTime.timeName());
171  Ufield.append(tmp<volVectorField>(U));
172  Pfield.append(tmp<volScalarField>(p));
173  nutFields.append(tmp<volScalarField>(nut));
174  counter++;
175  nextWrite += writeEvery;
176  writeMu(mu_now);
177  // --- Fill in the mu_samples with parameters (time, mu) to be used for the PODI sample points
178  mu_samples.conservativeResize(mu_samples.rows() + 1, mu_now.size() + 1);
179  mu_samples(mu_samples.rows() - 1, 0) = atof(runTime.timeName().c_str());
180 
181  for (label i = 0; i < mu_now.size(); i++)
182  {
183  mu_samples(mu_samples.rows() - 1, i + 1) = mu_now[i];
184  }
185  }
186 
187  runTime++;
188  }
189 
190  // Resize to Unitary if not initialized by user (i.e. non-parametric problem)
191  if (mu.cols() == 0)
192  {
193  mu.resize(1, 1);
194  }
195 
196  if (mu_samples.rows() == nsnapshots * mu.cols())
197  {
198  ITHACAstream::exportMatrix(mu_samples, "mu_samples", "eigen",
199  "./ITHACAoutput/Offline");
200  }
201 }
