116  // Method to compute the lifting function for temperature
117  void liftSolveT()
118  {
119  for (label k = 0; k < inletIndexT.rows(); k++)
120  {
121  Time& runTime = _runTime();
122  fvMesh& mesh = _mesh();
123  volScalarField& T = _T();
124  volVectorField& U = _U();
125  surfaceScalarField& phi = _phi();
126  phi = linearInterpolate(U) & mesh.Sf();
127  simpleControl simple(mesh);
128  // IOMRFZoneList& MRF = _MRF();
129  // singlePhaseTransportModel& laminarTransport = _laminarTransport();
130  // volScalarField& nut = _nut();
131  volScalarField& alphat = _alphat();
132  // dimensionedScalar& nu = _nu();
133  dimensionedScalar& Pr = _Pr();
134  dimensionedScalar& Prt = _Prt();
135  label BCind = inletIndexT(k, 0);
136  volScalarField Tlift("Tlift" + name(k), T);
137  instantList Times = runTime.times();
138  runTime.setTime(Times[1], 1);
139  Info << "Solving a lifting Problem" << endl;
140  scalar t1 = 1;
141  scalar t0 = 0;
142  alphat = turbulence->nut() / Prt;
143  alphat.correctBoundaryConditions();
144  volScalarField alphaEff("alphaEff", turbulence->nu() / Pr + alphat);
145 
146  for (label j = 0; j < T.boundaryField().size(); j++)
147  {
148  if (j == BCind)
149  {
150  assignBC(Tlift, j, t1);
151  assignIF(Tlift, t0);
152  }
153  else if (T.boundaryField()[BCind].type() == "fixedValue")
154  {
155  assignBC(Tlift, j, t0);
156  assignIF(Tlift, t0);
157  }
158  else
159  {
160  }
161  }
162 
163  while (simple.correctNonOrthogonal())
164  {
165  fvScalarMatrix TEqn
166  (
167  fvm::div(phi, Tlift)
168  - fvm::laplacian(alphaEff, Tlift)
169  );
170  TEqn.solve();
171  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
172  << " ClockTime = " << runTime.elapsedClockTime() << " s"
173  << nl << endl;
174  }
175 
176  Tlift.write();
177  liftfieldT.append(Tlift);
178  }
179  }

187 int main(int argc, char* argv[])
188 {
189  // Construct the tutorial object
190  tutorial11 example(argc, argv);
191  // the offline samples for the boundary conditions
192  word par_offline_BC("./par_offline_BC");
193  // the samples which will be used for setting the boundary condition in the online stage
194  word par_online_BC("./par_online_BC");
195  Eigen::MatrixXd par_off_BC = ITHACAstream::readMatrix(par_offline_BC);
196  Eigen::MatrixXd par_on_BC = ITHACAstream::readMatrix(par_online_BC);
197  // Read some parameters from file
198  ITHACAparameters* para = ITHACAparameters::getInstance(example._mesh(),
199  example._runTime());
200  int NmodesUproj = para->ITHACAdict->lookupOrDefault<int>("NmodesUproj", 5);
201  int NmodesPproj = para->ITHACAdict->lookupOrDefault<int>("NmodesPproj", 5);
202  int NmodesPrghproj = para->ITHACAdict->lookupOrDefault<int>("NmodesPrghproj",
203  5);
204  int NmodesTproj = para->ITHACAdict->lookupOrDefault<int>("NmodesTproj", 5);
205  int NmodesSUPproj = para->ITHACAdict->lookupOrDefault<int>("NmodesSUPproj", 5);
206  int NmodesOut = para->ITHACAdict->lookupOrDefault<int>("NmodesOut", 15);
207  // Set the number of parameters
208  example.Pnumber = 1;
209  // Set samples
210  example.Tnumber = 1;
211  // Set the parameters infos
212  example.setParameters();
213  // Set the parameter ranges
214  example.mu_range(0, 0) = 0.00001;
215  example.mu_range(0, 1) = 0.00001;
216  // Generate equispaced samples inside the parameter range
217  example.genEquiPar();
218  // Set the inlet Temperature boundaries where we have non homogeneous boundary conditions
219  example.inletIndexT.resize(3, 1);
220  example.inletIndexT << 1, 2, 3;
221  // Set the inlet Velocity boundaries where we have non homogeneous boundary conditions
222  example.inletIndex.resize(1, 2);
223  example.inletIndex << 3, 0;
224  // Time parameters
225  example.startTime = 0.0;
226  example.finalTime = 5.0;
227  example.timeStep = 0.002;
228  example.writeEvery = 0.01;
229  // Perform the Offline Solve;
230  example.offlineSolve(par_off_BC);
231  // Search the lift function for the velocity
232  example.liftSolve();
233  // Search the lift function for the temperature
234  example.liftSolveT();
235  // Create homogeneous basis functions for velocity
236  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
237  // Create homogeneous basis functions for temperature
238  example.computeLiftT(example.Tfield, example.liftfieldT, example.Tomfield);
239  // Perform a POD decomposition for velocity ,temperature and pressure fields
240  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example._U().name(),
241  example.podex, 0, 0,
242  NmodesOut);
243  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example._p().name(),
244  example.podex, 0, 0,
245  NmodesOut);
246  ITHACAPOD::getModes(example.Prghfield, example.Prghmodes,
247  example._p_rgh().name(), example.podex, 0, 0,
248  NmodesOut);
249  ITHACAPOD::getModes(example.Tomfield, example.Tmodes, example._T().name(),
250  example.podex, 0, 0,
251  NmodesOut);
252  // Solve the supremizer problem
253  example.solvesupremizer("modes");
254  // Create a list with number of modes for which the projection needs to be performed
255  Eigen::MatrixXd List_of_modes(NmodesOut - 5, 1);
256 
257  for (int i = 0; i < List_of_modes.rows(); i++)
258  {
259  List_of_modes(i, 0) = i + 1;
260  }
261 
262  // Export with number of modes for which the projection needs to be performed
263  ITHACAstream::exportMatrix(List_of_modes, "List_of_modes", "eigen",
264  "./ITHACAoutput/l2error");
265  // Create locally the temperature modes
266  PtrList<volScalarField> TLmodes;
267 
268  for (label k = 0; k < example.liftfieldT.size(); k++)
269  {
270  TLmodes.append(example.liftfieldT[k]);
271  }
272 
273  for (label k = 0; k < List_of_modes.size(); k++)
274  {
275  TLmodes.append(example.Tmodes[k]);
276  }
277 
278  // Create locally the velocity modes
279  PtrList<volVectorField> ULmodes;
280 
281  for (label k = 0; k < example.liftfield.size(); k++)
282  {
283  ULmodes.append(example.liftfield[k]);
284  }
285 
286  for (label k = 0; k < List_of_modes.size(); k++)
287  {
288  ULmodes.append(example.Umodes[k]);
289  }
290 
291  for (label k = 0; k < NmodesSUPproj; k++)
292  {
293  ULmodes.append(example.supmodes[k]);
294  }
295 
296  // Perform the projection for all number of modes in List_of_modes for temperature and velocity
297  Eigen::MatrixXd L2errorProjMatrixU(example.Ufield.size(), List_of_modes.rows());
298  Eigen::MatrixXd L2errorProjMatrixT(example.Tfield.size(), List_of_modes.rows());
299 
300  // Calculate the coefficients and L2 error and store the error in a matrix for each number of modes
301  for (int i = 0; i < List_of_modes.rows(); i++)
302  {
303  Eigen::MatrixXd coeffU = ITHACAutilities::getCoeffs(example.Ufield, ULmodes,
304  List_of_modes(i, 0) + example.liftfield.size() + NmodesSUPproj);
305  Eigen::MatrixXd coeffT = ITHACAutilities::getCoeffs(example.Tfield, TLmodes,
306  List_of_modes(i, 0) + example.liftfieldT.size());
307  PtrList<volVectorField> rec_fieldU = ITHACAutilities::reconstructFromCoeff(
308  ULmodes, coeffU, List_of_modes(i,
309  0) + example.liftfield.size() + NmodesSUPproj);
310  PtrList<volScalarField> rec_fieldT = ITHACAutilities::reconstructFromCoeff(
311  TLmodes, coeffT, List_of_modes(i, 0) + example.liftfieldT.size());
312  Eigen::MatrixXd L2errorProjU = ITHACAutilities::errorL2Rel(example.Ufield,
313  rec_fieldU);
314  Eigen::MatrixXd L2errorProjT = ITHACAutilities::errorL2Rel(example.Tfield,
315  rec_fieldT);
316  L2errorProjMatrixU.col(i) = L2errorProjU;
317  L2errorProjMatrixT.col(i) = L2errorProjT;
318  }
319 
320  // Export the matrix containing the error
321  ITHACAstream::exportMatrix(L2errorProjMatrixU, "L2errorProjMatrixU", "eigen",
322  "./ITHACAoutput/l2error");
323  ITHACAstream::exportMatrix(L2errorProjMatrixT, "L2errorProjMatrixT", "eigen",
324  "./ITHACAoutput/l2error");
325  // Get reduced matrices
326  example.projectSUP("./Matrices", NmodesUproj, NmodesPrghproj, NmodesTproj,
327  NmodesSUPproj);
328  // Resize the modes for projection
329  example.Tmodes.resize(NmodesTproj);
330  example.Umodes.resize(NmodesUproj);
331  example.Pmodes.resize(NmodesPproj);
332  example.Prghmodes.resize(NmodesPrghproj);
333  // Online part
334  ReducedUnsteadyBB reduced(example);
335  // Set values of the online solve
336  reduced.nu = 0.00001;
337  reduced.Pr = 0.71;
338  reduced.tstart = 0.0;
339  reduced.finalTime = 5.0;
340  reduced.dt = 0.002;
341  // Set the online velocity
342  Eigen::MatrixXd vel_now_BC(1, 1);
343  vel_now_BC(0, 0) = 0.0157;
344 
345  // Set the online temperature BC and solve reduced model
346  for (label k = 0; k < par_on_BC.rows(); k++)
347  {
348  Eigen::MatrixXd temp_now_BC(3, 1);
349  temp_now_BC(0, 0) = par_on_BC(k, 0);
350  temp_now_BC(1, 0) = par_on_BC(k, 1);
351  temp_now_BC(2, 0) = par_on_BC(k, 2);
352  reduced.solveOnline_sup(temp_now_BC, vel_now_BC, k, par_on_BC.rows());
353  reduced.reconstruct_sup("./ITHACAoutput/ReconstructionSUP/", 5);
354  }
355 
356  // Performing full order simulation for second parameter set of temp_now_BC
357  tutorial11 HFonline2(argc, argv);
358  HFonline2.Pnumber = 1;
359  HFonline2.Tnumber = 1;
360  HFonline2.setParameters();
361  HFonline2.mu_range(0, 0) = 0.00001;
362  HFonline2.mu_range(0, 1) = 0.00001;
363  HFonline2.genEquiPar();
364  HFonline2.inletIndexT.resize(3, 1);
365  HFonline2.inletIndexT << 1, 2, 3;
366  HFonline2.inletIndex.resize(1, 2);
367  HFonline2.inletIndex << 3, 0;
368  HFonline2.startTime = 0.0;
369  HFonline2.finalTime = 5.0;
370  HFonline2.timeStep = 0.002;
371  HFonline2.writeEvery = 0.01;
372  // Reconstruct the online solution
373  HFonline2.onlineSolveFull(par_on_BC, 1,
374  "./ITHACAoutput/high_fidelity_online2");
375  // Reading in the high-fidelity solutions for the parameter set
376  // for which the offline solve has been performed
377  example.onlineSolveRead("./ITHACAoutput/Offline/");
378  // Reading in the high-fidelity solutions for the second parameter set
379  example.onlineSolveRead("./ITHACAoutput/high_fidelity_online2/");
380  // Calculate error between online- and corresponding full order solution
381  Eigen::MatrixXd L2errorMatrixU = ITHACAutilities::errorL2Rel(
382  example.Ufield_on, reduced.UREC);
383  Eigen::MatrixXd L2errorMatrixT = ITHACAutilities::errorL2Rel(
384  example.Tfield_on, reduced.TREC);
385  //Export the matrix containing the error
386  ITHACAstream::exportMatrix(L2errorMatrixU, "L2errorMatrixU", "eigen",
387  "./ITHACAoutput/l2error");
388  ITHACAstream::exportMatrix(L2errorMatrixT, "L2errorMatrixT", "eigen",
389  "./ITHACAoutput/l2error");
390  exit(0);
391 }
