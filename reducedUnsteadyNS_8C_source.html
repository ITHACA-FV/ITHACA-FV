46 reducedUnsteadyNS::reducedUnsteadyNS(unsteadyNS& problem, word tipo)
47 {
48  B_matrix = problem.B_matrix;
49  C_matrix = problem.C_matrix;
50  M_matrix = problem.M_matrix;
51  K_matrix = problem.K_matrix;
52  N_BC = problem.inletIndex.rows();
53 
54  Nphi_u = B_matrix.rows();
55  Nphi_p = K_matrix.cols();
56 
57  // Pressure Poisson equation approach
58  if (tipo == "PPE")
59  {
60  D_matrix = problem.D_matrix;
61  G_matrix = problem.G_matrix;
62  BC1_matrix = problem.BC1_matrix;
63  BC2_matrix = problem.BC2_matrix;
64  BC3_matrix = problem.BC3_matrix;
65  newton_object_PPE = newton_unsteadyNS_PPE(Nphi_u + Nphi_p , Nphi_u + Nphi_p, problem);
66  newton_object_sup.nu = nu;
67  }
68 
69  // Supremizer approach
70  if (tipo == "SUP")
71  {
72  P_matrix = problem.P_matrix;
73  newton_object_sup = newton_unsteadyNS_sup(Nphi_u + Nphi_p , Nphi_u + Nphi_p, problem);
74  newton_object_sup.nu = nu;
75  }
76 
77  // Create locally the velocity modes
78  for (label k = 0; k < problem.liftfield.size(); k++)
79  {
80  Umodes.append(problem.liftfield[k]);
81  }
82  for (label k = 0; k < problem.NUmodes; k++)
83  {
84  Umodes.append(problem.Umodes[k]);
85  }
86  for (label k = 0; k < problem.NSUPmodes; k++)
87  {
88  Umodes.append(problem.supmodes[k]);
89  }
90 
91  // Create locally the pressure modes
92  for (label k = 0; k < problem.NPmodes; k++)
93  {
94  Pmodes.append(problem.Pmodes[k]);
95  }
96 
97  // Store locally the snapshots for projections
98  for (label k = 0; k < problem.Ufield.size(); k++)
99  {
100  Usnapshots.append(problem.Ufield[k]);
101  Psnapshots.append(problem.Pfield[k]);
102  }
103 }

108 // Operator to evaluate the residual for the supremizer approach
109 int newton_unsteadyNS_sup::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
110 {
111  Eigen::VectorXd a_dot(Nphi_u);
112  Eigen::VectorXd a_tmp(Nphi_u);
113  Eigen::VectorXd b_tmp(Nphi_p);
114  a_tmp = x.head(Nphi_u);
115  b_tmp = x.tail(Nphi_p);
116  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
117 
118  // Convective term
119  Eigen::MatrixXd cc(1, 1);
120  // Mom Term
121  Eigen::VectorXd M1 = B_matrix * a_tmp * nu;
122  // Gradient of pressure
123  Eigen::VectorXd M2 = K_matrix * b_tmp;
124  // Mass Term
125  Eigen::VectorXd M5 = M_matrix * a_dot;
126  // Pressure Term
127  Eigen::VectorXd M3 = P_matrix * a_tmp;
128 
129  for (label i = 0; i < Nphi_u; i++)
130  {
131  cc = a_tmp.transpose() * C_matrix[i] * a_tmp;
132  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M2(i);
133  }
134  for (label j = 0; j < Nphi_p; j++)
135  {
136  label k = j + Nphi_u;
137  fvec(k) = M3(j);
138  }
139  for (label j = 0; j < N_BC; j++)
140  {
141  fvec(j) = x(j) - BC(j);
142  }
143  return 0;
144 }

157 // Operator to evaluate the residual for the supremizer approach
158 int newton_unsteadyNS_PPE::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
159 {
160  Eigen::VectorXd a_dot(Nphi_u);
161  Eigen::VectorXd a_tmp(Nphi_u);
162  Eigen::VectorXd b_tmp(Nphi_p);
163  a_tmp = x.head(Nphi_u);
164  b_tmp = x.tail(Nphi_p);
165  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
166 
167  // Convective terms
168  Eigen::MatrixXd cc(1, 1);
169  Eigen::MatrixXd gg(1, 1);
170  Eigen::MatrixXd bb(1, 1);
171  // Mom Term
172  Eigen::VectorXd M1 = B_matrix * a_tmp * nu;
173  // Gradient of pressure
174  Eigen::VectorXd M2 = K_matrix * b_tmp;
175  // Mass Term
176  Eigen::VectorXd M5 = M_matrix * a_dot;
177  // Pressure Term
178  Eigen::VectorXd M3 = D_matrix * b_tmp;
179 
180  // BC PPE
181  Eigen::VectorXd M6 = BC1_matrix * a_tmp * nu;
182 
183  // BC PPE
184  Eigen::VectorXd M7 = BC3_matrix * a_tmp * nu;
185 
186 
187 
188  for (label i = 0; i < Nphi_u; i++)
189  {
190  cc = a_tmp.transpose() * C_matrix[i] * a_tmp;
191  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M2(i);
192  }
193  for (label j = 0; j < Nphi_p; j++)
194  {
195  label k = j + Nphi_u;
196  gg = a_tmp.transpose() * G_matrix[j] * a_tmp;
197  bb = a_tmp.transpose() * BC2_matrix[j] * a_tmp;
198  //fvec(k) = M3(j, 0) - gg(0, 0) - M6(j, 0) + bb(0, 0);
199  fvec(k) = M3(j, 0) + gg(0, 0) - M7(j, 0);
200  }
201  for (label j = 0; j < N_BC; j++)
202  {
203  fvec(j) = x(j) - BC(j);
204  }
205  return 0;
206 }
