170 void unsteadyNST::liftSolveT()
171 {
172 
173  for (label k = 0; k < inletIndexT.rows(); k++)
174  {
175  Time& runTime = _runTime();
176  surfaceScalarField& phi = _phi();
177  fvMesh& mesh = _mesh();
178  volScalarField T = _T();
179  volScalarField p = _p();
180  volVectorField U = _U();
181  dimensionedScalar DT = _DT();
182  IOMRFZoneList& MRF = _MRF();
183  label BCind = inletIndexT(k,0);
184  volScalarField Tlift("Tlift" + name(k), T);
185  volVectorField Ulift("Ulift" + name(k), U);
186  pisoControl potentialFlow(mesh, "potentialFlow");
187  instantList Times = runTime.times();
188  runTime.setTime(Times[1], 1);
189 
190  simpleControl simple(mesh);
191 
192  dimensionedScalar nu = _nu();
193  singlePhaseTransportModel& laminarTransport = _laminarTransport();
194 
195 
196  runTime.setEndTime(finalTime);
197 
198 
199  scalar t1 = 1;
200  scalar t0 = 0;
201 
202 
203  for (label j = 0; j < T.boundaryField().size(); j++)
204  {
205 
206  if (j == BCind)
207  {
208  assignBC(Tlift, j, t1);
209  assignIF(Tlift,t0);
210 
211  if (T.boundaryField()[BCind].type() == "zeroGradient")
212  {
213  assignBC(Tlift, j, t0);
214  assignIF(Tlift,t1);
215  }
216 
217  }
218  else if (T.boundaryField()[BCind].type() == "fixedValue")
219  {
220  assignBC(Tlift, j, t0);
221  }
222 
223  else
224  {
225  }
226 
227  }
228 
229  while (simple.correctNonOrthogonal())
230  {
231  fvScalarMatrix TEqn
232  (
233 
234  fvm::ddt(Tlift)
235  - fvm::laplacian(DT, Tlift)
236  );
237 
238  TEqn.solve();
239 
240 
241 
242  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
243  << " ClockTime = " << runTime.elapsedClockTime() << " s"
244  << nl << endl;
245  }
246 
247  scalar totalTime = mesh.time().value();
248  scalar dt = mesh.time().deltaTValue();
249  forAll(Tlift, i)
250  {
251  Tlift[i] = (totalTime * Tlift[i] + dt * Tlift[i] ) / (totalTime + dt);
252  }
253  Tlift.write();
254  liftfieldT.append(Tlift);
255 
256  }
257 
258 }

437 void unsteadyNST::liftSolve()
438 {
439  for (label k = 0; k < inletIndex.rows(); k++)
440  {
441  Time& runTime = _runTime();
442  surfaceScalarField& phi = _phi();
443  fvMesh& mesh = _mesh();
444  volScalarField p = _p();
445  volVectorField U = _U();
446  dimensionedScalar nu = _nu();
447  IOMRFZoneList& MRF = _MRF();
448  label BCind = inletIndex(k, 0);
449  volVectorField Ulift("Ulift" + name(k), U);
450  instantList Times = runTime.times();
451  runTime.setTime(Times[1], 1);
452 
453  pisoControl potentialFlow(mesh, "potentialFlow");
454 
455  Info << "Solving a lifting Problem" << endl;
456 
457  Vector<double> v1(0, 0, 0);
458  v1[inletIndex(k, 1)] = 1;
459 
460  Vector<double> v0(0, 0, 0);
461 
462  for (label j = 0; j < U.boundaryField().size(); j++)
463  {
464  if (j == BCind)
465  {
466  assignBC(Ulift, j, v1);
467  }
468  else if (U.boundaryField()[BCind].type() == "fixedValue")
469  {
470  assignBC(Ulift, j, v0);
471  }
472  else
473  {
474 
475  }
476  assignIF(Ulift, v0);
477  phi = linearInterpolate(Ulift) & mesh.Sf();
478  }
479  Info << "Constructing velocity potential field Phi\n" << endl;
480  volScalarField Phi
481  (
482  IOobject
483  (
484  "Phi",
485  runTime.timeName(),
486  mesh,
487  IOobject::READ_IF_PRESENT,
488  IOobject::NO_WRITE
489  ),
490  mesh,
491  dimensionedScalar("Phi", dimLength * dimVelocity, 0),
492  p.boundaryField().types()
493  );
494 
495  label PhiRefCell = 0;
496  scalar PhiRefValue = 0;
497  setRefCell
498  (
499  Phi,
500  potentialFlow.dict(),
501  PhiRefCell,
502  PhiRefValue
503  );
504 
505  mesh.setFluxRequired(Phi.name());
506 
507  runTime.functionObjects().start();
508 
509  MRF.makeRelative(phi);
510  adjustPhi(phi, Ulift, p);
511 
512 
513  while (potentialFlow.correctNonOrthogonal())
514  {
515  fvScalarMatrix PhiEqn
516  (
517  fvm::laplacian(dimensionedScalar("1", dimless, 1), Phi)
518  //fvm::laplacian(nu, Phi)
519  ==
520  fvc::div(phi)
521  );
522 
523  PhiEqn.setReference(PhiRefCell, PhiRefValue);
524  PhiEqn.solve();
525 
526  if (potentialFlow.finalNonOrthogonalIter())
527  {
528  phi -= PhiEqn.flux();
529  }
530  }
531 
532  MRF.makeAbsolute(phi);
533 
534  Info << "Continuity error = "
535  << mag(fvc::div(phi))().weightedAverage(mesh.V()).value()
536  << endl;
537 
538  Ulift = fvc::reconstruct(phi);
539  Ulift.correctBoundaryConditions();
540 
541  Info << "Interpolated velocity error = "
542  << (sqrt(sum(sqr((fvc::interpolate(U) & mesh.Sf()) - phi)))
543  / sum(mesh.magSf())).value()
544  << endl;
545  Ulift.write();
546  liftfield.append(Ulift);
547  }
548 }
