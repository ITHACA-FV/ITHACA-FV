160 // Method to compute the lifting function for temperature
161 void unsteadyNST::liftSolveT()
162 {
163  for (label k = 0; k < inletIndexT.rows(); k++)
164  {
165  Time& runTime = _runTime();
166  surfaceScalarField& phi = _phi();
167  fvMesh& mesh = _mesh();
168  volScalarField T = _T();
169  volScalarField p = _p();
170  volVectorField U = _U();
171  dimensionedScalar DT = _DT();
172  IOMRFZoneList& MRF = _MRF();
173  label BCind = inletIndexT(k, 0);
174  volScalarField Tlift("Tlift" + name(k), T);
175  volVectorField Ulift("Ulift" + name(k), U);
176  pisoControl potentialFlow(mesh, "potentialFlow");
177  instantList Times = runTime.times();
178  runTime.setTime(Times[1], 1);
179  simpleControl simple(mesh);
180  dimensionedScalar nu = _nu();
181  singlePhaseTransportModel& laminarTransport = _laminarTransport();
182  runTime.setEndTime(finalTime);
183  scalar t1 = 1;
184  scalar t0 = 0;
185 
186  for (label j = 0; j < T.boundaryField().size(); j++)
187  {
188  if (j == BCind)
189  {
190  assignBC(Tlift, j, t1);
191  assignIF(Tlift, t0);
192 
193  if (T.boundaryField()[BCind].type() == "zeroGradient")
194  {
195  assignBC(Tlift, j, t0);
196  assignIF(Tlift, t1);
197  }
198  }
199  else if (T.boundaryField()[BCind].type() == "fixedValue")
200  {
201  assignBC(Tlift, j, t0);
202  }
203  else
204  {
205  }
206  }
207 
208  while (simple.correctNonOrthogonal())
209  {
210  fvScalarMatrix TEqn
211  (
212  fvm::ddt(Tlift)
213  - fvm::laplacian(DT, Tlift)
214  );
215  TEqn.solve();
216  Info << "ExecutionTime = " << runTime.elapsedCpuTime() << " s"
217  << " ClockTime = " << runTime.elapsedClockTime() << " s"
218  << nl << endl;
219  }
220 
221  scalar totalTime = mesh.time().value();
222  scalar dt = mesh.time().deltaTValue();
223  forAll(Tlift, i)
224  {
225  Tlift[i] = (totalTime * Tlift[i] + dt * Tlift[i] ) / (totalTime + dt);
226  }
227  Tlift.write();
228  liftfieldT.append(Tlift);
229  }
230 }

253 List< Eigen::MatrixXd > unsteadyNST::convective_term_temperature(label NUmodes,
254  label NTmodes, label NSUPmodes)
255 {
256  label Qsize = NUmodes + liftfield.size() + NSUPmodes;
257  label Qsizet = NTmodes + liftfieldT.size() ;

259  Q_matrix.setSize(Qsizet);
260 
261  for (label j = 0; j < Qsizet; j++)

263  Q_matrix[j].resize(Qsize, Qsizet);
264  }
265 
266  PtrList<volVectorField> Together(0);
267  PtrList<volScalarField> Togethert(0);
268 
269  if (liftfield.size() != 0)
270  {
271  for (label k = 0; k < liftfield.size(); k++)
272  {
273  Together.append(liftfield[k]);
274  }
275  }
276 
277  if (NUmodes != 0)
278  {
279  for (label k = 0; k < NUmodes; k++)
280  {
281  Together.append(Umodes[k]);
282  }
283  }
284 
285  if (NSUPmodes != 0)
286  {
287  for (label k = 0; k < NSUPmodes; k++)
288  {
289  Together.append(supmodes[k]);
290  }
291  }
292 
293  if (liftfieldT.size() != 0)
294  {
295  for (label k = 0; k < liftfieldT.size(); k++)
296  {
297  Togethert.append(liftfieldT[k]);
298  }
299  }
300 
301  if (NTmodes != 0)
302  {
303  for (label k = 0; k < NTmodes; k++)
304  {
305  Togethert.append(LTmodes[k]);
306  }
307  }
308 
309  {
310  for (label i = 0; i < Qsizet; i++)
311  {
312  for (label j = 0; j < Qsize; j++)
313  {
314  for (label k = 0; k < Qsizet; k++)
315  {
316  Q_matrix[i](j, k) = fvc::domainIntegrate(Togethert[i] * fvc::div(
317  fvc::interpolate(Together[j]) & Together[j].mesh().Sf(), Togethert[k])).value();
318  }
319  }
320  }
321  }
322 
323  // Export the matrix
324  ITHACAstream::exportMatrix(Q_matrix, "Q", "python", "./ITHACAoutput/Matrices/");
325  ITHACAstream::exportMatrix(Q_matrix, "Q", "matlab", "./ITHACAoutput/Matrices/");
326  ITHACAstream::exportMatrix(Q_matrix, "Q", "eigen", "./ITHACAoutput/Matrices/Q");
327  return Q_matrix;
328 }

414 // Calculate lifting function for velocity
415 void unsteadyNST::liftSolve()
416 {
417  for (label k = 0; k < inletIndex.rows(); k++)
418  {
419  Time& runTime = _runTime();
420  surfaceScalarField& phi = _phi();
421  fvMesh& mesh = _mesh();
422  volScalarField p = _p();
423  volVectorField U = _U();
424  dimensionedScalar nu = _nu();
425  IOMRFZoneList& MRF = _MRF();
426  label BCind = inletIndex(k, 0);
427  volVectorField Ulift("Ulift" + name(k), U);
428  instantList Times = runTime.times();
429  runTime.setTime(Times[1], 1);
430  pisoControl potentialFlow(mesh, "potentialFlow");
431  Info << "Solving a lifting Problem" << endl;
432  Vector<double> v1(0, 0, 0);
433  v1[inletIndex(k, 1)] = 1;
434  Vector<double> v0(0, 0, 0);
435 
436  for (label j = 0; j < U.boundaryField().size(); j++)
437  {
438  if (j == BCind)
439  {
440  assignBC(Ulift, j, v1);
441  }
442  else if (U.boundaryField()[BCind].type() == "fixedValue")
443  {
444  assignBC(Ulift, j, v0);
445  }
446  else
447  {
448  }
449 
450  assignIF(Ulift, v0);
451  phi = linearInterpolate(Ulift) & mesh.Sf();
452  }
453 
454  Info << "Constructing velocity potential field Phi\n" << endl;
455  volScalarField Phi
456  (
457  IOobject
458  (
459  "Phi",
460  runTime.timeName(),
461  mesh,
462  IOobject::READ_IF_PRESENT,
463  IOobject::NO_WRITE
464  ),
465  mesh,
466  dimensionedScalar("Phi", dimLength * dimVelocity, 0),
467  p.boundaryField().types()
468  );
469  label PhiRefCell = 0;
470  scalar PhiRefValue = 0;
471  setRefCell
472  (
473  Phi,
474  potentialFlow.dict(),
475  PhiRefCell,
476  PhiRefValue
477  );
478  mesh.setFluxRequired(Phi.name());
479  runTime.functionObjects().start();
480  MRF.makeRelative(phi);
481  adjustPhi(phi, Ulift, p);
482 
483  while (potentialFlow.correctNonOrthogonal())
484  {
485  fvScalarMatrix PhiEqn
486  (
487  fvm::laplacian(dimensionedScalar("1", dimless, 1), Phi)
488  //fvm::laplacian(nu, Phi)
489  ==
490  fvc::div(phi)
491  );
492  PhiEqn.setReference(PhiRefCell, PhiRefValue);
493  PhiEqn.solve();
494 
495  if (potentialFlow.finalNonOrthogonalIter())
496  {
497  phi -= PhiEqn.flux();
498  }
499  }
500 
501  MRF.makeAbsolute(phi);
502  Info << "Continuity error = "
503  << mag(fvc::div(phi))().weightedAverage(mesh.V()).value()
504  << endl;
505  Ulift = fvc::reconstruct(phi);
506  Ulift.correctBoundaryConditions();
507  Info << "Interpolated velocity error = "
508  << (sqrt(sum(sqr((fvc::interpolate(U) & mesh.Sf()) - phi)))
509  / sum(mesh.magSf())).value()
510  << endl;
511  Ulift.write();
512  liftfield.append(Ulift);
513  }
514 }
