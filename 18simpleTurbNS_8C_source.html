97 int main(int argc, char* argv[])
98 {
99  // Construct the tutorial object
100  tutorial18 example(argc, argv);
101  // Read some parameters from file
102  ITHACAparameters* para = ITHACAparameters::getInstance(example._mesh(),
103  example._runTime());
104  // Read the par file where the parameters are stored
105  std::ifstream exFileOff("./parsOff_mat.txt");
106 
107  if (exFileOff)
108  {
109  example.mu = ITHACAstream::readMatrix("./parsOff_mat.txt");
110  }
111  else
112  {
113  example.mu = Eigen::VectorXd::LinSpaced(50, 1.00e-04, 1.00e-05);
114  ITHACAstream::exportMatrix(example.mu, "parsOff", "eigen", "./");
115  }
116 
117  Eigen::MatrixXd parOn;
118  std::ifstream exFileOn("./parsOn_mat.txt");
119 
120  if (exFileOn)
121  {
122  parOn = ITHACAstream::readMatrix("./parsOn_mat.txt");
123  }
124  else
125  {
126  parOn = ITHACAutilities::rand(20, 1, 1.00e-04, 1.00e-05);
127  ITHACAstream::exportMatrix(parOn, "parsOn", "eigen", "./");
128  }
129 
130  // Set the inlet boundaries patch 0 directions x and y
131  example.inletIndex.resize(1, 2);
132  example.inletIndex(0, 0) = 0;
133  example.inletIndex(0, 1) = 0;
134  // Set the maximum iterations number for the offline phase
135  example.maxIter = para->ITHACAdict->lookupOrDefault<int>("maxIter", 2000);
136  // Perform the offline solve
137  example.offlineSolve();
138  ITHACAstream::read_fields(example.liftfield, example.U, "./lift/");
139  // Homogenize the snapshots
140  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
141  // Perform POD on velocity and pressure and store the first 10 modes
142  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example._U().name(),
143  example.podex, 0, 0,
144  example.NUmodesOut);
145  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example._p().name(),
146  example.podex, 0, 0,
147  example.NPmodesOut);
148 
149  if (ITHACAutilities::isTurbulent())
150  {
151  ITHACAPOD::getModes(example.nutFields, example.nutModes, "nut",
152  example.podex, 0, 0, example.NNutModesOut);
153  // Create the RBF for turbulence
154  example.getTurbRBF(example.NNutModes);
155  }
156 
157  // Create the reduced object
158  reducedSimpleSteadyNS reduced(example);
159  PtrList<volVectorField> U_rec_list;
160  PtrList<volScalarField> P_rec_list;
161  // Reads inlet volocities boundary conditions.
162  word vel_file(para->ITHACAdict->lookup("online_velocities"));
163  Eigen::MatrixXd vel = ITHACAstream::readMatrix(vel_file);
164  // Set the maximum iterations number for the online phase
165  reduced.maxIterOn = para->ITHACAdict->lookupOrDefault<int>("maxIterOn", 2000);
166 
167  //Perform the online solutions
168  for (label k = 0; k < parOn.size(); k++)
169  {
170  scalar mu_now = parOn(k, 0);
171  example.restart();
172  example.change_viscosity(mu_now);
173  reduced.setOnlineVelocity(vel);
174 
175  if (ITHACAutilities::isTurbulent())
176  {
177  reduced.solveOnline_Simple(mu_now, example.NUmodes, example.NPmodes,
178  example.NNutModes);
179  }
180  else
181  {
182  reduced.solveOnline_Simple(mu_now, example.NUmodes, example.NPmodes,
183  example.NNutModes);
184  }
185  }
186 
187  exit(0);
