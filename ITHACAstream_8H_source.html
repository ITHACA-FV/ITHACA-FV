373 template <typename T, int whatever, typename IND>
374 void ITHACAstream::SaveSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m, word folder, word MatrixName)
375 {
376  typedef Eigen::Triplet<int> Trip;
377  std::vector<Trip> res;
378  m.makeCompressed();
379  mkDir(folder);
380 
381  std::fstream writeFile;
382  writeFile.open(folder + MatrixName, std::ios::binary | std::ios::out);
383 
384  if (writeFile.is_open())
385  {
386  IND rows, cols, nnzs, outS, innS;
387  rows = m.rows() ;
388  cols = m.cols() ;
389  nnzs = m.nonZeros() ;
390  outS = m.outerSize();
391  innS = m.innerSize();
392  writeFile.write(reinterpret_cast<const char *> (&rows), sizeof(IND));
393  writeFile.write(reinterpret_cast<const char *> (&cols), sizeof(IND));
394  writeFile.write(reinterpret_cast<const char *> (&nnzs), sizeof(IND));
395  writeFile.write(reinterpret_cast<const char *> (&outS), sizeof(IND));
396  writeFile.write(reinterpret_cast<const char *> (&innS), sizeof(IND));
397  writeFile.write(reinterpret_cast<const char *>(m.valuePtr()), sizeof(T ) * m.nonZeros());
398  writeFile.write(reinterpret_cast<const char *>(m.outerIndexPtr()), sizeof(IND) * m.outerSize());
399  writeFile.write(reinterpret_cast<const char *>(m.innerIndexPtr()), sizeof(IND) * m.nonZeros());
400  writeFile.close();
401  }

404 template <typename T, int whatever, typename IND>
405 void ITHACAstream::ReadSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m, word folder, word MatrixName)
406 {
407  std::fstream readFile;
408  readFile.open(folder + MatrixName, std::ios::binary | std::ios::in);
409  if (!readFile.good())
410  {
411  std::cout << folder + MatrixName << " file does not exist, try to rerun the Offline Stage!" << std::endl;
412  exit(EXIT_FAILURE);
413  }
414  else if (readFile.is_open())
415  {
416  IND rows, cols, nnz, inSz, outSz;
417  readFile.read(reinterpret_cast<char *>(&rows ), sizeof(IND));
418  readFile.read(reinterpret_cast<char *>(&cols ), sizeof(IND));
419  readFile.read(reinterpret_cast<char *>(&nnz ), sizeof(IND));
420  readFile.read(reinterpret_cast<char *>(&inSz ), sizeof(IND));
421  readFile.read(reinterpret_cast<char *>(&outSz), sizeof(IND));
422  m.resize(rows, cols);
423  m.makeCompressed();
424  m.resizeNonZeros(nnz);
425  readFile.read(reinterpret_cast<char *>(m.valuePtr()) , sizeof(T ) * nnz );
426  readFile.read(reinterpret_cast<char *>(m.outerIndexPtr()), sizeof(IND) * outSz);
427  readFile.read(reinterpret_cast<char *>(m.innerIndexPtr()), sizeof(IND) * nnz );
428  m.finalize();
429  readFile.close();
430  }

435 void ITHACAstream::SaveDenseMatrix(MatrixType& Matrix, word folder, word MatrixName)
436 {
437  mkDir(folder);
438  std::ofstream out(folder + MatrixName, std::ios::out | std::ios::binary | std::ios::trunc);
439  typename MatrixType::Index rows = Matrix.rows(), cols = Matrix.cols();
440  out.write(reinterpret_cast<char *> (&rows), sizeof(typename MatrixType::Index));
441  out.write(reinterpret_cast<char *> (&cols), sizeof(typename MatrixType::Index));
442  out.write(reinterpret_cast<char *> (Matrix.data()), rows * cols * sizeof(typename MatrixType::Scalar) );
443  out.close();
444 }
