611 template <typename T, int whatever, typename IND>
612 void ITHACAstream::SaveSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
613  word folder, word MatrixName)
614 {
615  typedef Eigen::Triplet<int> Trip;
616  std::vector<Trip> res;
617  m.makeCompressed();
618  mkDir(folder);
619  std::fstream writeFile;
620  writeFile.open(folder + MatrixName, std::ios::binary | std::ios::out);
621 
622  if (writeFile.is_open())
623  {
624  IND rows, cols, nnzs, outS, innS;
625  rows = m.rows() ;
626  cols = m.cols() ;
627  nnzs = m.nonZeros() ;
628  outS = m.outerSize();
629  innS = m.innerSize();
630  writeFile.write(reinterpret_cast<const char*> (&rows), sizeof(IND));
631  writeFile.write(reinterpret_cast<const char*> (&cols), sizeof(IND));
632  writeFile.write(reinterpret_cast<const char*> (&nnzs), sizeof(IND));
633  writeFile.write(reinterpret_cast<const char*> (&outS), sizeof(IND));
634  writeFile.write(reinterpret_cast<const char*> (&innS), sizeof(IND));
635  writeFile.write(reinterpret_cast<const char*>(m.valuePtr()),
636  sizeof(T ) * m.nonZeros());
637  writeFile.write(reinterpret_cast<const char*>(m.outerIndexPtr()),
638  sizeof(IND) * m.outerSize());
639  writeFile.write(reinterpret_cast<const char*>(m.innerIndexPtr()),
640  sizeof(IND) * m.nonZeros());
641  writeFile.close();
642  }

645 template <typename T, int whatever, typename IND>
646 void ITHACAstream::ReadSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
647  word folder, word MatrixName)
648 {
649  std::fstream readFile;
650  readFile.open(folder + MatrixName, std::ios::binary | std::ios::in);
651 
652  if (!readFile.good())
653  {
654  std::cout << folder + MatrixName <<
655  " file does not exist, try to rerun the Offline Stage!" << std::endl;
656  exit(EXIT_FAILURE);
657  }
658  else if (readFile.is_open())
659  {
660  IND rows, cols, nnz, inSz, outSz;
661  readFile.read(reinterpret_cast<char*>(&rows ), sizeof(IND));
662  readFile.read(reinterpret_cast<char*>(&cols ), sizeof(IND));
663  readFile.read(reinterpret_cast<char*>(&nnz ), sizeof(IND));
664  readFile.read(reinterpret_cast<char*>(&inSz ), sizeof(IND));
665  readFile.read(reinterpret_cast<char*>(&outSz), sizeof(IND));
666  m.resize(rows, cols);
667  m.makeCompressed();
668  m.resizeNonZeros(nnz);
669  readFile.read(reinterpret_cast<char*>(m.valuePtr()), sizeof(T ) * nnz );
670  readFile.read(reinterpret_cast<char*>(m.outerIndexPtr()), sizeof(IND) * outSz);
671  readFile.read(reinterpret_cast<char*>(m.innerIndexPtr()), sizeof(IND) * nnz );
672  m.finalize();
673  readFile.close();
674  }
675 }
