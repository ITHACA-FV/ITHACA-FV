542 template <typename T, int whatever, typename IND>
543 void ITHACAstream::SaveSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
544  word folder, word MatrixName)
545 {
546  typedef Eigen::Triplet<int> Trip;
547  std::vector<Trip> res;
548  m.makeCompressed();
549  mkDir(folder);
550  std::fstream writeFile;
551  writeFile.open(folder + MatrixName, std::ios::binary | std::ios::out);
552 
553  if (writeFile.is_open())
554  {
555  IND rows, cols, nnzs, outS, innS;
556  rows = m.rows() ;
557  cols = m.cols() ;
558  nnzs = m.nonZeros() ;
559  outS = m.outerSize();
560  innS = m.innerSize();
561  writeFile.write(reinterpret_cast<const char*> (&rows), sizeof(IND));
562  writeFile.write(reinterpret_cast<const char*> (&cols), sizeof(IND));
563  writeFile.write(reinterpret_cast<const char*> (&nnzs), sizeof(IND));
564  writeFile.write(reinterpret_cast<const char*> (&outS), sizeof(IND));
565  writeFile.write(reinterpret_cast<const char*> (&innS), sizeof(IND));
566  writeFile.write(reinterpret_cast<const char*>(m.valuePtr()),
567  sizeof(T ) * m.nonZeros());
568  writeFile.write(reinterpret_cast<const char*>(m.outerIndexPtr()),
569  sizeof(IND) * m.outerSize());
570  writeFile.write(reinterpret_cast<const char*>(m.innerIndexPtr()),
571  sizeof(IND) * m.nonZeros());
572  writeFile.close();
573  }

576 template <typename T, int whatever, typename IND>
577 void ITHACAstream::ReadSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
578  word folder, word MatrixName)
579 {
580  std::fstream readFile;
581  readFile.open(folder + MatrixName, std::ios::binary | std::ios::in);
582 
583  if (!readFile.good())
584  {
585  std::cout << folder + MatrixName <<
586  " file does not exist, try to rerun the Offline Stage!" << std::endl;
587  exit(EXIT_FAILURE);
588  }
589  else if (readFile.is_open())
590  {
591  IND rows, cols, nnz, inSz, outSz;
592  readFile.read(reinterpret_cast<char*>(&rows ), sizeof(IND));
593  readFile.read(reinterpret_cast<char*>(&cols ), sizeof(IND));
594  readFile.read(reinterpret_cast<char*>(&nnz ), sizeof(IND));
595  readFile.read(reinterpret_cast<char*>(&inSz ), sizeof(IND));
596  readFile.read(reinterpret_cast<char*>(&outSz), sizeof(IND));
597  m.resize(rows, cols);
598  m.makeCompressed();
599  m.resizeNonZeros(nnz);
600  readFile.read(reinterpret_cast<char*>(m.valuePtr()), sizeof(T ) * nnz );
601  readFile.read(reinterpret_cast<char*>(m.outerIndexPtr()), sizeof(IND) * outSz);
602  readFile.read(reinterpret_cast<char*>(m.innerIndexPtr()), sizeof(IND) * nnz );
603  m.finalize();
604  readFile.close();
605  }
606 }
