81 int main(int argc, char *argv[])
82 {
83  // Construct the tutorial object
84  tutorial03 example(argc, argv);
85 
86  // Read some parameters from file
87  ITHACAparameters para;
88  int NmodesUout = para.ITHACAdict->lookupOrDefault<int>("NmodesUout", 15);
89  int NmodesPout = para.ITHACAdict->lookupOrDefault<int>("NmodesPout", 15);
90  int NmodesSUPout = para.ITHACAdict->lookupOrDefault<int>("NmodesSUPout", 15);
91  int NmodesUproj = para.ITHACAdict->lookupOrDefault<int>("NmodesUproj", 10);
92  int NmodesPproj = para.ITHACAdict->lookupOrDefault<int>("NmodesPproj", 10);
93  int NmodesSUPproj = para.ITHACAdict->lookupOrDefault<int>("NmodesSUPproj", 10);
94 
95 
96 
97  // Read the par file where the parameters are stored
98  word filename("./par");
99  example.mu = ITHACAstream::readMatrix(filename);
100 
101  // Set the inlet boundaries patch 0 directions x and y
102  example.inletIndex.resize(1, 2);
103  example.inletIndex(0, 0) = 0;
104  example.inletIndex(0, 1) = 0;
105 
106  // Perform the offline solve
107  example.offlineSolve();
108 
109  // Solve the supremizer problem
110  example.solvesupremizer();
111 
112  ITHACAstream::read_fields(example.liftfield, example.U, "./lift/");
113 
114  // Homogenize the snapshots
115  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
116 
117  // Perform POD on velocity pressure and supremizers and store the first 10 modes
118  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example.podex, 0, 0, NmodesUout);
119  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example.podex, 0, 0, NmodesPout);
120  ITHACAPOD::getModes(example.supfield, example.supmodes, example.podex, example.supex, 1, NmodesSUPout);
121 
122  // Perform the Galerkin Projection
123  example.projectSUP("./Matrices", NmodesUproj, NmodesPproj, NmodesSUPproj);
124 
125  // Create the reduced object
126  reducedSteadyNS ridotto(example);
127 
128  // Set the inlet velocity
129  Eigen::MatrixXd vel_now(2, 1);
130  vel_now(0, 0) = 1;
131  vel_now(1, 0) = 0;
132 
133  // Perform an online solve for the new values of inlet velocities
134  for (label k = 0; k < 20; k++)
135  {
136 
137  // Set the reduced viscosity
138  ridotto.nu = example.mu(k, 0);
139  ridotto.solveOnline_sup(vel_now);
140  Eigen::MatrixXd tmp_sol(ridotto.y.rows() + 1, 1);
141  tmp_sol(0) = k + 1;
142  tmp_sol.col(0).tail(ridotto.y.rows()) = ridotto.y;
143  ridotto.online_solution.append(tmp_sol);
144  }
145  // Save the online solution
146  ITHACAstream::exportMatrix(ridotto.online_solution, "red_coeff", "python", "./ITHACAoutput/red_coeff");
147  ITHACAstream::exportMatrix(ridotto.online_solution, "red_coeff", "matlab", "./ITHACAoutput/red_coeff");
148  ITHACAstream::exportMatrix(ridotto.online_solution, "red_coeff", "eigen", "./ITHACAoutput/red_coeff");
149 
150  // Reconstruct and export the solution
151  ridotto.reconstruct_sup("./ITHACAoutput/Reconstruction/");
152  exit(0);
153 }
