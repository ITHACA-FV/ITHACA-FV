37 void ITHACAPOD::getModes(PtrList<volVectorField>& snapshotsU, PtrList<volVectorField>& modes, bool podex, bool supex, bool sup, int nmodes)
38 {
39  if (nmodes == 0)
40  {
41  nmodes = snapshotsU.size() - 2;
42  }
43  M_Assert(nmodes <= snapshotsU.size() - 2, "The number of requested modes cannot be bigger than the number of Snapshots - 2");
44  if (podex == 0)
45  {
46  ITHACAparameters para;
47  Eigen::MatrixXd SnapMatrix = Foam2Eigen::PtrList2Eigen(snapshotsU);
48  List<Eigen::MatrixXd> SnapMatrixBC = Foam2Eigen::PtrList2EigenBC(snapshotsU);
49  int NBC = snapshotsU[0].boundaryField().size();
50  Eigen::VectorXd V = Foam2Eigen::field2Eigen(snapshotsU[0].mesh().V());
51  Eigen::VectorXd V3d = (V.replicate(3, 1));
52  auto VM = V3d.asDiagonal();
53  Eigen::MatrixXd _corMatrix = SnapMatrix.transpose() * VM * SnapMatrix;
54  Eigen::VectorXd eigenValueseig;
55  Eigen::MatrixXd eigenVectoreig;
56  modes.resize(nmodes);
57  Info << "####### Performing the POD using EigenDecomposition " << snapshotsU[0].name() << " #######" << endl;
58  int ncv = snapshotsU.size();
59  Spectra::DenseSymMatProd<double> op(_corMatrix);
60  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> esEg;
61  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra::DenseSymMatProd<double> > es(&op, nmodes, ncv);
62  if (para.eigensolver == "spectra")
63  {
64  std::cout << "Using Spectra EigenSolver " << std::endl;
65  es.init();
66  es.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
67  M_Assert(es.info() == Spectra::SUCCESSFUL, "The Eigenvalue Decomposition did not succeed");
68  eigenVectoreig = es.eigenvectors().real();
69  eigenValueseig = es.eigenvalues().real();
70  }
71  else if (para.eigensolver == "eigen")
72  {
73  std::cout << "Using Eigen EigenSolver " << std::endl;
74  esEg.compute(_corMatrix);
75  M_Assert(esEg.info() == Eigen::Success, "The Eigenvalue Decomposition did not succeed");
76  eigenVectoreig = esEg.eigenvectors().real().rowwise().reverse().leftCols(nmodes);
77  eigenValueseig = esEg.eigenvalues().real().reverse().head(nmodes);
78  }
79  Info << "####### End of the POD for " << snapshotsU[0].name() << " #######" << endl;
80  Eigen::VectorXd eigenValueseigLam = eigenValueseig.real().array().cwiseInverse().sqrt() ;
81  Eigen::MatrixXd modesEig = (SnapMatrix * eigenVectoreig) * eigenValueseigLam.asDiagonal();
82  List<Eigen::MatrixXd> modesEigBC;
83  modesEigBC.resize(NBC);
84  for (int i = 0; i < NBC; i++)
85  {
86  modesEigBC[i] = (SnapMatrixBC[i] * eigenVectoreig) * eigenValueseigLam.asDiagonal();
87  }
88 
89  for (int i = 0; i < modes.size(); i++)
90  {
91  volVectorField tmp(snapshotsU[0].name(), snapshotsU[0] * 0);
92  Eigen::VectorXd vec = modesEig.col(i);
93  tmp = Foam2Eigen::Eigen2field(tmp, vec);
94  for (int k = 0; k < tmp.boundaryField().size(); k++)
95  {
96  ITHACAutilities::assignBC(tmp, k, modesEigBC[k].col(i));
97  }
98  // const fvPatchList & patches = tmp.mesh().boundary();
99  // // Adjusting boundary conditions
100  // forAll(patches, iPatch)
101  // {
102  // UList<vector> myList(tmp.boundaryField()[iPatch].patchInternalField());
103  // forAll(tmp.boundaryFieldRef()[iPatch], iCell)
104  // {
105  // tmp.boundaryFieldRef()[iPatch][iCell] = myList[iCell];
106  // }
107  // }
108  modes.set(i, tmp);
109  }
110  eigenValueseig = eigenValueseig / eigenValueseig.sum();
111  Eigen::VectorXd cumEigenValues(eigenValueseig);
112  for (int j = 1; j < cumEigenValues.size(); ++j)
113  {
114  cumEigenValues(j) += cumEigenValues(j - 1);
115  }
116  Info << "####### Saving the POD bases for " << snapshotsU[0].name() << " #######" << endl;
117 
118  ITHACAPOD::exportBases(modes, snapshotsU, sup);
119  Eigen::saveMarketVector(eigenValueseig, "./ITHACAoutput/POD/Eigenvalues_" + snapshotsU[0].name(), para.precision, para.outytpe);
120  Eigen::saveMarketVector(cumEigenValues, "./ITHACAoutput/POD/CumEigenvalues_" + snapshotsU[0].name(), para.precision, para.outytpe);
121  }
122  else
123  {
124  Info << "Reading the existing modes" << endl;
125  if (sup == 1)
126  {
127  ITHACAstream::read_fields(modes, snapshotsU[0], "./ITHACAoutput/supremizer/");
128  }
129  else
130  {
131  ITHACAstream::read_fields (modes, snapshotsU[0], "./ITHACAoutput/POD/");
132  }
133  }

390 Eigen::MatrixXd ITHACAPOD::corMatrix(PtrList<volScalarField>& snapshots)
391 {
392  Info << "########## Filling the correlation matrix for " << snapshots[0].name() << "##########" << endl;
393  Eigen::MatrixXd matrix( snapshots.size(), snapshots.size());
394  for (label i = 0; i < snapshots.size(); i++)
395  {
396  Info << "Filling row " << i << " of the " << snapshots[0].name() << " correlation matrix" << endl;
397  for (label j = 0; j <= i; j++)
398  {
399  matrix(i, j) = fvc::domainIntegrate(snapshots[i] * snapshots[j]).value();
400  }
401  }
402  for (label i = 1; i < snapshots.size(); i++)
403  {
404  for (label j = 0; j < i; j++)
405  {
406  matrix(j, i) = matrix(i, j);
407  }
408  }
409  return matrix;

602 std::tuple<List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > ITHACAPOD::DEIMmodes(List<Eigen::SparseMatrix<double> >& A, List<Eigen::VectorXd>& b, int nmodesA, int nmodesB, word MatrixName)
603 {
604  List<Eigen::SparseMatrix<double> > ModesA(nmodesA);
605  List<Eigen::VectorXd> ModesB(nmodesB);
606 
607  if (!ITHACAutilities::check_folder("./ITHACAoutput/DEIM/" + MatrixName))
608  {
609  if (nmodesA > A.size() - 2 || nmodesB > A.size() - 2 )
610  {
611  std::cout << "The number of requested modes cannot be bigger than the number of Snapshots - 2" << std::endl;
612  exit(0);
613  }
614 
615  scalarField eigenValuesA(nmodesA); scalarField cumEigenValuesA(nmodesA);
616  scalarField eigenValuesB(nmodesB); scalarField cumEigenValuesB(nmodesB);
617  List<scalarField> eigenVectorA(nmodesA);
618  List<scalarField> eigenVectorB(nmodesB);
619 
620  for (label i = 0; i < nmodesA; i++)
621  {
622  eigenVectorA[i].setSize(A.size());
623  }
624  for (label i = 0; i < nmodesB; i++)
625  {
626  eigenVectorB[i].setSize(A.size());
627  }
628 
629  Eigen::MatrixXd corMatrixA = ITHACAPOD::corMatrix(A);
630  Eigen::MatrixXd corMatrixB = ITHACAPOD::corMatrix(b);
631  Info << "####### Performing the POD for the Matrix List #######" << endl;
632  Spectra::DenseSymMatProd<double> opA(corMatrixA);
633  Spectra::DenseSymMatProd<double> opB(corMatrixB);
634  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double> > esA(&opA, nmodesA, nmodesA + 10);
635  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double> > esB(&opB, nmodesB, nmodesB + 10);
636  esA.init();
637  esB.init();
638  esA.compute();
639  if (nmodesB != 1)
640  {
641  esB.compute();
642  }
643  Info << "####### End of the POD for the Matrix List #######" << endl;
644  Eigen::VectorXd eigenValueseigA;
645  Eigen::MatrixXd eigenVectorseigA;
646  Eigen::VectorXd eigenValueseigB;
647  Eigen::MatrixXd eigenVectorseigB;
648 
649  if (esA.info() == Spectra::SUCCESSFUL)
650  {
651  eigenValueseigA = esA.eigenvalues().real();
652  eigenVectorseigA = esA.eigenvectors().real();
653  if (esB.info() == Spectra::SUCCESSFUL && nmodesB != 1)
654  {
655  eigenValueseigB = esB.eigenvalues().real();
656  eigenVectorseigB = esB.eigenvectors().real();
657  }
658  else if (nmodesB == 1)
659  {
660  eigenValueseigB.resize(1);
661  eigenVectorseigB.resize(A.size(), nmodesB);
662  eigenValueseigB(0) = 1;
663  eigenVectorseigB = eigenVectorseigB * 0;
664  eigenVectorseigB(0, 0) = 1;
665  }
666  else
667  {
668  Info << "The Eigenvalue solver in ITHACAPOD.H did not converge, exiting the code" << endl;
669  exit(0);
670  }
671  }
672  else
673  {
674  Info << "The Eigenvalue solver in ITHACAPOD.H did not converge, exiting the code" << endl;
675  exit(0);
676 
677  }
678  for (label i = 0; i < nmodesA; i++)
679  {
680  eigenValuesA[i] = eigenValueseigA(i) / eigenValueseigA.sum();
681  }
682  for (label i = 0; i < nmodesB; i++)
683  {
684  eigenValuesB[i] = eigenValueseigB(i) / eigenValueseigB.sum();
685  }
686  for (label i = 0; i < nmodesA; i++)
687  {
688  for (label k = 0; k < A.size(); k++)
689  {
690  eigenVectorA[i][k] = eigenVectorseigA(k, i);
691  }
692  }
693  for (label i = 0; i < nmodesB; i++)
694  {
695  for (label k = 0; k < A.size(); k++)
696  {
697  eigenVectorB[i][k] = eigenVectorseigB(k, i);
698  }
699  }
700  cumEigenValuesA[0] = eigenValuesA[0];
701  cumEigenValuesB[0] = eigenValuesB[0];
702  for (label i = 1; i < nmodesA; i++)
703  {
704  cumEigenValuesA[i] = cumEigenValuesA[i - 1] + eigenValuesA[i];
705  }
706  for (label i = 1; i < nmodesB; i++)
707  {
708  cumEigenValuesB[i] = cumEigenValuesB[i - 1] + eigenValuesB[i];
709  }
710  Eigen::SparseMatrix<double> tmp_A;
711  Eigen::VectorXd tmp_B;
712  for (label i = 0; i < nmodesA; i++)
713  {
714  tmp_A = eigenVectorA[i][0] * A[0];
715  for (label k = 1; k < A.size(); k++)
716  {
717  tmp_A += eigenVectorA[i][k] * A[k];
718 
719  }
720  ModesA[i] = tmp_A;
721  }
722  for (label i = 0; i < nmodesB; i++)
723  {
724  tmp_B = eigenVectorB[i][0] * b[0];
725  for (label k = 1; k < A.size(); k++)
726  {
727  tmp_B += eigenVectorB[i][k] * b[k];
728  }
729  ModesB[i] = tmp_B;
730  }
731  ITHACAstream::exportList(eigenValuesA, "./ITHACAoutput/DEIM/" + MatrixName + "/", "eigenValuesA_" + MatrixName);
732  ITHACAstream::exportList(cumEigenValuesA, "./ITHACAoutput/DEIM/" + MatrixName + "/", "cumEigenValuesA_" + MatrixName);
733  ITHACAstream::exportList(eigenValuesB, "./ITHACAoutput/DEIM/" + MatrixName + "/", "eigenValuesB_" + MatrixName);
734  ITHACAstream::exportList(cumEigenValuesB, "./ITHACAoutput/DEIM/" + MatrixName + "/", "cumEigenValuesB_" + MatrixName);
735  for (int i = 0; i < ModesA.size(); i++)
736  {
737  ITHACAstream::SaveSparseMatrix(ModesA[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "A_" + MatrixName + name(i));
738  }
739  for (int i = 0; i < ModesB.size(); i++)
740  {
741  ITHACAstream::SaveDenseMatrix(ModesB[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "B_" + MatrixName + name(i));
742  }
743  }
744  else
745  {
746  for (label i = 0; i < nmodesA; i++)
747  {
748  ITHACAstream::ReadSparseMatrix(ModesA[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "A_" + MatrixName + name(i));
749  }
750  for (label i = 0; i < nmodesB; i++)
751  {
752  ITHACAstream::ReadDenseMatrix(ModesB[i], "./ITHACAoutput/DEIM/" + MatrixName + "/" , "B_" + MatrixName + name(i));
753  }
754 
755  }
756  std::tuple <List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > tupla;
757  tupla = std::make_tuple(ModesA, ModesB);

761 void ITHACAPOD::GrammSchmidt(Eigen::MatrixXd & Matrix)
762 {
763  Eigen::MatrixXd Ortho = Matrix;
764  Ortho = Matrix;
765  for (int i = 0; i < Matrix.cols(); i++)
766  {
767  for (int k = 0; k < i; k++)
768  {
769  double num = Ortho.col(k).transpose() * Matrix.col(i);
770  double den = (Ortho.col(k).transpose() * Ortho.col(k));
771  double fact = num / den;
772  Ortho.col(i) -= fact * Ortho.col(k) ;
773  }
774  Ortho.col(i).normalize();
775  }
776  Matrix = Ortho;
