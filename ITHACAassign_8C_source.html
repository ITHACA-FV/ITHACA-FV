30 template<class TypeField>
31 PtrList<TypeField> averageSubtract(PtrList<TypeField>
32  fields, Eigen::MatrixXd ind, PtrList<TypeField>& ave)
33 {
34  PtrList<TypeField> aveSubtracted;
35  Eigen::VectorXd newInd;
36  newInd.resize(ind.size() + 1);
37  newInd.head(ind.size()) = ind;
38  newInd(ind.size()) = fields.size();
39 
40  for (label i = 0; i < ind.size(); i++)
41  {
42  TypeField aveTemp("nut", fields[0] * 0);
43 
44  for (label j = newInd(i); j < newInd(i + 1); j++)
45  {
46  aveTemp += fields[j];
47  }
48 
49  aveTemp /= newInd(i + 1) - newInd(i);
50  ave.append(aveTemp.clone());
51  }
52 
53  for (label i = 0; i < ind.size(); i++)
54  {
55  for (label j = newInd(i); j < newInd(i + 1); j++)
56  {
57  TypeField newfield("nut", fields[0] * 0);
58  newfield = fields[j] - ave[i];
59  aveSubtracted.append(newfield.clone());
60  }
61  }
62 
63  return aveSubtracted;
64 }

173 // Assign a BC for a scalar field
174 void assignBC(GeometricField<scalar, fvPatchField, volMesh>& s, label BC_ind,
175  List<double> valueList)
176 {
177  word typeBC = s.boundaryField()[BC_ind].type();
178  label sizeBC = s.boundaryField()[BC_ind].size();
179  M_Assert(sizeBC == valueList.size(),
180  "The size of the given values list has to be equal to the dimension of the boundaryField");
181 
182  if (typeBC == "fixedGradient")
183  {
184  fixedGradientFvPatchScalarField& Tpatch =
185  refCast<fixedGradientFvPatchScalarField>(s.boundaryFieldRef()[BC_ind]);
186  scalarField& gradTpatch = Tpatch.gradient();
187  forAll(gradTpatch, faceI)
188  {
189  double value = valueList[faceI];
190  gradTpatch[faceI] = value;
191  }
192  }
193  else if (typeBC == "freestream")
194  {
195  for (label i = 0; i < sizeBC; i++)
196  {
197  double value = valueList[i];
198  s.boundaryFieldRef()[BC_ind][i] = value;
199  }
200 
201  freestreamFvPatchField<scalar>& Tpatch =
202  refCast<freestreamFvPatchField<scalar>>(s.boundaryFieldRef()[BC_ind]);
203  scalarField& gradTpatch = Tpatch.freestreamValue();
204  forAll(gradTpatch, faceI)
205  {
206  double value = valueList[faceI];
207  gradTpatch[faceI] = value;
208  }
209  }
210  else if (typeBC == "empty" || typeBC == "zeroGradient")
211  {}
212  else
213  {
214  try
215  {
216  if (typeBC != "fixedGradient" && typeBC != "freestream" && typeBC != "empty"
217  && typeBC != "zeroGradient" && typeBC != "fixedValue" && typeBC != "calculated"
218  && typeBC != "fixedFluxPressure" && typeBC != "processor"
219  && typeBC != "nutkWallFunction" && typeBC != "mixedEnergy")
220  {
221  word message = "Pay attention, your typeBC " + typeBC + " for " + s.name() +
222  " is not included into the developed ones. Your BC will be treated as a classical fixedValue.";
223  throw (message);
224  }
225  }
226  catch (const word message)
227  {
228  std::cerr << "WARNING: " << message << endl;
229  }
230 
231  for (label i = 0; i < sizeBC; i++)
232  {
233  double value = valueList[i];
234  s.boundaryFieldRef()[BC_ind][i] = value;
235  }
236  }
