136 void cnpy::parse_npy_header(FILE* fp, unsigned int& word_size,
137  unsigned int*& shape, unsigned int& ndims, bool& fortran_order)
138 {
139  char buffer[256];
140  size_t res = fread(buffer, sizeof(char), 11, fp);
141 
142  if (res != 11)
143  {
144  throw std::runtime_error("parse_npy_header: failed fread");
145  }
146 
147  std::string header = fgets(buffer, 256, fp);
148  assert(header[header.size() - 1] == '\n');
149  int loc1, loc2;
150  //fortran order
151  loc1 = header.find("fortran_order") + 16;
152  fortran_order = (header.substr(loc1, 5) == "True" ? true : false);
153  //shape
154  loc1 = header.find("(");
155  loc2 = header.find(")");
156  std::string str_shape = header.substr(loc1 + 1, loc2 - loc1 - 1);
157 
158  if (str_shape[str_shape.size() - 1] == ',')
159  {
160  ndims = 1;
161  }
162  else
163  {
164  ndims = std::count(str_shape.begin(), str_shape.end(), ',') + 1;
165  }
166 
167  shape = new unsigned int[ndims];
168 
169  for (unsigned int i = 0; i < ndims; i++)
170  {
171  loc1 = str_shape.find(",");
172  shape[i] = atoi(str_shape.substr(0, loc1).c_str());
173  str_shape = str_shape.substr(loc1 + 1);
174  }
175 
176  //endian, word size, data type
177  //byte order code | stands for not applicable.
178  //not sure when this applies except for byte array
179  loc1 = header.find("descr") + 9;
180  bool littleEndian = (header[loc1] == '<' || header[loc1] == '|' ? true : false);
181  assert(littleEndian);
182  //char type = header[loc1+1];
183  //assert(type == map_type(T));
184  std::string str_ws = header.substr(loc1 + 2);
185  loc2 = str_ws.find("'");
186  word_size = atoi(str_ws.substr(0, loc2).c_str());
187 }

189 void cnpy::parse_zip_footer(FILE* fp, unsigned short& nrecs,
190  unsigned int& global_header_size, unsigned int& global_header_offset)
191 {
192  std::vector<char> footer(22);
193  fseek(fp, -22, SEEK_END);
194  size_t res = fread(&footer[0], sizeof(char), 22, fp);
195 
196  if (res != 22)
197  {
198  throw std::runtime_error("parse_zip_footer: failed fread");
199  }
200 
201  unsigned short disk_no, disk_start, nrecs_on_disk, comment_len;
202  disk_no = *(unsigned short*) &footer[4];
203  disk_start = *(unsigned short*) &footer[6];
204  nrecs_on_disk = *(unsigned short*) &footer[8];
205  nrecs = *(unsigned short*) &footer[10];
206  global_header_size = *(unsigned int*) &footer[12];
207  global_header_offset = *(unsigned int*) &footer[16];
208  comment_len = *(unsigned short*) &footer[20];
209  assert(disk_no == 0);
210  assert(disk_start == 0);
211  assert(nrecs_on_disk == nrecs);
212  assert(comment_len == 0);
213 }

215 cnpy::NpyArray load_the_npy_file(FILE* fp)
216 {
217  unsigned int* shape;
218  unsigned int ndims, word_size;
219  bool fortran_order;
220  cnpy::parse_npy_header(fp, word_size, shape, ndims, fortran_order);
221  unsigned long long size =
222  1; //long long so no overflow when multiplying by word_size
223 
224  for (unsigned int i = 0; i < ndims; i++)
225  {
226  size *= shape[i];
227  }
228 
229  cnpy::NpyArray arr;
230  arr.word_size = word_size;
231  arr.shape = std::vector<unsigned int>(shape, shape + ndims);
232  delete[] shape;
233  arr.data = new char[size * word_size];
234  arr.fortran_order = fortran_order;
235  size_t nread = fread(arr.data, word_size, size, fp);
236 
237  if (nread != size)
238  {
239  throw std::runtime_error("load_the_npy_file: failed fread");
240  }
241 
242  return arr;

245 cnpy::npz_t cnpy::npz_load(std::string fname)
246 {
247  FILE* fp = fopen(fname.c_str(), "rb");
248 
249  if (!fp)
250  {
251  printf("npz_load: Error! Unable to open file %s!\n", fname.c_str());
252  }
253 
254  assert(fp);
255  cnpy::npz_t arrays;
256 
257  while (1)
258  {
259  std::vector<char> local_header(30);
260  size_t headerres = fread(&local_header[0], sizeof(char), 30, fp);
261 
262  if (headerres != 30)
263  {
264  throw std::runtime_error("npz_load: failed fread");
265  }
266 
267  //if we've reached the global header, stop reading
268  if (local_header[2] != 0x03 || local_header[3] != 0x04)
269  {
270  break;
271  }
272 
273  //read in the variable name
274  unsigned short name_len = *(unsigned short*) &local_header[26];
275  std::string varname(name_len, ' ');
276  size_t vname_res = fread(&varname[0], sizeof(char), name_len, fp);
277 
278  if (vname_res != name_len)
279  {
280  throw std::runtime_error("npz_load: failed fread");
281  }
282 
283  //erase the lagging .npy
284  varname.erase(varname.end() - 4, varname.end());
285  //read in the extra field
286  unsigned short extra_field_len = *(unsigned short*) &local_header[28];
287 
288  if (extra_field_len > 0)
289  {
290  std::vector<char> buff(extra_field_len);
291  size_t efield_res = fread(&buff[0], sizeof(char), extra_field_len, fp);
292 
293  if (efield_res != extra_field_len)
294  {
295  throw std::runtime_error("npz_load: failed fread");
296  }
297  }
298 
299  arrays[varname] = load_the_npy_file(fp);
300  }
301 
302  fclose(fp);
303  return arrays;
304 }
