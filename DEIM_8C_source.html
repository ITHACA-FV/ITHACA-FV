34 template<typename T>
35 DEIM<T>::DEIM (PtrList<T>& s, label MaxModes, word FunctionName)
36  :
37  SnapShotsMatrix(s),
38  MaxModes(MaxModes),
39  FunctionName(FunctionName)
40 {
41  Eigen::MatrixXd A;
42  Eigen::VectorXd b;
43  Eigen::VectorXd c;
44  Eigen::VectorXd r;
45  Eigen::VectorXd rho(1);
46  modes = ITHACAPOD::DEIMmodes(SnapShotsMatrix, MaxModes, FunctionName);
47  MatrixModes = Foam2Eigen::PtrList2Eigen(modes);
48  label ind_max, c1;
49  double max = MatrixModes.cwiseAbs().col(0).maxCoeff(&ind_max, &c1);
50  rho(0) = max;
51  magicPoints.append(ind_max);
52  U = MatrixModes.col(0);
53  P.resize(MatrixModes.rows(), 1);
54  P.insert(ind_max, 0) = 1;
55 
56  for (label i = 1; i < MaxModes; i++)
57  {
58  A = P.transpose() * U;
59  b = P.transpose() * MatrixModes.col(i);
60  c = A.fullPivLu().solve(b);
61  r = MatrixModes.col(i) - U * c;
62  max = r.cwiseAbs().maxCoeff(&ind_max, &c1);
63  P.conservativeResize(MatrixModes.rows(), i + 1);
64  P.insert(ind_max, i) = 1;
65  U.conservativeResize(MatrixModes.rows(), i + 1);
66  U.col(i) = MatrixModes.col(i);
67  rho.conservativeResize(i + 1);
68  rho(i) = max;
69  magicPoints.append(ind_max);
70  }
71 
72  MatrixOnline = U * ((P.transpose() * U).fullPivLu().inverse());
73 }

77 DEIM<T>::DEIM (PtrList<T>& s, label MaxModesA, label MaxModesB, word MatrixName)
78  :
79  SnapShotsMatrix(s),

85  runSubMeshB(false)
86

88  Eigen::MatrixXd AA;
89  Eigen::VectorXd bA;
90  Eigen::MatrixXd cA;
91  Eigen::SparseMatrix<double> rA;
92  Eigen::VectorXd rhoA(1);
93  Matrix_Modes = ITHACAPOD::DEIMmodes(SnapShotsMatrix, MaxModesA, MaxModesB,
94  MatrixName);
95  Ncells = getNcells(std::get<1>(Matrix_Modes)[0].rows());
96  label ind_rowA, ind_colA, xyz_rowA, xyz_colA;
97  ind_rowA = ind_colA = xyz_rowA = xyz_colA = 0;
98  double maxA = EigenFunctions::max(std::get<0>(Matrix_Modes)[0], ind_rowA,
99  ind_colA);
100  label ind_rowAOF = ind_rowA;
101  label ind_colAOF = ind_colA;
102  check3DIndices(ind_rowAOF, ind_colAOF, xyz_rowA, xyz_colA);
103  Pair <label> indA(ind_rowAOF, ind_colAOF);
104  Pair <label> xyzA(xyz_rowA, xyz_colA);
105  xyz_A.append(xyzA);
106  rhoA(0) = maxA;
107  magicPointsA.append(indA);
108  UA.append(std::get<0>(Matrix_Modes)[0]);
109  Eigen::SparseMatrix<double> Pnow(std::get<0>(Matrix_Modes)[0].rows(),
110  std::get<0>(Matrix_Modes)[0].cols());
111  Pnow.insert(ind_rowA, ind_colA) = 1;
112  PA.append(Pnow);
113 
114  for (label i = 1; i < MaxModesA; i++)
115  {
116  AA = EigenFunctions::innerProduct(PA, UA);
117  bA = EigenFunctions::innerProduct(PA, std::get<0>(Matrix_Modes)[i]);
118  cA = AA.fullPivLu().solve(bA);
119  rA = std::get<0>(Matrix_Modes)[i] - EigenFunctions::MVproduct(UA, cA);
120  double maxA = EigenFunctions::max(rA, ind_rowA, ind_colA);
121  rhoA.conservativeResize(i + 1);
122  rhoA(i) = maxA;
123  label ind_rowAOF = ind_rowA;
124  label ind_colAOF = ind_colA;
125  check3DIndices(ind_rowAOF, ind_colAOF, xyz_rowA, xyz_colA);
126  Pair <label> indA(ind_rowAOF, ind_colAOF);
127  Pair <label> xyzA(xyz_rowA, xyz_colA);
128  xyz_A.append(xyzA);
129  magicPointsA.append(indA);
130  UA.append(std::get<0>(Matrix_Modes)[i]);
131  Eigen::SparseMatrix<double> Pnow(std::get<0>(Matrix_Modes)[0].rows(),
132  std::get<0>(Matrix_Modes)[0].cols());
133  Pnow.insert(ind_rowA, ind_colA) = 1;
134  PA.append(Pnow);
135  }
136 
137  Eigen::MatrixXd Aaux = EigenFunctions::innerProduct(PA,
138  UA).fullPivLu().inverse();
139  MatrixOnlineA = EigenFunctions::MMproduct(UA, Aaux);
140  Eigen::MatrixXd AB;
141  Eigen::VectorXd bB;
142  Eigen::VectorXd cB;
143  Eigen::VectorXd rB;
144  Eigen::VectorXd rhoB(1);
145  label ind_rowB, xyz_rowB, c1;
146  double maxB = std::get<1>(Matrix_Modes)[0].cwiseAbs().maxCoeff(&ind_rowB,
147  &c1);
148  label ind_rowBOF = ind_rowB;
149  check3DIndices(ind_rowBOF, xyz_rowB);
150  rhoB(0) = maxB;
151  xyz_B.append(xyz_rowB);
152  magicPointsB.append(ind_rowBOF);
153  UB = std::get<1>(Matrix_Modes)[0];
154  PB.resize(UB.rows(), 1);
155  PB.insert(ind_rowB, 0) = 1;
156 
157  for (label i = 1; i < MaxModesB; i++)
158  {
159  AB = PB.transpose() * UB;
160  bB = PB.transpose() * std::get<1>(Matrix_Modes)[i];
161  cB = AB.fullPivLu().solve(bB);
162  rB = std::get<1>(Matrix_Modes)[i] - UB * cB;
163  maxB = rB.cwiseAbs().maxCoeff(&ind_rowB, &c1);
164  ind_rowBOF = ind_rowB;
165  check3DIndices(ind_rowBOF, xyz_rowB);
166  xyz_B.append(xyz_rowB);
167  PB.conservativeResize((std::get<1>(Matrix_Modes)[i]).size(), i + 1);
168  PB.insert(ind_rowB, i) = 1;
169  UB.conservativeResize((std::get<1>(Matrix_Modes)[i]).size(), i + 1);
170  UB.col(i) = std::get<1>(Matrix_Modes)[i];
171  rhoB.conservativeResize(i + 1);
172  rhoB(i) = maxB;
173  magicPointsB.append(ind_rowBOF);
174  }
175 
176  if (MaxModesB == 1 && std::get<1>(Matrix_Modes)[0].norm() < 1e-8)
177  {
178  MatrixOnlineB = Eigen::MatrixXd::Zero(std::get<1>(Matrix_Modes)[0].rows(), 1);
179  }
180  else if (MaxModesB != 1)
181  {
182  MatrixOnlineB = UB * ((PB.transpose() * UB).fullPivLu().inverse());
183  }
184  else
185  {
186  Eigen::MatrixXd aux = PB.transpose() * UB;
187  MatrixOnlineB = UB * 1 / aux(0, 0);
188  }
189 }
