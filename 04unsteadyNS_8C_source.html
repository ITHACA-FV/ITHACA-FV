87  int NmodesSUPout = para.ITHACAdict->lookupOrDefault<int>("NmodesSUPout", 15);
88  int NmodesUproj = para.ITHACAdict->lookupOrDefault<int>("NmodesUproj", 10);

91 
92 
94  example.Pnumber = 1;
96  example.Tnumber = 5;
98  example.setParameters();
99  // Set the parameter ranges
100  example.mu_range(0, 0) = 0.005;
101  example.mu_range(0, 1) = 0.01;
102  // Generate equispaced samples inside the parameter range
103  example.genEquiPar();
104 
105  // Set the inlet boundaries where you have non homogeneous boundary conditions
106  example.inletIndex.resize(1, 2);
107  example.inletIndex(0, 0) = 0;
108  example.inletIndex(0, 1) = 0;
109 
110  // Time parameters
111  example.startTime = 180;
112  example.finalTime = 190;
113  example.timeStep = 0.01;
114  example.writeEvery = 0.05;
115 
116  // Perform The Offline Solve;
117  example.offlineSolve();
118 
119  // Solve the supremizer problem
120  example.solvesupremizer();
121 
122  // Search the lift function
123  example.liftSolve();
124 
125  // Create homogeneous basis functions for velocity
126  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
127 
128  // Perform a POD decomposition for velocity and pressure
129  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example.podex, 0, 0, NmodesUout);
130  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example.podex, 0, 0, NmodesPout);
131  ITHACAPOD::getModes(example.supfield, example.supmodes, example.podex, example.supex, 1, NmodesSUPout);
132 
133  example.projectSUP("./Matrices", NmodesUproj, NmodesPproj, NmodesSUPproj);
134 
135  reducedUnsteadyNS ridotto(example, "SUP");
136  //unsteadyNSreduced ridotto(example, "PPE");
137 
138  // Set values of the ridotto stuff

140  ridotto.tstart = 0;

142  ridotto.dt = 0.01;
143 
144  // Set the online velocity

146  vel_now(0, 0) = 1;
147  ridotto.solveOnline_sup(vel_now);
148  // Reconstruct the solution and export it
149  ridotto.reconstruct_sup(example, "./ITHACAoutput/ReconstructionSUP/", 5);
150  //ridotto.reconstruct_PPE(example,"./ITHACAoutput/Reconstruction/",4);
