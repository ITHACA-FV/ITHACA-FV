84 template<typename T> void npy_save(std::string fname, const T* data,
85  const unsigned int* shape, const unsigned int ndims, std::string mode = "w")
86 {
87  FILE* fp = NULL;
88 
89  if (mode == "a")
90  {
91  fp = fopen(fname.c_str(), "r+b");
92  }
93 
94  if (fp )
95  {
96  //file exists. we need to append to it. read the header, modify the array size
97  unsigned int word_size, tmp_dims;
98  unsigned int* tmp_shape = 0;
99  bool fortran_order;
100  parse_npy_header(fp, word_size, tmp_shape, tmp_dims, fortran_order);
101  assert(!fortran_order);
102 
103  if (word_size != sizeof(T))
104  {
105  std::cout << "libnpy error: " << fname << " has word size " << word_size <<
106  " but npy_save appending data sized " << sizeof(T) << "\n";
107  assert( word_size == sizeof(T) );
108  }
109 
110  if (tmp_dims != ndims)
111  {
112  std::cout <<
113  "libnpy error: npy_save attempting to append misdimensioned data to " << fname
114  << "\n";
115  assert(tmp_dims == ndims);
116  }
117 
118  for (unsigned int i = 1; i < ndims; i++)
119  {
120  if (shape[i] != tmp_shape[i])
121  {
122  std::cout << "libnpy error: npy_save attempting to append misshaped data to " <<
123  fname << "\n";
124  assert(shape[i] == tmp_shape[i]);
125  }
126  }
127 
128  tmp_shape[0] += shape[0];
129  fseek(fp, 0, SEEK_SET);
130  std::vector<char> header = create_npy_header(data, tmp_shape, ndims);
131  fwrite(&header[0], sizeof(char), header.size(), fp);
132  fseek(fp, 0, SEEK_END);
133  delete[] tmp_shape;
134  }
135  else
136  {
137  fp = fopen(fname.c_str(), "wb");
138  std::vector<char> header = create_npy_header(data, shape, ndims);
139  fwrite(&header[0], sizeof(char), header.size(), fp);
140  }
141 
142  unsigned int nels = 1;
143 
144  for (unsigned int i = 0; i < ndims; i++)
145  {
146  nels *= shape[i];
147  }
148 
149  fwrite(data, sizeof(T), nels, fp);
150  fclose(fp);
151 }

153 template<typename T> void npz_save(std::string zipname, std::string fname,
154  const T* data, const unsigned int* shape, const unsigned int ndims,
155  std::string mode = "w")
156 {
157  //first, append a .npy to the fname
158  fname += ".npy";
159  //now, on with the show
160  FILE* fp = NULL;
161  unsigned short nrecs = 0;
162  unsigned int global_header_offset = 0;
163  std::vector<char> global_header;
164 
165  if (mode == "a")
166  {
167  fp = fopen(zipname.c_str(), "r+b");
168  }
169 
170  if (fp != NULL)
171  {
172  //zip file exists. we need to add a new npy file to it.
173  //first read the footer. this gives us the offset and size of the global header
174  //then read and store the global header.
175  //below, we will write the the new data at the start of the global header then append the global header and footer below it
176  unsigned int global_header_size;
177  parse_zip_footer(fp, nrecs, global_header_size, global_header_offset);
178  fseek(fp, global_header_offset, SEEK_SET);
179  global_header.resize(global_header_size);
180  size_t res = fread(&global_header[0], sizeof(char), global_header_size, fp);
181 
182  if (res != global_header_size)
183  {
184  throw std::runtime_error("npz_save: header read error while adding to existing zip");
185  }
186 
187  fseek(fp, global_header_offset, SEEK_SET);
188  }
189  else
190  {
191  fp = fopen(zipname.c_str(), "wb");
192  }
193 
194  std::vector<char> npy_header = create_npy_header(data, shape, ndims);
195  unsigned long nels = 1;
196 
197  for (int m = 0; m < ndims; m++ )
198  {
199  nels *= shape[m];
200  }
201 
202  int nbytes = nels * sizeof(T) + npy_header.size();
203  //get the CRC of the data to be added
204  unsigned int crc = crc32(0L, (unsigned char*)&npy_header[0], npy_header.size());
205  crc = crc32(crc, (unsigned char*)data, nels * sizeof(T));
206  //build the local header
207  std::vector<char> local_header;
208  local_header += "PK"; //first part of sig
209  local_header += (unsigned short) 0x0403; //second part of sig
210  local_header += (unsigned short) 20; //min version to extract
211  local_header += (unsigned short) 0; //general purpose bit flag
212  local_header += (unsigned short) 0; //compression method
213  local_header += (unsigned short) 0; //file last mod time
214  local_header += (unsigned short) 0; //file last mod date
215  local_header += (unsigned int) crc; //crc
216  local_header += (unsigned int) nbytes; //compressed size
217  local_header += (unsigned int) nbytes; //uncompressed size
218  local_header += (unsigned short) fname.size(); //fname length
219  local_header += (unsigned short) 0; //extra field length
220  local_header += fname;
221  //build global header
222  global_header += "PK"; //first part of sig
223  global_header += (unsigned short) 0x0201; //second part of sig
224  global_header += (unsigned short) 20; //version made by
225  global_header.insert(global_header.end(), local_header.begin() + 4,
226  local_header.begin() + 30);
227  global_header += (unsigned short) 0; //file comment length
228  global_header += (unsigned short) 0; //disk number where file starts
229  global_header += (unsigned short) 0; //internal file attributes
230  global_header += (unsigned int) 0; //external file attributes
231  global_header += (unsigned int)
232  global_header_offset; //relative offset of local file header, since it begins where the global header used to begin
233  global_header += fname;
234  //build footer
235  std::vector<char> footer;
236  footer += "PK"; //first part of sig
237  footer += (unsigned short) 0x0605; //second part of sig
238  footer += (unsigned short) 0; //number of this disk
239  footer += (unsigned short) 0; //disk where footer starts
240  footer += (unsigned short) (nrecs + 1); //number of records on this disk
241  footer += (unsigned short) (nrecs + 1); //total number of records
242  footer += (unsigned int) global_header.size(); //nbytes of global headers
243  footer += (unsigned int) (global_header_offset + nbytes +
244  local_header.size()); //offset of start of global headers, since global header now starts after newly written array
245  footer += (unsigned short) 0; //zip file comment length
246  //write everything
247  fwrite(&local_header[0], sizeof(char), local_header.size(), fp);
248  fwrite(&npy_header[0], sizeof(char), npy_header.size(), fp);
249  fwrite(data, sizeof(T), nels, fp);
250  fwrite(&global_header[0], sizeof(char), global_header.size(), fp);
251  fwrite(&footer[0], sizeof(char), footer.size(), fp);
252  fclose(fp);
253 }

255 template<typename T> std::vector<char> create_npy_header(const T* data,
256  const unsigned int* shape, const unsigned int ndims)
257 {
258  std::vector<char> dict;
259  dict += "{'descr': '";
260  dict += BigEndianTest();
261  dict += map_type(typeid(T));
262  dict += tostring(sizeof(T));
263  dict += "', 'fortran_order': False, 'shape': (";
264  dict += tostring(shape[0]);
265 
266  for (unsigned int i = 1; i < ndims; i++)
267  {
268  dict += ", ";
269  dict += tostring(shape[i]);
270  }
271 
272  if (ndims == 1)
273  {
274  dict += ",";
275  }
276 
277  dict += "), }";
278  //pad with spaces so that preamble+dict is modulo 16 bytes. preamble is 10 bytes. dict needs to end with \n
279  int remainder = 16 - (10 + dict.size()) % 16;
280  dict.insert(dict.end(), remainder, ' ');
281  dict.back() = '\n';
282  std::vector<char> header;
283  header += (char) 0x93;
284  header += "NUMPY";
285  header += (char) 0x01; //major version of numpy format
286  header += (char) 0x00; //minor version of numpy format
287  header += (unsigned short) dict.size();
288  header.insert(header.end(), dict.begin(), dict.end());
289  return header;
290 }
