53 void reducedLaplacian::solveOnline(Eigen::MatrixXd mu)
54 {
55  if (mu.cols() != A_matrices.size())
56  {
57  Info << "wrong dimension of online parameters" << endl;
58  exit(0);
59  }
60  Eigen::MatrixXd A;
61  A.setZero(NTmodes, NTmodes);
62  for (int i = 0; i < A_matrices.size() ; i++)
63  {
64  A += A_matrices[i] * mu(0, i);
65  }
66  Eigen::MatrixXd x;
67  x = A.colPivHouseholderQr().solve(-source);
68  online_solution.conservativeResize(count_online_solve, NTmodes + 1);
69  online_solution(count_online_solve - 1, 0) = count_online_solve;
70  online_solution.row(count_online_solve - 1).tail(NTmodes) = x.transpose();
71  count_online_solve += 1;
72 }

74 void reducedLaplacian::reconstruct(laplacianProblem& problem, fileName folder, int printevery)
75 {
76  mkDir(folder);

78 
79  int counter = 0;

84  {
85  if (counter == nextwrite)
86  {
87  volScalarField T_rec("T_rec", Tmodes[0] * 0);
88  for (label j = 0; j < NTmodes; j++)
89  {
90  T_rec += Tmodes[j] * online_solution(i, j + 1);
91  }
92  problem.exportSolution(T_rec, name(online_solution(i, 0)), folder);
93 
94  nextwrite += printevery;
95  counter2 ++;
96 
97  }
98  counter++;
99  }
