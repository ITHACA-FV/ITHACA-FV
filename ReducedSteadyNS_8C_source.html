70 int newton_steadyNS::operator()(const Eigen::VectorXd& x,
71  Eigen::VectorXd& fvec) const
72 {
73  Eigen::VectorXd a_tmp(Nphi_u);
74  Eigen::VectorXd b_tmp(Nphi_p);
75  a_tmp = x.head(Nphi_u);
76  b_tmp = x.tail(Nphi_p);
77  // Convective term
78  Eigen::MatrixXd cc(1, 1);
79  // Mom Term
80  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
81  // Gradient of pressure
82  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
83  // Pressure Term
84  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
85  // Penalty term
86  Eigen::MatrixXd penaltyU = Eigen::MatrixXd::Zero(Nphi_u, N_BC);
87 
88  // Term for penalty method
89  if (problem->bcMethod == "penalty")
90  {
91  for (int l = 0; l < N_BC; l++)
92  {
93  penaltyU.col(l) = BC(l) * problem->bcVelVec[l] - problem->bcVelMat[l] *
94  a_tmp;
95  }
96  }
97 
98  for (int i = 0; i < Nphi_u; i++)
99  {
100  cc = a_tmp.transpose() * Eigen::SliceFromTensor(problem->C_tensor, 0,
101  i) * a_tmp;
102  fvec(i) = M1(i) - cc(0, 0) - M2(i);
103 
104  if (problem->bcMethod == "penalty")
105  {
106  fvec(i) += (penaltyU.row(i) * tauU)(0, 0);
107  }
108  }
109 
110  for (int j = 0; j < Nphi_p; j++)
111  {
112  int k = j + Nphi_u;
113  fvec(k) = M3(j);
114  }
115 
116  if (problem->bcMethod == "lift")
117  {
118  for (int j = 0; j < N_BC; j++)
119  {
120  fvec(j) = x(j) - BC(j);
121  }
122  }
123 
124  return 0;
125 }

146 void reducedSteadyNS::solveOnline_sup(Eigen::MatrixXd vel)
147 {
148  if (problem->bcMethod == "lift")
149  {
150  vel_now = setOnlineVelocity(vel);
151  }
152  else if (problem->bcMethod == "penalty")
153  {
154  vel_now = vel;
155  }
156 
157  y.resize(Nphi_u + Nphi_p, 1);
158  y.setZero();
159 
160  // Change initial condition for the lifting function
161  if (problem->bcMethod == "lift")
162  {
163  for (int j = 0; j < N_BC; j++)
164  {
165  y(j) = vel_now(j, 0);
166  }
167  }
168 
169  Color::Modifier red(Color::FG_RED);
170  Color::Modifier green(Color::FG_GREEN);
171  Color::Modifier def(Color::FG_DEFAULT);
172  Eigen::HybridNonLinearSolver<newton_steadyNS> hnls(newton_object);
173  newton_object.BC.resize(N_BC);
174  newton_object.tauU = tauU;
175 
176  for (int j = 0; j < N_BC; j++)
177  {
178  newton_object.BC(j) = vel_now(j, 0);
179  }
180 
181  newton_object.nu = nu;
182  hnls.solve(y);
183  Eigen::VectorXd res(y);
184  newton_object.operator()(y, res);
185  Info << "################## Online solve N° " << count_online_solve <<
186  " ##################" << endl;
187 
188  if (Pstream::master())
189  {
190  std::cout << "Solving for the parameter: " << vel_now << std::endl;
191  }
192 
193  if (res.norm() < 1e-5 && Pstream::master())
194  {
195  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
196  hnls.iter << " iterations " << def << std::endl << std::endl;
197  }
198  else if (Pstream::master())
199  {
200  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
201  hnls.iter << " iterations " << def << std::endl << std::endl;
202  }
203 
204  count_online_solve += 1;
205 }

290 double reducedSteadyNS::inf_sup_constant()
291 {
292  double a;
293  Eigen::VectorXd sup(Nphi_u);
294  Eigen::VectorXd inf(Nphi_p);
295 
296  for (int i = 0; i < Nphi_p; i++)
297  {
298  for (int j = 0; j < Nphi_u; j++)
299  {
300  sup(j) = fvc::domainIntegrate(fvc::div(Umodes[j]) * Pmodes[i]).value() /
301  ITHACAutilities::H1Seminorm(Umodes[j]) / ITHACAutilities::L2Norm(Pmodes[i]);
302  }
303 
304  inf(i) = sup.maxCoeff();
305  }
306 
307  a = inf.minCoeff();
308  return a;

374 Eigen::MatrixXd reducedSteadyNS::setOnlineVelocity(Eigen::MatrixXd vel)
375 {
376  assert(problem->inletIndex.rows() == vel.rows()
377  && "Imposed boundary conditions dimensions do not match given values matrix dimensions");
378  Eigen::MatrixXd vel_scal;
379  vel_scal.resize(vel.rows(), vel.cols());
380 
381  for (int k = 0; k < problem->inletIndex.rows(); k++)
382  {
383  int p = problem->inletIndex(k, 0);
384  int l = problem->inletIndex(k, 1);
385  scalar area = gSum(problem->liftfield[0].mesh().magSf().boundaryField()[p]);
386  scalar u_lf = gSum(problem->liftfield[k].mesh().magSf().boundaryField()[p] *
387  problem->liftfield[k].boundaryField()[p]).component(l) / area;
388  vel_scal(k, 0) = vel(k, 0) / u_lf;
389  }
390 
391  return vel_scal;
392 }
