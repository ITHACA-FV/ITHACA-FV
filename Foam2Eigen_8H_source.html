262 {
263  List<Eigen::SparseMatrix<double> > SM_list;
264  List<Eigen::VectorXd> V_list;
265  int LSize = MatrixList.size();
266  SM_list.resize(LSize);
267  V_list.resize(LSize);
268  Eigen::SparseMatrix<double> A;
269  Eigen::VectorXd b;
270  for (int j = 0; j < LSize; j++)
271  {
272  fvMatrix2Eigen(MatrixList[j], A, b);
273  SM_list[j] = A;

275  }
276  std::tuple <List<Eigen::SparseMatrix<double> >, List<Eigen::VectorXd> > tupla;
277  tupla = std::make_tuple(SM_list, V_list);
278  return tupla;

281 template <class type_m, class type_PtrList>
282 std::tuple<Eigen::MatrixXd, Eigen::VectorXd> Foam2Eigen::projectFvMatrix(type_m& matrix, type_PtrList& modes, int Nmodes)
283 {
284  Eigen::SparseMatrix<double> A;
285  Eigen::MatrixXd Ar;
286  Eigen::VectorXd b;
287  Eigen::VectorXd br;
288  Eigen::MatrixXd Eig_Modes = PtrList2Eigen(modes, Nmodes);
289  Foam2Eigen::fvMatrix2Eigen(matrix, A, b);
290  Eigen::VectorXd Volumes = field2Eigen(modes[0].mesh());
291  Eigen::MatrixXd VolumesN(Volumes.rows(), Nmodes);
292  if (Volumes.rows() != Eig_Modes.rows())
293  {
294  VolumesN.resize(Eig_Modes.rows(), Nmodes);
295  }
296  if (Volumes.rows() == Eig_Modes.rows())
297  {
298  for (auto i = 0; i < Nmodes; i++)
299  {
300  VolumesN.col(i) = Volumes;
301  }
302  }
303  else
304  {
305  for (auto i = 0; i < Nmodes; i++)
306  {
307  VolumesN.col(i).segment(0, Volumes.rows()) = Volumes;
308  VolumesN.col(i).segment(Volumes.rows() + 1, Volumes.rows() * 2) = Volumes;
309  VolumesN.col(i).segment(Volumes.rows() * 2 + 1, Volumes.rows() * 3) = Volumes;
310  }
311  }
312  Ar = Eig_Modes.transpose() * A * Eig_Modes;
313  br = Eig_Modes.transpose() * b;
314  std::tuple <Eigen::MatrixXd, Eigen::VectorXd> tupla;
315  tupla = std::make_tuple(Ar, br);
316  return tupla;
317 }

352 Eigen::VectorXd Foam2Eigen::projectField(type_f& field, type_PtrList& modes, int Nmodes)
353 {
354  Eigen::VectorXd fr;
355  Eigen::MatrixXd Eig_Modes = PtrList2Eigen(modes, Nmodes);
356  Eigen::VectorXd f = Foam2Eigen::field2Eigen(field);
357  Eigen::VectorXd Volumes = field2Eigen(modes[0].mesh());
358  Eigen::MatrixXd VolumesN(Volumes.rows(), 1);
359  VolumesN = Volumes;
360  if (Volumes.rows() != Eig_Modes.rows())
361  {
362  VolumesN.resize(Eig_Modes.rows(), 1);
363  VolumesN.col(0).segment(0, Volumes.rows()) = Volumes;
364  VolumesN.col(0).segment(Volumes.rows() + 1, Volumes.rows() * 2) = Volumes;
365  VolumesN.col(0).segment(Volumes.rows() * 2 + 1, Volumes.rows() * 3) = Volumes;
366  }
367  fr = Eig_Modes.transpose() * (f.cwiseProduct(VolumesN));
