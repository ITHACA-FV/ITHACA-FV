69 DenseMatrix Function::centralDifference(DenseVector x) const
70 {
71  DenseMatrix dx(1, x.size());
72 
73  double h = 1e-6; // perturbation step size
74  double hForward = 0.5*h;
75  double hBackward = 0.5*h;
76 
77  for (unsigned int i = 0; i < getNumVariables(); ++i)
78  {
79  DenseVector xForward(x);
80  xForward(i) = xForward(i) + hForward;
81 
82  DenseVector xBackward(x);
83  xBackward(i) = xBackward(i) - hBackward;
84 
85  double yForward = eval(xForward);
86  double yBackward = eval(xBackward);
87 
88  dx(i) = (yForward - yBackward)/(hBackward + hForward);
89  }

92 }
