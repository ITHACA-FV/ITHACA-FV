87 int main(int argc, char* argv[])
88 {
89  // Construct the tutorial object
90  tutorial03 example(argc, argv);
91  // Read some parameters from file
92  ITHACAparameters* para = ITHACAparameters::getInstance(example._mesh(),
93  example._runTime());
94  int NmodesUout = para->ITHACAdict->lookupOrDefault<int>("NmodesUout", 15);
95  int NmodesPout = para->ITHACAdict->lookupOrDefault<int>("NmodesPout", 15);
96  int NmodesSUPout = para->ITHACAdict->lookupOrDefault<int>("NmodesSUPout", 15);
97  int NmodesUproj = para->ITHACAdict->lookupOrDefault<int>("NmodesUproj", 10);
98  int NmodesPproj = para->ITHACAdict->lookupOrDefault<int>("NmodesPproj", 10);
99  int NmodesSUPproj = para->ITHACAdict->lookupOrDefault<int>("NmodesSUPproj", 10);
100  // Read the par file where the parameters are stored
101  word filename("./par");
102  example.mu = ITHACAstream::readMatrix(filename);
103  // Set the inlet boundaries patch 0 directions x and y
104  example.inletIndex.resize(1, 2);
105  example.inletIndex(0, 0) = 0;
106  example.inletIndex(0, 1) = 0;
107  // Perform the offline solve
108  example.offlineSolve();
109  // Solve the supremizer problem
110  example.solvesupremizer();
111 
112  if (example.bcMethod == "lift")
113  {
114  ITHACAstream::read_fields(example.liftfield, example.U, "./lift/");
115  ITHACAutilities::normalizeFields(example.liftfield);
116  // Homogenize the snapshots
117  example.computeLift(example.Ufield, example.liftfield, example.Uomfield);
118  // Perform POD on the velocity snapshots
119  ITHACAPOD::getModes(example.Uomfield, example.Umodes, example._U().name(),
120  example.podex, 0, 0, NmodesUout);
121  }
122  else
123  {
124  ITHACAPOD::getModes(example.Ufield, example.Umodes, example._U().name(),
125  example.podex, 0, 0, NmodesUout);
126  }
127 
128  // Perform POD on velocity pressure and supremizers and store the first 10 modes
129  ITHACAPOD::getModes(example.Pfield, example.Pmodes, example._p().name(),
130  example.podex, 0, 0,
131  NmodesPout);
132  ITHACAPOD::getModes(example.supfield, example.supmodes, example._U().name(),
133  example.podex,
134  example.supex, 1, NmodesSUPout);
135  // Perform the Galerkin Projection
136  example.projectSUP("./Matrices", NmodesUproj, NmodesPproj, NmodesSUPproj);
137  // Create the reduced object
138  reducedSteadyNS ridotto(example);
139  // Set the inlet velocity
140  Eigen::MatrixXd vel_now(1, 1);
141  vel_now(0, 0) = 1;
142  // used only for penalty approach
143  ridotto.tauU = Eigen::MatrixXd::Zero(1, 1);
144  ridotto.tauU(0, 0) = 1e-1;
145 
146  // Perform an online solve for the new values of inlet velocities
147  for (label k = 0; k < example.mu.size(); k++)
148  {
149  // Set the reduced viscosity
150  ridotto.nu = example.mu(0, k);
151  ridotto.solveOnline_sup(vel_now);
152  Eigen::MatrixXd tmp_sol(ridotto.y.rows() + 1, 1);
153  tmp_sol(0) = k + 1;
154  tmp_sol.col(0).tail(ridotto.y.rows()) = ridotto.y;
155  ridotto.online_solution.append(tmp_sol);
156  }
157 
158  // Save the online solution
159  ITHACAstream::exportMatrix(ridotto.online_solution, "red_coeff", "python",
160  "./ITHACAoutput/red_coeff");
161  ITHACAstream::exportMatrix(ridotto.online_solution, "red_coeff", "matlab",
162  "./ITHACAoutput/red_coeff");
163  ITHACAstream::exportMatrix(ridotto.online_solution, "red_coeff", "eigen",
164  "./ITHACAoutput/red_coeff");
165  // Reconstruct and export the solution
166  ridotto.reconstruct(true, "./ITHACAoutput/Reconstruction/");
167  return 0;
168 }
