49 template<class FieldType>
50 void ITHACADMD<FieldType>::getModes(int SVD_rank, bool exact,
51  bool exportDMDmodes)
52 {
53  // Check the rank if rank < 0, full rank.
54  if (SVD_rank < 0)
55  {
56  SVD_rank = NSnaps - 1;
57  }
58 
59  std::string assertMessage = "The SVD_rank is equal to: " + name(
60  SVD_rank) +
61  ", it cannot be bigger than the number of snapshots - 1. NSnapshots is equal to: "
62  + name(NSnaps);
63  SVD_rank_public = SVD_rank;
64  M_Assert(SVD_rank < NSnaps, assertMessage.c_str());
65  // Convert the OpenFoam Snapshots to Matrix
66  Eigen::MatrixXd SnapEigen = Foam2Eigen::PtrList2Eigen(snapshotsDMD);
67  List<Eigen::MatrixXd> SnapEigenBC = Foam2Eigen::PtrList2EigenBC(snapshotsDMD);
68  Eigen::MatrixXd Xm = SnapEigen.leftCols(NSnaps - 1);
69  Eigen::MatrixXd Ym = SnapEigen.rightCols(NSnaps - 1);
70  List<Eigen::MatrixXd> XmBC(SnapEigenBC.size());
71  List<Eigen::MatrixXd> YmBC(SnapEigenBC.size());
72 
73  for (int i = 0 ; i < SnapEigenBC.size() ; i++)
74  {
75  XmBC[i] = SnapEigenBC[i].leftCols(NSnaps - 1);
76  YmBC[i] = SnapEigenBC[i].rightCols(NSnaps - 1);
77  }
78 
79  Eigen::MatrixXcd U;
80  Eigen::MatrixXcd V;
81  Eigen::VectorXd S;
82 
83  if (redSVD)
84  {
85  Info << "SVD using Randomized method" << endl;
86  RedSVD::RedSVD<Eigen::MatrixXd> svd(Xm, SVD_rank);
87  U = svd.matrixU();
88  V = svd.matrixV();
89  S = svd.singularValues().array().cwiseInverse();
90  }
91  else
92  {
93  Info << "SVD using Divide and Conquer method" << endl;
94  Eigen::BDCSVD<Eigen::MatrixXd> svd(Xm,
95  Eigen::ComputeThinU | Eigen::ComputeThinV);
96  U = svd.matrixU().leftCols(SVD_rank);
97  V = svd.matrixV().leftCols(SVD_rank);
98  S = svd.singularValues().head(SVD_rank).array().cwiseInverse();
99  }
100 
101  Eigen::MatrixXcd A_tilde = U.transpose().conjugate() * Ym *
102  V *
103  S.asDiagonal();
104  Eigen::ComplexEigenSolver<Eigen::MatrixXcd> esEg(A_tilde);
105  eigenValues = esEg.eigenvalues();
106  Eigen::VectorXd ln = eigenValues.array().log().imag().abs();
107  // Sort based on The Frequencies
108  typedef std::pair<double, int> mypair;
109  std::vector<mypair> sortedList(ln.size());
110 
111  for (int i = 0; i < ln.size(); i++)
112  {
113  sortedList[i].first = ln(i);
114  sortedList[i].second = i;
115  }
116 
117  std::sort(sortedList.begin(), sortedList.end(),
118  std::less<std::pair<double, int>>());
119 
120  if (exact)
121  {
122  DMDEigenModes = Ym * V * S.asDiagonal() * esEg.eigenvectors();
123  DMDEigenModesBC.resize(YmBC.size());
124 
125  for (int i = 0; i < YmBC.size(); i++)
126  {
127  DMDEigenModesBC[i] = YmBC[i] * V * S.asDiagonal() * esEg.eigenvectors();
128  }
129  }
130  else
131  {
132  Eigen::VectorXd eigenValueseigLam =
133  S.array().sqrt();
134  PODm = (Xm * V) * eigenValueseigLam.asDiagonal();
135  PODmBC.resize(XmBC.size());
136 
137  for (int i = 0; i < XmBC.size(); i++)
138  {
139  PODmBC[i] = (XmBC[i] * V) * eigenValueseigLam.asDiagonal();
140  }
141 
142  DMDEigenModes = PODm * esEg.eigenvectors();
143  DMDEigenModesBC.resize(PODmBC.size());
144 
145  for (int i = 0; i < PODmBC.size(); i++)
146  {
147  DMDEigenModesBC[i] = PODmBC[i] * esEg.eigenvectors();
148  }
149  }
150 
151  Amplitudes = DMDEigenModes.real().bdcSvd(Eigen::ComputeThinU |
152  Eigen::ComputeThinV).solve(Xm.col(0));
153 
154  if (exportDMDmodes)
155  {
156  convert2Foam();
157  Info << "exporting the DMDmodes for " << snapshotsDMD[0].name() << endl;
158  ITHACAstream::exportFields(DMDmodesReal.toPtrList(), "ITHACAoutput/DMD/",
159  snapshotsDMD[0].name() + "_Modes_" + name(SVD_rank) + "_Real");
160  ITHACAstream::exportFields(DMDmodesImag.toPtrList(), "ITHACAoutput/DMD/",
161  snapshotsDMD[0].name() + "_Modes_" + name(SVD_rank) + "_Imag");
162  }
163 }

165 template<class FieldType>
166 void ITHACADMD<FieldType>::getDynamics(double tStart, double tFinal, double dt)
167 {
168  Eigen::VectorXcd omega = eigenValues.array().log() / originalDT;
169  int ncols = static_cast<int>((tFinal - tStart) / dt ) + 1;
170  dynamics.resize(SVD_rank_public, ncols);
171  int i = 0;
172 
173  for (double t = tStart; t <= tFinal; t += dt)
174  {
175  Eigen::VectorXcd coli = (omega * t).array().exp() * Amplitudes.array();
176  dynamics.col(i) = coli;
177  i++;
178  }
179 }

193 template<class FieldType>
194 void ITHACADMD<FieldType>::reconstruct(word exportFolder, word fieldName)
195 {
196  PtrList<GeometricField<FieldType, fvPatchField, volMesh>> snapshotsrec;
197 
198  for (int i = 0; i < dynamics.cols(); i++)
199  {
200  Eigen::MatrixXcd col = DMDEigenModes * dynamics.col(i);
201  Eigen::VectorXd vec = col.real();
202  GeometricField<FieldType, fvPatchField, volMesh> tmp("TMP",
203  snapshotsDMD[0] * 0);
204  tmp = Foam2Eigen::Eigen2field(tmp, vec);
205 
206  for (int k = 0; k < tmp.boundaryField().size(); k++)
207  {
208  Eigen::VectorXd vecBC = (DMDEigenModesBC[k] * dynamics.col(i)).real();
209  ITHACAutilities::assignBC(tmp, k, vecBC);
210  }
211 
212  snapshotsrec.append(tmp);
213  }
214 
215  ITHACAstream::exportFields(snapshotsrec, exportFolder, fieldName);
216 }
217 template<class FieldType>
218 void ITHACADMD<FieldType>::convert2Foam()
219 {
220  DMDmodesReal.resize(DMDEigenModes.cols());
221  DMDmodesImag.resize(DMDEigenModes.cols());
222  GeometricField<FieldType, fvPatchField, volMesh> tmpReal(
223  snapshotsDMD[0].name(), snapshotsDMD[0] * 0);
224  GeometricField<FieldType, fvPatchField, volMesh> tmpImag(
225  snapshotsDMD[0].name(), snapshotsDMD[0] * 0);
226 
227  for (label i = 0; i < DMDmodesReal.size(); i++)
228  {
229  Eigen::VectorXd vecReal = DMDEigenModes.col(i).real();
230  Eigen::VectorXd vecImag = DMDEigenModes.col(i).imag();
231  tmpReal = Foam2Eigen::Eigen2field(tmpReal, vecReal);
232  tmpImag = Foam2Eigen::Eigen2field(tmpImag, vecImag);
233 
234  // Adjusting boundary conditions
235  for (int k = 0; k < tmpReal.boundaryField().size(); k++)
236  {
237  ITHACAutilities::assignBC(tmpReal, k, DMDEigenModesBC[k].col(i).real());
238  ITHACAutilities::assignBC(tmpImag, k, DMDEigenModesBC[k].col(i).imag());
239  }
240 
241  DMDmodesReal.set(i, tmpReal);
242  DMDmodesImag.set(i, tmpImag);
243  }
244 }
