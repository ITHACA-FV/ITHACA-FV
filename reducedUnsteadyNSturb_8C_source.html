88 // Operator to evaluate the residual for the supremizer approach
89 int newton_unsteadyNSturb_sup::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
90 {
91  Eigen::VectorXd a_dot(Nphi_u);
92  Eigen::VectorXd a_tmp(Nphi_u);
93  Eigen::VectorXd b_tmp(Nphi_p);
94  a_tmp = x.head(Nphi_u);
95  b_tmp = x.tail(Nphi_p);
96  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
97 
98  // Convective term
99  Eigen::MatrixXd cc(1, 1);
100  // Mom Term
101  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
102  // Gradient of pressure
103  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
104  // Mass Term
105  Eigen::VectorXd M5 = problem->M_matrix * a_dot;
106  // Pressure Term
107  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
108 
109  for (label i = 0; i < Nphi_u; i++)
110  {
111  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp - nu_c.transpose() * problem->C_total_matrix[i] * a_tmp;
112  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M2(i);
113  }
114  for (label j = 0; j < Nphi_p; j++)
115  {
116  label k = j + Nphi_u;
117  fvec(k) = M3(j);
118 
119  }
120  for (label j = 0; j < N_BC; j++)
121  {
122  fvec(j) = x(j) - BC(j);
123  }
124  return 0;
125 }

138 // Operator to evaluate the residual for the supremizer approach
139 int newton_unsteadyNSturb_PPE::operator()(const Eigen::VectorXd &x, Eigen::VectorXd &fvec) const
140 {
141  Eigen::VectorXd a_dot(Nphi_u);
142  Eigen::VectorXd a_tmp(Nphi_u);
143  Eigen::VectorXd b_tmp(Nphi_p);
144  a_tmp = x.head(Nphi_u);
145  b_tmp = x.tail(Nphi_p);
146  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
147 
148  // Convective terms
149  Eigen::MatrixXd cc(1, 1);
150  Eigen::MatrixXd gg(1, 1);
151  Eigen::MatrixXd bb(1, 1);
152  // Mom Term
153  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
154  // Gradient of pressure
155  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
156  // Mass Term
157  Eigen::VectorXd M5 = problem->M_matrix * a_dot;
158  // Pressure Term
159  Eigen::VectorXd M3 = problem->D_matrix * b_tmp;
160 
161  // BC PPE
162  Eigen::VectorXd M6 = problem->BC1_matrix * a_tmp * nu;
163 
164  // BC PPE
165  Eigen::VectorXd M7 = problem->BC3_matrix * a_tmp * nu;
166 
167 
168 
169  for (label i = 0; i < Nphi_u; i++)
170  {
171  cc = a_tmp.transpose() * problem->C_matrix[i] * a_tmp - nu_c.transpose() * problem->C_total_matrix[i] * a_tmp;
172  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M2(i);
173  }
174  for (label j = 0; j < Nphi_p; j++)
175  {
176  label k = j + Nphi_u;
177  gg = a_tmp.transpose() * problem->G_matrix[j] * a_tmp;
178  bb = a_tmp.transpose() * problem->BC2_matrix[j] * a_tmp;
179  //fvec(k) = M3(j, 0) - gg(0, 0) - M6(j, 0) + bb(0, 0);
180  fvec(k) = M3(j, 0) + gg(0, 0) - M7(j, 0);
181  }
182  for (label j = 0; j < N_BC; j++)
183  {
184  fvec(j) = x(j) - BC(j);
185  }
186  return 0;
187 }

205  y.setZero();

207

209  y.tail(Nphi_p) = ITHACAutilities::get_coeffs_ortho(Psnapshots[startSnap], Pmodes);
210 
211 
212  // Change initial condition for the lifting function
213  for (label j = 0; j < N_BC; j++)
214  {
215  y(j) = vel_now(j, 0);
216  }
217 
218  // Set some properties of the newton object
219  newton_object_sup.nu = nu;
220  newton_object_sup.y_old = y;
221  newton_object_sup.dt = dt;
222  newton_object_sup.BC.resize(N_BC);

225  {
226  newton_object_sup.BC(j) = vel_now(j, 0);
227  }
228 
229

231  int Ntsteps = static_cast<int>((finalTime - tstart) / dt);
232  online_solution.resize(Ntsteps);
233

235  time = tstart;
236 
237  // Counting variable
238  int counter = 0;
239 
240  // Create vector to store temporal solution and save initial condition as first solution
241  Eigen::MatrixXd tmp_sol(Nphi_u + Nphi_p + 1, 1);
242  tmp_sol(0) = time;
243  tmp_sol.col(0).tail(y.rows()) = y;
244  if(time!=0)
245  {
246  online_solution[counter] = tmp_sol;
247  counter ++;
248  }

251 
252  // Set output colors for fancy output
253  Color::Modifier red(Color::FG_RED);
254  Color::Modifier green(Color::FG_GREEN);
255  Color::Modifier def(Color::FG_DEFAULT);
256 
257 
258  // This part up to the while loop is just to compute the eddy viscosity field at time = startTime if time is not equal to zero
259  if(time!=0)
260  {
261 
262  volScalarField nut_rec("nut_rec", problem->nuTmodes[0] * 0);
263  Eigen::VectorXd nu_c0;
264  nu_c0.resize(Nphi_nut);
265  std::vector<double> tv0;
266  tv0.resize(1);
267  tv0[0]=time;
268 
269  for(label i=0;i< Nphi_nut;i++)
270  {
271 
272  nu_c0(i)= problem->rbfsplines[i]->eval(tv0);
273  }
274 
275  for (label j = 0; j < Nphi_nut; j++)
276  {
277 
278  nut_rec += problem->nuTmodes[j] * nu_c0(j);
279  }
280  nutREC.append(nut_rec);
281  }
282 
283  // Start the time loop
284  while (time < finalTime)
285  {
286  time = time + dt;
287 
288  std::vector<double> tv;
289  tv.resize(1);
290  tv[0]=time;
291 
292  for(label i=0;i< Nphi_nut;i++)
293  {
294  newton_object_sup.nu_c(i)= problem->rbfsplines[i]->eval(tv);
295  }
296 
297  volScalarField nut_rec("nut_rec", problem->nuTmodes[0] * 0);
298  for (label j = 0; j < Nphi_nut; j++)
299  {
300  nut_rec += problem->nuTmodes[j] * newton_object_sup.nu_c(j);
301  }
302 
303  nutREC.append(nut_rec);
304 
305 
306  Eigen::VectorXd res(y);
307  res.setZero();
308  hnls.solve(y);
309 
310  for (label j = 0; j < N_BC; j++)
311  {
312  y(j) = vel_now(j, 0);
313  }
314 
315 
316 
317  newton_object_sup.operator()(y, res);
318  newton_object_sup.y_old = y;
319 
320 
321 
322  std::cout << "################## Online solve N° " << count_online_solve << " ##################" << std::endl;
323  Info << "Time = " << time << endl;
324  std::cout << "Solving for the parameter: " << vel_now << std::endl;
325  if (res.norm() < 1e-5)
326  {
327  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
328  }
329  else
330  {
331  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " << hnls.iter << " iterations " << def << std::endl << std::endl;
332  }
333  count_online_solve += 1;
334  tmp_sol(0) = time;
335  tmp_sol.col(0).tail(y.rows()) = y;
336  if (counter >= online_solution.size())
337  {
338  online_solution.append(tmp_sol);
339  }
340  else
341  {
342  online_solution[counter] = tmp_sol;
343  }
344  counter ++;
345  }
346  // Save the solution
347  ITHACAstream::exportMatrix(online_solution, "red_coeff", "python", "./ITHACAoutput/red_coeff");
348  ITHACAstream::exportMatrix(online_solution, "red_coeff", "matlab", "./ITHACAoutput/red_coeff");
349  count_online_solve += 1;
