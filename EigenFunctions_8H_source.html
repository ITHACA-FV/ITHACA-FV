227 template <typename T>
228 Eigen::SparseMatrix<T> EigenFunctions::vectorize(Eigen::SparseMatrix<T>& Matrix)
229 {
230  typedef Eigen::Triplet<double> Trip;
231  std::vector<Trip> tripletList;
232  tripletList.reserve(Matrix.nonZeros());
233  Eigen::SparseMatrix<T> vec;
234  vec.resize(1, Matrix.size());
235  vec.reserve(Matrix.nonZeros());
236 
237  for (int k = 0; k < Matrix.outerSize(); ++k)
238  {
239  for (Eigen::SparseMatrix<double>::InnerIterator it(Matrix, k); it; ++it)
240  {
241  tripletList.push_back(Trip(0, it.row() + it.col()*Matrix.rows(), it.value()));
242  }
243  }
244 
245  vec.setFromTriplets(tripletList.begin(), tripletList.end());
246 }
