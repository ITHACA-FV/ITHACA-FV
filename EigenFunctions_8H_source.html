226 template <typename T>
227 Eigen::SparseMatrix<T> EigenFunctions::vectorize(Eigen::SparseMatrix<T>& Matrix)
228 {
229  typedef Eigen::Triplet<double> Trip;
230  std::vector<Trip> tripletList;
231  tripletList.reserve(Matrix.nonZeros());
232  Eigen::SparseMatrix<T> vec;
233  vec.resize(1, Matrix.size());
234  vec.reserve(Matrix.nonZeros());
235 
236  for (int k = 0; k < Matrix.outerSize(); ++k)
237  {
238  for (Eigen::SparseMatrix<double>::InnerIterator it(Matrix, k); it; ++it)
239  {
240  tripletList.push_back(Trip(0, it.row() + it.col()*Matrix.rows(), it.value()));
241  }
242  }
243 
244  vec.setFromTriplets(tripletList.begin(), tripletList.end());
245 }
