228 template <typename T>
229 Eigen::SparseMatrix<T> EigenFunctions::vectorize(Eigen::SparseMatrix<T>& Matrix)
230 {
231  typedef Eigen::Triplet<double> Trip;
232  std::vector<Trip> tripletList;
233  tripletList.reserve(Matrix.nonZeros());
234  Eigen::SparseMatrix<T> vec;
235  vec.resize(1, Matrix.size());
236  vec.reserve(Matrix.nonZeros());
237 
238  for (int k = 0; k < Matrix.outerSize(); ++k)
239  {
240  for (Eigen::SparseMatrix<double>::InnerIterator it(Matrix, k); it; ++it)
241  {
242  tripletList.push_back(Trip(0, it.row() + it.col()*Matrix.rows(), it.value()));
243  }
244  }
245  vec.setFromTriplets(tripletList.begin(), tripletList.end());
246 }
