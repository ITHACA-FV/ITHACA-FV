426 template<>
427 void Foam2Eigen::fvMatrix2Eigen(fvMatrix<scalar> foam_matrix,
428  Eigen::MatrixXd& A,
429  Eigen::VectorXd& b)
430 {
431  label sizeA = foam_matrix.diag().size();
432  A.setZero(sizeA, sizeA);
433  b.setZero(sizeA);
434 
435  for (auto i = 0; i < sizeA; i++)
436  {
437  A(i, i) = foam_matrix.diag()[i];
438  b(i, 0) = foam_matrix.source()[i];
439  }
440 
441  const lduAddressing& addr = foam_matrix.lduAddr();
442  const labelList& lowerAddr = addr.lowerAddr();
443  const labelList& upperAddr = addr.upperAddr();
444  forAll(lowerAddr, i)
445  {
446  A(lowerAddr[i], upperAddr[i]) = foam_matrix.upper()[i];
447  A(upperAddr[i], lowerAddr[i]) = foam_matrix.lower()[i];
448  }
449  forAll(foam_matrix.psi().boundaryField(), I)
450  {
451  const fvPatch& ptch = foam_matrix.psi().boundaryField()[I].patch();
452  forAll(ptch, J)
453  {
454  label w = ptch.faceCells()[J];
455  const double intern = foam_matrix.internalCoeffs()[I][J];
456  A(w, w) += intern;
457  b(w, 0) += foam_matrix.boundaryCoeffs()[I][J];
458  }
459  }
460 }
