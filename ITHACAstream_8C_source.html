352 {

363  file_count++;

367  closedir (dirp);

371  word matname = folder + "/" + mat_name + name(j) + "_mat.txt";
372  Eigen::MatrixXd temp = readMatrix(matname);
373  result.append(temp);
374  }
375 
376  return result;

379 Eigen::MatrixXd readMatrix(word filename)
380 {
381  int cols = 0, rows = 0;
382  double buff[MAXBUFSIZE];
383  // Read numbers from file into buffer.
384  std::ifstream infile;
385  infile.open(filename.c_str());
386  std::string message = "The matrix file \"" + filename +
387  "\" does not exist. Check the existence of the file or the way it is named.";
388  M_Assert(infile.good() != 0, message.c_str()
389  );
390 
391  while (! infile.eof())
392  {
393  string line;
394  getline(infile, line);
395  int temp_cols = 0;
396  std::stringstream stream(line);
397 
398  while (! stream.eof())
399  {
400  stream >> buff[cols * rows + temp_cols++];
401  }
402 
403  if (temp_cols == 0)
404  {
405  continue;
406  }
407 
408  if (cols == 0)
409  {
410  cols = temp_cols;
411  }
412 
413  rows++;
414  }
415 
416  infile.close();
417  rows--;
418  // Populate matrix with numbers.
419  Eigen::MatrixXd result(rows, cols);
420 
421  for (int i = 0; i < rows; i++)
422  {
423  for (int j = 0; j < cols; j++)
424  {
425  result(i, j) = buff[ cols * i + j ];
426  }
427  }
428 
429  return result;
430 }
