349 Eigen::MatrixXd ITHACAstream::readMatrix(word filename)
350 {
351  int cols = 0, rows = 0;
352  double buff[MAXBUFSIZE];
353  // Read numbers from file into buffer.
354  std::ifstream infile;
355  infile.open(filename.c_str());
356  std::string message = "The matrix file \"" + filename +
357  "\" does not exist. Check the existence of the file or the way it is named.";
358  M_Assert(infile.good() != 0, message.c_str()
359  );
360 
361  while (! infile.eof())
362  {
363  string line;
364  getline(infile, line);
365  int temp_cols = 0;
366  std::stringstream stream(line);
367 
368  while (! stream.eof())
369  {
370  stream >> buff[cols * rows + temp_cols++];
371  }
372 
373  if (temp_cols == 0)
374  {
375  continue;
376  }
377 
378  if (cols == 0)
379  {
380  cols = temp_cols;
381  }
382 
383  rows++;
384  }
385 
386  infile.close();
387  rows--;
388  // Populate matrix with numbers.
389  Eigen::MatrixXd result(rows, cols);
390 
391  for (int i = 0; i < rows; i++)
392  {
393  for (int j = 0; j < cols; j++)
394  {
395  result(i, j) = buff[ cols * i + j ];
396  }
397  }
398 
399  return result;
400 }
